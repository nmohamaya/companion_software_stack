// config/payload_config.h
#pragma once
#include <string>
#include <vector>
#include <nlohmann/json.hpp>

struct PayloadConfig {
    // Gimbal
    bool gimbal_enabled         = true;
    int  gimbal_pwm_pitch       = 0;
    int  gimbal_pwm_roll        = 1;
    int  gimbal_pwm_yaw         = 2;
    int  gimbal_imu_bus         = 1;  // I2C bus
    int  gimbal_imu_addr        = 0x68;
    double gimbal_rate_hz       = 200.0;

    // Camera
    bool camera_enabled         = true;
    int  camera_trigger_gpio    = 12;
    double default_interval_s   = 2.0;
    std::string geotag_dir      = "/opt/drone/data/geotags";

    // Aux payloads
    bool spotlight_enabled      = false;
    int  spotlight_pwm_channel  = 3;
    bool drop_enabled           = false;
    int  drop_gpio_pin          = 17;
    double drop_hold_time_s     = 2.0;

    // General
    std::string log_level       = "info";
};
