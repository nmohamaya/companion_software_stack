=== Listing 9-1: Command Dispatcher with Plugin Routing ===

// dispatch/command_dispatcher.h
#pragma once
#include "plugin/payload_plugin.h"
#include "ipc/payload_command_reader.h"
#include <vector>
#include <memory>
#include <unordered_map>

class CommandDispatcher {
public:
    void register_plugin(PayloadPlugin* plugin,
                          const std::vector<PayloadCmdType>& handled_cmds) {
        for (auto cmd : handled_cmds) {
            cmd_routes_[cmd] = plugin;
        }
        plugins_.push_back(plugin);
    }

    void dispatch(const PayloadCmd& cmd) {
        // Skip NONE commands
        if (cmd.type == PayloadCmdType::NONE) return;

        auto it = cmd_routes_.find(cmd.type);
        if (it != cmd_routes_.end()) {
            auto* plugin = it->second;
            if (plugin->state() == PluginState::READY ||
                plugin->state() == PluginState::ACTIVE) {
                if (!plugin->handle_command(cmd)) {
                    spdlog::warn("Plugin [{}] rejected command type {}",
                                  plugin->name(), static_cast<int>(cmd.type));
                }
            } else {
                spdlog::warn("Plugin [{}] not ready (state={}), dropping cmd",
                              plugin->name(), static_cast<int>(plugin->state()));
            }
        } else {
            spdlog::debug("No handler for command type {}",
                           static_cast<int>(cmd.type));
        }
    }

    const std::vector<PayloadPlugin*>& plugins() const { return plugins_; }

private:
    std::unordered_map<PayloadCmdType, PayloadPlugin*> cmd_routes_;
    std::vector<PayloadPlugin*> plugins_;
};
