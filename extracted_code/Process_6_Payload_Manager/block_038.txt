# CMakeLists.txt — Process 6: Payload Manager
cmake_minimum_required(VERSION 3.18)
project(payload_manager VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# ── Compiler flags ──────────────────────────────────────────────
add_compile_options(
    -Wall -Wextra -Wpedantic -Werror
    -O2
    -march=armv8.2-a+fp16
)

# ── Dependencies ────────────────────────────────────────────────
find_package(Eigen3 3.4 REQUIRED NO_MODULE)
find_package(spdlog REQUIRED)
find_package(nlohmann_json 3.11 REQUIRED)
find_package(Threads REQUIRED)

# ── Sources ─────────────────────────────────────────────────────
set(PAYLOAD_SOURCES
    src/main.cpp
    src/plugin/plugin_registry.cpp
    src/gimbal/gimbal_plugin.cpp
    src/gimbal/gimbal_thread.cpp
    src/gimbal/pid_controller.cpp
    src/camera/camera_plugin.cpp
    src/camera/camera_thread.cpp
    src/camera/geo_tag.cpp
    src/aux/spotlight_plugin.cpp
    src/aux/drop_plugin.cpp
    src/aux/aux_sensor_thread.cpp
    src/dispatch/command_dispatcher.cpp
    src/dispatch/dispatcher_thread.cpp
    src/hardware/gpio_output.cpp
    src/hardware/pwm_output.cpp
    src/hardware/i2c_imu.cpp
    src/ipc/pose_reader.cpp
    src/ipc/payload_command_reader.cpp
    src/ipc/mission_status_reader.cpp
    src/ipc/payload_status_writer.cpp
    src/config/config_loader.cpp
    src/util/realtime.cpp
)

# ── Executable ──────────────────────────────────────────────────
add_executable(payload_manager ${PAYLOAD_SOURCES})

target_include_directories(payload_manager PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/include
)

target_link_libraries(payload_manager PRIVATE
    Eigen3::Eigen
    spdlog::spdlog
    nlohmann_json::nlohmann_json
    Threads::Threads
    rt          # POSIX shared memory
    pthread     # POSIX threads
)

# ── Install ─────────────────────────────────────────────────────
install(TARGETS payload_manager DESTINATION /opt/drone/bin)
install(FILES config/payload_config.json DESTINATION /opt/drone/etc)

# ── Tests ───────────────────────────────────────────────────────
option(BUILD_TESTS "Build unit tests" ON)
if(BUILD_TESTS)
    enable_testing()
    find_package(GTest REQUIRED)

    add_executable(test_pid
        tests/test_pid.cpp
        src/gimbal/pid_controller.cpp
    )
    target_include_directories(test_pid PRIVATE include)
    target_link_libraries(test_pid GTest::gtest_main Eigen3::Eigen)
    add_test(NAME PIDTest COMMAND test_pid)

    add_executable(test_trigger_modes tests/test_trigger_modes.cpp)
    target_include_directories(test_trigger_modes PRIVATE include)
    target_link_libraries(test_trigger_modes GTest::gtest_main Eigen3::Eigen)
    add_test(NAME TriggerModeTest COMMAND test_trigger_modes)

    add_executable(test_plugin_registry
        tests/test_plugin_registry.cpp
        src/plugin/plugin_registry.cpp
    )
    target_include_directories(test_plugin_registry PRIVATE include)
    target_link_libraries(test_plugin_registry GTest::gtest_main)
    add_test(NAME PluginRegistryTest COMMAND test_plugin_registry)

    add_executable(test_geotag
        tests/test_geotag.cpp
        src/camera/geo_tag.cpp
    )
    target_include_directories(test_geotag PRIVATE include)
    target_link_libraries(test_geotag GTest::gtest_main Eigen3::Eigen)
    add_test(NAME GeoTagTest COMMAND test_geotag)

    add_executable(test_command_dispatcher
        tests/test_command_dispatcher.cpp
        src/dispatch/command_dispatcher.cpp
    )
    target_include_directories(test_command_dispatcher PRIVATE include)
    target_link_libraries(test_command_dispatcher GTest::gtest_main)
    add_test(NAME CommandDispatcherTest COMMAND test_command_dispatcher)
endif()
