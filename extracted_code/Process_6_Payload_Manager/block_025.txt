=== Listing 11-2: PWM Output with RAII (servo control) ===

// hardware/pwm_output.h
#pragma once
#include <string>
#include <fstream>
#include <spdlog/spdlog.h>

class PWMOutput {
public:
    PWMOutput() = default;

    bool init(int channel, int period_ns = 20000000,  // 20 ms (50 Hz servo)
              int min_pulse_ns = 1000000,              // 1 ms
              int max_pulse_ns = 2000000) {            // 2 ms
        channel_ = channel;
        min_pulse_ = min_pulse_ns;
        max_pulse_ = max_pulse_ns;

        std::string base = "/sys/class/pwm/pwmchip0/pwm" + std::to_string(channel);

        // Export PWM channel
        write_file("/sys/class/pwm/pwmchip0/export", std::to_string(channel));

        // Configure period
        if (!write_file(base + "/period", std::to_string(period_ns))) return false;

        // Set initial duty cycle to centre
        int centre = (min_pulse_ns + max_pulse_ns) / 2;
        if (!write_file(base + "/duty_cycle", std::to_string(centre))) return false;

        // Enable
        if (!write_file(base + "/enable", "1")) return false;

        duty_path_ = base + "/duty_cycle";
        duty_fd_.open(duty_path_);
        if (!duty_fd_.is_open()) return false;

        initialised_ = true;
        spdlog::info("PWM channel {} initialised (period={}ns, range={}-{}ns)",
                      channel, period_ns, min_pulse_ns, max_pulse_ns);
        return true;
    }

    // value: -1.0 to +1.0, mapped to min_pulse → max_pulse
    void set_normalised(double value) {
        if (!initialised_) return;
        value = std::clamp(value, -1.0, 1.0);
        double t = (value + 1.0) / 2.0;  // 0.0 → 1.0
        int pulse = min_pulse_ + static_cast<int>(t * (max_pulse_ - min_pulse_));
        duty_fd_.seekp(0);
        duty_fd_ << pulse << std::flush;
    }

    ~PWMOutput() {
        if (initialised_) {
            // Return to centre before disabling
            set_normalised(0.0);
            std::string base = "/sys/class/pwm/pwmchip0/pwm" +
                               std::to_string(channel_);
            duty_fd_.close();
            write_file(base + "/enable", "0");
            write_file("/sys/class/pwm/pwmchip0/unexport",
                        std::to_string(channel_));
            spdlog::debug("PWM channel {} unexported", channel_);
        }
    }

    PWMOutput(const PWMOutput&) = delete;
    PWMOutput& operator=(const PWMOutput&) = delete;
    PWMOutput(PWMOutput&&) noexcept = default;

private:
    int channel_ = -1;
    int min_pulse_ = 1000000;
    int max_pulse_ = 2000000;
    bool initialised_ = false;
    std::string duty_path_;
    std::ofstream duty_fd_;

    static bool write_file(const std::string& path, const std::string& val) {
        std::ofstream f(path);
        if (!f) return false;
        f << val;
        return true;
    }
};
