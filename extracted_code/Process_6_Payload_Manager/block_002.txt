=== Listing 4-1: Payload Command SHM Reader ===

// ipc/payload_command_reader.h
#pragma once
#include <atomic>
#include <cstring>
#include <sys/mman.h>
#include <fcntl.h>
#include <unistd.h>

enum class PayloadCmdType : uint8_t {
    NONE = 0,
    GIMBAL_SET_ANGLE,       // set pitch/roll/yaw target
    GIMBAL_SET_RATE,        // set angular rate
    GIMBAL_SET_MODE,        // stabilised / follow / lock / FPV
    CAMERA_TRIGGER,         // single shot
    CAMERA_START_INTERVAL,  // interval capture (survey)
    CAMERA_STOP_INTERVAL,
    CAMERA_START_VIDEO,
    CAMERA_STOP_VIDEO,
    AUX_SPOTLIGHT_ON,
    AUX_SPOTLIGHT_OFF,
    AUX_SPOTLIGHT_BRIGHTNESS,
    AUX_DROP_RELEASE,
    AUX_SENSOR_START,
    AUX_SENSOR_STOP,
};

struct PayloadCmd {
    PayloadCmdType type;
    float param1;   // e.g., pitch angle, brightness 0-1
    float param2;   // e.g., roll angle, interval_s
    float param3;   // e.g., yaw angle
    uint64_t timestamp_ns;
};

struct ShmPayloadCmd {
    std::atomic<uint64_t> seq;
    PayloadCmd cmd;
};

class PayloadCommandReader {
public:
    bool open(const char* name = "/payload_commands") {
        fd_ = shm_open(name, O_RDONLY, 0);
        if (fd_ < 0) return false;
        ptr_ = static_cast<ShmPayloadCmd*>(
            mmap(nullptr, sizeof(ShmPayloadCmd), PROT_READ, MAP_SHARED, fd_, 0));
        return ptr_ != MAP_FAILED;
    }

    bool read(PayloadCmd& out) const {
        if (!ptr_) return false;
        for (int attempt = 0; attempt < 4; ++attempt) {
            uint64_t s1 = ptr_->seq.load(std::memory_order_acquire);
            if (s1 & 1) continue;
            std::memcpy(&out, &ptr_->cmd, sizeof(PayloadCmd));
            std::atomic_thread_fence(std::memory_order_acquire);
            uint64_t s2 = ptr_->seq.load(std::memory_order_relaxed);
            if (s1 == s2) return true;
        }
        return false;
    }

    ~PayloadCommandReader() {
        if (ptr_ && ptr_ != MAP_FAILED) munmap(ptr_, sizeof(ShmPayloadCmd));
        if (fd_ >= 0) close(fd_);
    }

private:
    int fd_ = -1;
    ShmPayloadCmd* ptr_ = nullptr;
};
