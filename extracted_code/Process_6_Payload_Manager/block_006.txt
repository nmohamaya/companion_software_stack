=== Listing 5-2: Plugin Registry with Self-Registration Macro ===

// plugin/plugin_registry.h
#pragma once
#include "plugin/payload_plugin.h"
#include <functional>
#include <unordered_map>
#include <memory>
#include <string>

class PluginRegistry {
public:
    using FactoryFn = std::function<std::unique_ptr<PayloadPlugin>()>;

    static PluginRegistry& instance() {
        static PluginRegistry reg;
        return reg;
    }

    void register_type(const std::string& type_name, FactoryFn factory) {
        factories_[type_name] = std::move(factory);
        spdlog::debug("Plugin type registered: {}", type_name);
    }

    std::unique_ptr<PayloadPlugin> create(const std::string& type_name) const {
        auto it = factories_.find(type_name);
        if (it == factories_.end()) {
            spdlog::error("Unknown plugin type: {}", type_name);
            return nullptr;
        }
        return it->second();
    }

    std::vector<std::string> registered_types() const {
        std::vector<std::string> types;
        for (const auto& [name, _] : factories_)
            types.push_back(name);
        return types;
    }

private:
    PluginRegistry() = default;
    std::unordered_map<std::string, FactoryFn> factories_;
};

// Macro for self-registration
#define REGISTER_PLUGIN(TypeName, ClassName) \
    static bool _registered_##ClassName = []() { \
        PluginRegistry::instance().register_type( \
            TypeName, \
            []() { return std::make_unique<ClassName>(); }); \
        return true; \
    }();
