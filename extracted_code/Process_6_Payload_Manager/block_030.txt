// data/payload_status.h
#pragma once
#include <cstdint>
#include <atomic>

struct PayloadStatus {
    // Gimbal
    float gimbal_pitch;           // degrees
    float gimbal_roll;
    float gimbal_yaw;
    uint8_t gimbal_mode;          // GimbalMode enum
    bool gimbal_connected;

    // Camera
    uint32_t images_captured;
    bool camera_active;
    bool camera_connected;

    // Aux
    float spotlight_brightness;   // 0.0-1.0
    bool drop_armed;
    bool drop_released;

    // Overall
    uint8_t num_plugins_active;
    uint8_t num_plugins_error;
};

struct ShmPayloadStatus {
    std::atomic<uint64_t> seq;
    uint64_t timestamp_ns;
    PayloadStatus status;
};
