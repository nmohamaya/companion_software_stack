// ipc/payload_status_writer.h
#pragma once
#include "data/payload_status.h"
#include <sys/mman.h>
#include <fcntl.h>
#include <unistd.h>

class PayloadStatusWriter {
public:
    bool create(const char* name = "/payload_status") {
        fd_ = shm_open(name, O_CREAT | O_RDWR, 0666);
        if (fd_ < 0) return false;
        ftruncate(fd_, sizeof(ShmPayloadStatus));
        ptr_ = static_cast<ShmPayloadStatus*>(
            mmap(nullptr, sizeof(ShmPayloadStatus),
                 PROT_READ | PROT_WRITE, MAP_SHARED, fd_, 0));
        if (ptr_ == MAP_FAILED) return false;
        ptr_->seq.store(0);
        return true;
    }

    void write(const PayloadStatus& status) {
        if (!ptr_) return;
        uint64_t s = ptr_->seq.load(std::memory_order_relaxed);
        ptr_->seq.store(s + 1, std::memory_order_release);
        ptr_->timestamp_ns = current_time_ns();
        ptr_->status = status;
        ptr_->seq.store(s + 2, std::memory_order_release);
    }

    ~PayloadStatusWriter() {
        if (ptr_ && ptr_ != MAP_FAILED) munmap(ptr_, sizeof(ShmPayloadStatus));
        if (fd_ >= 0) { shm_unlink("/payload_status"); close(fd_); }
    }

private:
    int fd_ = -1;
    ShmPayloadStatus* ptr_ = nullptr;
    uint64_t current_time_ns() const;
};
