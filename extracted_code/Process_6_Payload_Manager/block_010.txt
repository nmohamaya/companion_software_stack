=== Listing 6-2: GimbalPlugin with PID Stabilisation ===

// gimbal/gimbal_plugin.h
#pragma once
#include "plugin/payload_plugin.h"
#include "gimbal/pid_controller.h"
#include "hardware/pwm_output.h"
#include "hardware/i2c_imu.h"
#include "ipc/pose_reader.h"
#include <Eigen/Core>
#include <Eigen/Geometry>

enum class GimbalMode : uint8_t {
    STABILISED, FOLLOW, LOCK, RATE, RETRACT
};

class GimbalPlugin : public PayloadPlugin {
public:
    std::string name() const override { return "gimbal"; }
    std::string type() const override { return "gimbal"; }

    void update() override {
        // Read vehicle attitude from SLAM
        Eigen::Vector3d pos;
        Eigen::Quaterniond vehicle_quat;
        Eigen::Vector3d vel;
        uint8_t quality;
        if (!pose_reader_.read(pos, vehicle_quat, vel, quality)) return;

        auto vehicle_euler = quat_to_euler(vehicle_quat);

        // Compute desired gimbal angles based on mode
        Eigen::Vector3d target = compute_target(vehicle_euler);

        // Read gimbal IMU for feedback (if available)
        Eigen::Vector3d gimbal_euler = gimbal_imu_ ?
            gimbal_imu_->read_euler() : vehicle_euler;

        // PID for each axis
        double dt = 1.0 / 200.0;  // 200 Hz
        double pitch_cmd = pitch_pid_.compute(target.x(), gimbal_euler.x(), dt);
        double roll_cmd  = roll_pid_.compute(target.y(), gimbal_euler.y(), dt);
        double yaw_cmd   = yaw_pid_.compute(target.z(), gimbal_euler.z(), dt);

        // Output to servos
        pwm_pitch_.set_normalised(pitch_cmd);  // -1.0 → +1.0
        pwm_roll_.set_normalised(roll_cmd);
        pwm_yaw_.set_normalised(yaw_cmd);
    }

    bool handle_command(const PayloadCmd& cmd) override {
        switch (cmd.type) {
            case PayloadCmdType::GIMBAL_SET_ANGLE:
                target_pitch_ = cmd.param1;
                target_roll_  = cmd.param2;
                target_yaw_   = cmd.param3;
                return true;

            case PayloadCmdType::GIMBAL_SET_MODE:
                mode_ = static_cast<GimbalMode>(static_cast<int>(cmd.param1));
                spdlog::info("Gimbal mode: {}", static_cast<int>(mode_));
                if (mode_ == GimbalMode::RETRACT) {
                    target_pitch_ = 0.0f;
                    target_roll_  = 0.0f;
                    target_yaw_   = 0.0f;
                }
                return true;

            case PayloadCmdType::GIMBAL_SET_RATE:
                rate_pitch_ = cmd.param1;
                rate_roll_  = cmd.param2;
                rate_yaw_   = cmd.param3;
                return true;

            default:
                return false;
        }
    }

    nlohmann::json status() const override {
        return {
            {"type", "gimbal"},
            {"mode", static_cast<int>(mode_)},
            {"pitch", current_pitch_},
            {"roll", current_roll_},
            {"yaw", current_yaw_},
            {"state", static_cast<int>(state_)}
        };
    }

protected:
    bool do_init(const nlohmann::json& config) override;
    void do_shutdown() override;

private:
    PoseReader pose_reader_;
    std::unique_ptr<I2CIMU> gimbal_imu_;
    PWMOutput pwm_pitch_, pwm_roll_, pwm_yaw_;
    PitchRollPID pitch_pid_{PIDGains{2.0, 0.1, 0.5, -1, 1, 0.3, 0.02}};
    PitchRollPID roll_pid_{PIDGains{2.0, 0.1, 0.5, -1, 1, 0.3, 0.02}};
    YawPID yaw_pid_{PIDGains{1.5, 0.05, 0.3, -1, 1, 0.3, 0.02}};

    GimbalMode mode_ = GimbalMode::STABILISED;
    float target_pitch_ = 0.0f, target_roll_ = 0.0f, target_yaw_ = 0.0f;
    float rate_pitch_ = 0.0f, rate_roll_ = 0.0f, rate_yaw_ = 0.0f;
    float current_pitch_ = 0.0f, current_roll_ = 0.0f, current_yaw_ = 0.0f;

    Eigen::Vector3d compute_target(const Eigen::Vector3d& vehicle_euler);
    Eigen::Vector3d quat_to_euler(const Eigen::Quaterniond& q);
};

REGISTER_PLUGIN("gimbal", GimbalPlugin);
