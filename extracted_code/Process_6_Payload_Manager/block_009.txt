=== Listing 6-1: CRTP PID Controller with Derivative Filtering ===

// gimbal/pid_controller.h
#pragma once
#include <algorithm>
#include <cmath>

struct PIDGains {
    double kp = 1.0;
    double ki = 0.0;
    double kd = 0.0;
    double output_min = -1.0;
    double output_max =  1.0;
    double integral_max = 0.5;   // anti-windup clamp
    double d_filter_tau = 0.02;  // derivative low-pass filter time constant
};

template <typename Derived>
class PIDControllerBase {
public:
    explicit PIDControllerBase(const PIDGains& gains)
        : gains_(gains) {}

    double compute(double setpoint, double measurement, double dt) {
        double error = static_cast<Derived*>(this)->wrap_error(
            setpoint - measurement);

        // Proportional
        double p_term = gains_.kp * error;

        // Integral with anti-windup
        integral_ += error * dt;
        integral_ = std::clamp(integral_, -gains_.integral_max,
                                gains_.integral_max);
        double i_term = gains_.ki * integral_;

        // Derivative with low-pass filter
        double raw_d = (error - prev_error_) / dt;
        double alpha = dt / (gains_.d_filter_tau + dt);
        filtered_d_ = alpha * raw_d + (1.0 - alpha) * filtered_d_;
        double d_term = gains_.kd * filtered_d_;

        prev_error_ = error;

        double output = p_term + i_term + d_term;
        return std::clamp(output, gains_.output_min, gains_.output_max);
    }

    void reset() {
        integral_ = 0.0;
        prev_error_ = 0.0;
        filtered_d_ = 0.0;
    }

    PIDGains& gains() { return gains_; }

protected:
    PIDGains gains_;
    double integral_ = 0.0;
    double prev_error_ = 0.0;
    double filtered_d_ = 0.0;
};

// Pitch/Roll PID — error is linear angle in radians
class PitchRollPID : public PIDControllerBase<PitchRollPID> {
public:
    using PIDControllerBase::PIDControllerBase;
    double wrap_error(double e) const { return e; }  // no wrapping needed
};

// Yaw PID — error wraps around ±π
class YawPID : public PIDControllerBase<YawPID> {
public:
    using PIDControllerBase::PIDControllerBase;
    double wrap_error(double e) const {
        while (e >  M_PI) e -= 2.0 * M_PI;
        while (e < -M_PI) e += 2.0 * M_PI;
        return e;
    }
};
