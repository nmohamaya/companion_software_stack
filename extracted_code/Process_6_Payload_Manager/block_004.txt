=== Listing 5-1: PayloadPlugin Interface (Template Method Pattern) ===

// plugin/payload_plugin.h
#pragma once
#include <string>
#include <memory>
#include <nlohmann/json.hpp>
#include <spdlog/spdlog.h>

enum class PluginState : uint8_t {
    UNINITIALIZED,
    INITIALIZING,
    READY,
    ACTIVE,
    ERROR,
    SHUTDOWN
};

class PayloadPlugin {
public:
    virtual ~PayloadPlugin() = default;

    // Template Method: init lifecycle
    bool init(const nlohmann::json& config) {
        state_ = PluginState::INITIALIZING;
        spdlog::info("Plugin [{}] initialising...", name());

        if (!do_init(config)) {
            state_ = PluginState::ERROR;
            spdlog::error("Plugin [{}] init failed", name());
            return false;
        }

        state_ = PluginState::READY;
        spdlog::info("Plugin [{}] ready", name());
        return true;
    }

    // Template Method: shutdown lifecycle
    void shutdown() {
        spdlog::info("Plugin [{}] shutting down...", name());
        do_shutdown();
        state_ = PluginState::SHUTDOWN;
    }

    // Called by the appropriate thread at its update rate
    virtual void update() = 0;

    // Handle a command routed to this plugin
    virtual bool handle_command(const PayloadCmd& cmd) = 0;

    // Status query
    virtual nlohmann::json status() const = 0;

    // Identity
    virtual std::string name() const = 0;
    virtual std::string type() const = 0;

    PluginState state() const { return state_; }

protected:
    // Customisation points (Template Method hooks)
    virtual bool do_init(const nlohmann::json& config) = 0;
    virtual void do_shutdown() = 0;

    PluginState state_ = PluginState::UNINITIALIZED;
};
