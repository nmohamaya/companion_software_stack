// The tracker OWNS tracks via unique_ptr
class ObjectTracker {
    std::vector<std::unique_ptr<Track>> active_tracks_;

public:
    void create_track(const Detection2D& det) {
        // make_unique: single allocation, exception-safe
        active_tracks_.push_back(
            std::make_unique<Track>(det, next_track_id_++));
    }

    // Fusion thread OBSERVES tracks (does not own them)
    // Returns raw pointer — caller must not delete or store long-term
    const Track* get_track(uint32_t id) const {
        for (auto& t : active_tracks_) {
            if (t->id == id) return t.get();  // non-owning view
        }
        return nullptr;
    }

    void remove_stale_tracks() {
        // erase-remove idiom: unique_ptr destructor frees the Track
        active_tracks_.erase(
            std::remove_if(active_tracks_.begin(), active_tracks_.end(),
                [](const auto& t) { return t->is_stale(); }),
            active_tracks_.end());
        // Stale Track objects are automatically freed — RAII!
    }
};
