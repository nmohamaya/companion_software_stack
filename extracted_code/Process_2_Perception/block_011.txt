/// 3D point from LiDAR
struct Point3D {
    float x, y, z;           // meters, in LiDAR frame
    float intensity;          // reflectance (0-255)
    uint8_t ring;             // laser ring ID (for structured LiDARs)
};

/// LiDAR point cloud for one scan
struct PointCloud {
    std::vector<Point3D> points;
    uint64_t timestamp_ns;
    uint32_t scan_id;

    void reserve(size_t n) { points.reserve(n); }
    void clear() { points.clear(); }
    size_t size() const { return points.size(); }
};

/// Clustered LiDAR points (after ground removal + clustering)
struct LiDARCluster {
    std::vector<Point3D> points;
    Eigen::Vector3f centroid;       // cluster center
    Eigen::Vector3f bbox_min;       // AABB minimum
    Eigen::Vector3f bbox_max;       // AABB maximum
    float distance;                 // range from sensor
};
