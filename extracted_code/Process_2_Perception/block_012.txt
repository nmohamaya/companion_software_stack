/// Radar detection (already processed by radar sensor firmware)
struct RadarDetection {
    float range;              // meters
    float azimuth;            // radians
    float elevation;          // radians (if 4D radar)
    float velocity_radial;    // m/s (Doppler) — positive = approaching
    float rcs;                // Radar Cross Section (dBsm)
    uint64_t timestamp_ns;
};

struct RadarDetectionList {
    std::vector<RadarDetection> detections;
    uint64_t timestamp_ns;
};
