// file: src/perception/kalman_tracker.h
#pragma once
#include <Eigen/Dense>

namespace drone::perception {

/// 2D Kalman filter for a single tracked object.
/// State: [cx, cy, w, h, vx, vy, vw, vh] — center, size, velocities
/// Measurement: [cx, cy, w, h] — from bounding box detection
class KalmanBoxTracker {
public:
    static constexpr int STATE_DIM = 8;
    static constexpr int MEAS_DIM  = 4;

    using StateVec = Eigen::Matrix<float, STATE_DIM, 1>;
    using StateMat = Eigen::Matrix<float, STATE_DIM, STATE_DIM>;
    using MeasVec  = Eigen::Matrix<float, MEAS_DIM, 1>;
    using MeasMat  = Eigen::Matrix<float, MEAS_DIM, STATE_DIM>;

    explicit KalmanBoxTracker(const Detection2D& initial_det, uint32_t id)
        : track_id(id)
    {
        // Initialize state from first detection
        x_ << initial_det.x + initial_det.w/2,  // cx
               initial_det.y + initial_det.h/2,  // cy
               initial_det.w,                     // w
               initial_det.h,                     // h
               0, 0, 0, 0;                        // velocities = 0

        // State transition: constant velocity model
        F_ = StateMat::Identity();
        // F(0,4)=dt, F(1,5)=dt, etc. — set in predict()

        // Measurement matrix: we observe [cx, cy, w, h]
        H_ = MeasMat::Zero();
        H_(0,0) = H_(1,1) = H_(2,2) = H_(3,3) = 1.0f;

        // Covariances (tuned for pixel coordinates)
        P_ = StateMat::Identity() * 10.0f;    // initial uncertainty
        Q_ = StateMat::Identity() * 0.01f;    // process noise
        Q_(4,4) = Q_(5,5) = Q_(6,6) = Q_(7,7) = 0.1f;  // velocity noise higher
        R_ = Eigen::Matrix<float, MEAS_DIM, MEAS_DIM>::Identity() * 1.0f;
    }

    /// Predict state forward by dt seconds
    void predict(float dt = 1.0f/30.0f) {
        F_ = StateMat::Identity();
        F_(0, 4) = dt;  // cx += vx * dt
        F_(1, 5) = dt;  // cy += vy * dt
        F_(2, 6) = dt;  // w  += vw * dt
        F_(3, 7) = dt;  // h  += vh * dt

        x_ = F_ * x_;
        P_ = F_ * P_ * F_.transpose() + Q_;
        age++;
    }

    /// Update state with a matched detection
    void update(const Detection2D& det) {
        MeasVec z;
        z << det.x + det.w/2, det.y + det.h/2, det.w, det.h;

        // Kalman gain
        auto S = H_ * P_ * H_.transpose() + R_;
        auto K = P_ * H_.transpose() * S.inverse();

        // State update
        x_ = x_ + K * (z - H_ * x_);
        P_ = (StateMat::Identity() - K * H_) * P_;

        hits++;
        consecutive_misses = 0;
        confidence = det.confidence;
    }

    /// Get current bounding box prediction
    Detection2D predicted_bbox() const {
        return {x_(0) - x_(2)/2, x_(1) - x_(3)/2,
                x_(2), x_(3), confidence, class_id, 0, 0};
    }

    /// Get estimated velocity (pixels/sec)
    Eigen::Vector2f velocity() const { return {x_(4), x_(5)}; }

    uint32_t track_id;
    ObjectClass class_id = ObjectClass::UNKNOWN;
    float confidence = 0;
    uint32_t age = 0;
    uint32_t hits = 0;
    uint32_t consecutive_misses = 0;

    bool is_confirmed() const { return hits >= 3; }
    bool is_stale() const { return consecutive_misses > 10; }

private:
    StateVec x_;              // state vector
    StateMat F_, P_, Q_;      // transition, covariance, process noise
    MeasMat H_;               // measurement matrix
    Eigen::Matrix<float, MEAS_DIM, MEAS_DIM> R_;  // measurement noise
};

} // namespace drone::perception
