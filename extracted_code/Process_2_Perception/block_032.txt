# file: process2_perception/CMakeLists.txt

cmake_minimum_required(VERSION 3.16)
project(perception LANGUAGES CXX CUDA)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

# ── Dependencies ──
find_package(OpenCV REQUIRED COMPONENTS core imgproc calib3d dnn)
find_package(Eigen3 3.3 REQUIRED NO_MODULE)
find_package(CUDA REQUIRED)
find_package(PkgConfig REQUIRED)
pkg_check_modules(SPDLOG REQUIRED spdlog)

# TensorRT (Jetson: typically in /usr/lib/aarch64-linux-gnu/)
find_path(TENSORRT_INCLUDE NvInfer.h PATHS /usr/include/aarch64-linux-gnu)
find_library(TENSORRT_LIB nvinfer PATHS /usr/lib/aarch64-linux-gnu)

# ── Library ──
add_library(perception_lib STATIC
    src/perception/lidar_thread.cpp
    src/perception/radar_thread.cpp
    src/perception/inference_thread.cpp
    src/perception/tracker_thread.cpp
    src/perception/fusion_thread.cpp
    src/perception/hungarian.cpp
)

target_include_directories(perception_lib PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}/include
    ${OpenCV_INCLUDE_DIRS}
    ${EIGEN3_INCLUDE_DIRS}
    ${TENSORRT_INCLUDE}
    ${SPDLOG_INCLUDE_DIRS}
    ${CUDA_INCLUDE_DIRS}
)

target_link_libraries(perception_lib PUBLIC
    ${OpenCV_LIBS}
    Eigen3::Eigen
    ${TENSORRT_LIB}
    ${CUDA_LIBRARIES}
    ${SPDLOG_LIBRARIES}
    rt pthread systemd
)

# ── Executable ──
add_executable(perception src/perception/main.cpp)
target_link_libraries(perception PRIVATE perception_lib)

install(TARGETS perception DESTINATION /opt/drone/bin)

# ── Tests ──
enable_testing()
add_executable(test_kalman tests/test_kalman.cpp)
target_link_libraries(test_kalman perception_lib)
add_test(NAME KalmanTest COMMAND test_kalman)

add_executable(test_hungarian tests/test_hungarian.cpp)
target_link_libraries(test_hungarian perception_lib)
add_test(NAME HungarianTest COMMAND test_hungarian)

add_executable(test_fusion tests/test_fusion.cpp)
target_link_libraries(test_fusion perception_lib)
add_test(NAME FusionTest COMMAND test_fusion)
