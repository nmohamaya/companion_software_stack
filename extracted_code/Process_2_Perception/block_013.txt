/// State of a tracked object (maintained by Kalman filter)
struct TrackedObject {
    uint32_t track_id;            // unique ID, monotonically increasing
    ObjectClass class_id;
    float confidence;              // decays over time without measurement

    // 2D state (camera frame)
    Eigen::Vector2f position_2d;   // center in pixels
    Eigen::Vector2f velocity_2d;   // pixels/sec

    // Track lifecycle
    uint32_t age;                  // frames since creation
    uint32_t hits;                 // frames with successful measurement
    uint32_t misses;               // consecutive frames without measurement
    uint64_t timestamp_ns;

    enum class State { TENTATIVE, CONFIRMED, LOST };
    State state;
};

/// List of all currently tracked objects
struct TrackedObjectList {
    std::vector<TrackedObject> objects;
    uint64_t timestamp_ns;
    uint64_t frame_sequence;
};
