// The Strategy interface — any detection algorithm implements this
class IDetector {
public:
    virtual ~IDetector() = default;
    virtual std::vector<Detection2D> detect(const cv::Mat& frame) = 0;
    virtual std::string name() const = 0;
};

// Concrete strategy 1: TensorRT-based YOLO
class TensorRTYoloDetector : public IDetector {
public:
    std::vector<Detection2D> detect(const cv::Mat& frame) override { /* ... */ }
    std::string name() const override { return "YOLOv8-TensorRT"; }
};

// Concrete strategy 2: OpenCV DNN fallback
class OpenCVDNNDetector : public IDetector {
    std::vector<Detection2D> detect(const cv::Mat& frame) override { /* ... */ }
    std::string name() const override { return "MobileNet-SSD-OpenCV"; }
};

// The context — doesn't know or care which detector it's using
class InferenceThread {
    std::unique_ptr<IDetector> detector_;  // owns the strategy
public:
    InferenceThread(std::unique_ptr<IDetector> det)
        : detector_(std::move(det)) {}
    // detector_ is used polymorphically — can be swapped at construction
};
