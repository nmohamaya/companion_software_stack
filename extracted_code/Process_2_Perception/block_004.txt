// RAII for a CUDA stream — used by inference thread
class CudaStream {
public:
    CudaStream() {
        cudaStreamCreate(&stream_);  // acquire
    }
    ~CudaStream() {
        cudaStreamDestroy(stream_);  // release — guaranteed even on exception
    }

    // Non-copyable (two objects can't own the same stream)
    CudaStream(const CudaStream&) = delete;
    CudaStream& operator=(const CudaStream&) = delete;

    // Movable (transfer ownership)
    CudaStream(CudaStream&& other) noexcept : stream_(other.stream_) {
        other.stream_ = nullptr;
    }

    cudaStream_t get() const { return stream_; }

private:
    cudaStream_t stream_ = nullptr;
};

// Usage: CudaStream is destroyed when it goes out of scope — even if
// an exception is thrown. No leak possible.
