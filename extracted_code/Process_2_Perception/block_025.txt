// file: src/perception/tracker_thread.cpp

void tracker_thread(
    SPSCRing<Detection2DList, 4>& input_queue,      // from inference thread
    SPSCRing<TrackedObjectList, 4>& output_queue,    // to fusion thread
    std::atomic<bool>& stop_flag)
{
    std::vector<std::unique_ptr<KalmanBoxTracker>> tracks;
    uint32_t next_id = 1;

    spdlog::info("Tracker thread started");

    while (!stop_flag.load(std::memory_order_relaxed)) {
        auto dets_opt = input_queue.try_pop();
        if (!dets_opt) {
            std::this_thread::sleep_for(std::chrono::microseconds(100));
            continue;
        }

        auto& dets = *dets_opt;
        float dt = 1.0f / 30.0f;  // TODO: compute actual dt from timestamps

        // Step 1: Predict all tracks forward
        for (auto& t : tracks) t->predict(dt);

        // Step 2: Build cost matrix (1 - IoU, so lower = better match)
        size_t num_dets = dets.detections.size();
        size_t num_tracks = tracks.size();
        std::vector<std::vector<double>> cost(num_dets,
            std::vector<double>(num_tracks, 1.0));
        for (size_t i = 0; i < num_dets; ++i) {
            for (size_t j = 0; j < num_tracks; ++j) {
                auto pred = tracks[j]->predicted_bbox();
                cost[i][j] = 1.0 - iou(dets.detections[i], pred);
            }
        }

        // Step 3: Hungarian assignment
        auto result = HungarianSolver::solve(cost, 0.7);

        // Step 4a: Update matched tracks
        for (size_t i = 0; i < result.assignment.size(); ++i) {
            int j = result.assignment[i];
            if (j >= 0) {
                tracks[j]->update(dets.detections[i]);
                tracks[j]->class_id = dets.detections[i].class_id;
            }
        }

        // Step 4b: Create new tracks for unmatched detections
        for (int i : result.unmatched_rows) {
            tracks.push_back(
                std::make_unique<KalmanBoxTracker>(
                    dets.detections[i], next_id++));
        }

        // Step 4c: Mark unmatched tracks
        for (int j : result.unmatched_cols) {
            tracks[j]->consecutive_misses++;
        }

        // Step 4d: Remove stale tracks
        tracks.erase(
            std::remove_if(tracks.begin(), tracks.end(),
                [](const auto& t) { return t->is_stale(); }),
            tracks.end());

        // Build output: only confirmed tracks
        TrackedObjectList output;
        output.timestamp_ns = dets.timestamp_ns;
        output.frame_sequence = dets.frame_sequence;
        for (auto& t : tracks) {
            if (t->is_confirmed()) {
                TrackedObject obj;
                obj.track_id = t->track_id;
                obj.class_id = t->class_id;
                obj.confidence = t->confidence;
                obj.position_2d = {t->predicted_bbox().x + t->predicted_bbox().w/2,
                                   t->predicted_bbox().y + t->predicted_bbox().h/2};
                obj.velocity_2d = t->velocity();
                obj.age = t->age;
                obj.hits = t->hits;
                obj.misses = t->consecutive_misses;
                obj.state = TrackedObject::State::CONFIRMED;
                obj.timestamp_ns = dets.timestamp_ns;
                output.objects.push_back(obj);
            }
        }

        output_queue.try_push(std::move(output));
    }
    spdlog::info("Tracker thread stopped");
}
