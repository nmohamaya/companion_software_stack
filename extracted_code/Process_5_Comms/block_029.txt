=== Listing 12-1: Transport Interface ===

// transport/transport_interface.h
#pragma once
#include <cstdint>
#include <vector>
#include <functional>
#include <memory>

class Transport {
public:
    virtual ~Transport() = default;

    virtual bool start() = 0;
    virtual void stop() = 0;
    virtual bool is_connected() const = 0;

    // Send raw bytes
    virtual bool send(const std::vector<uint8_t>& data) = 0;

    // Register receive callback
    using ReceiveCallback = std::function<void(const uint8_t*, std::size_t)>;
    virtual void on_receive(ReceiveCallback cb) = 0;

    // Transport info
    virtual std::string name() const = 0;
    virtual double bandwidth_bps() const = 0;  // estimated link bandwidth
};

// Factory
std::unique_ptr<Transport> create_transport(const nlohmann::json& config);
