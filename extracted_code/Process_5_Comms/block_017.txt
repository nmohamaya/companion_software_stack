=== Listing 8-2: Priority-Based Telemetry Scheduler ===

// telemetry/telemetry_scheduler.h
#pragma once
#include "transport/rate_limiter.h"
#include "mavlink/mavlink_wrapper.h"
#include <vector>
#include <functional>
#include <chrono>

struct TelemetryStream {
    std::string name;
    uint32_t msg_id;
    double rate_hz;                        // desired send rate
    uint8_t priority;                      // 0=highest, 255=lowest
    std::function<mavlink_message_t()> pack_fn;
    std::chrono::steady_clock::time_point last_sent;
};

class TelemetryScheduler {
public:
    explicit TelemetryScheduler(double link_bps, double burst = 4096.0)
        : limiter_(link_bps, burst) {}

    void add_stream(TelemetryStream stream) {
        stream.last_sent = std::chrono::steady_clock::now();
        streams_.push_back(std::move(stream));
        // Sort by priority (lower = higher priority)
        std::sort(streams_.begin(), streams_.end(),
                  [](const auto& a, const auto& b) {
                      return a.priority < b.priority;
                  });
    }

    // Called from downlink thread; returns messages ready to send
    std::vector<mavlink_message_t> tick() {
        auto now = std::chrono::steady_clock::now();
        std::vector<mavlink_message_t> out;

        for (auto& stream : streams_) {
            double interval = 1.0 / stream.rate_hz;
            auto elapsed = std::chrono::duration<double>(
                now - stream.last_sent).count();

            if (elapsed >= interval) {
                auto msg = stream.pack_fn();
                auto bytes = mavlink_msg_get_send_buffer_length(&msg);

                if (limiter_.try_consume(bytes)) {
                    out.push_back(msg);
                    stream.last_sent = now;
                }
                // If rate limited, skip this tick — will retry next tick
            }
        }
        return out;
    }

    void set_link_bandwidth(double bps) { limiter_.set_bandwidth(bps); }

private:
    TokenBucketRateLimiter limiter_;
    std::vector<TelemetryStream> streams_;
};
