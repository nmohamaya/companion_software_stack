=== Listing 8-1: Token Bucket Rate Limiter ===

// transport/rate_limiter.h
#pragma once
#include <chrono>
#include <algorithm>

class TokenBucketRateLimiter {
public:
    // bandwidth_bps: link bandwidth in bytes per second
    // burst_bytes:  maximum burst size
    TokenBucketRateLimiter(double bandwidth_bps, double burst_bytes)
        : rate_(bandwidth_bps), max_tokens_(burst_bytes),
          tokens_(burst_bytes),
          last_refill_(std::chrono::steady_clock::now()) {}

    bool try_consume(std::size_t bytes) {
        refill();
        if (tokens_ >= static_cast<double>(bytes)) {
            tokens_ -= static_cast<double>(bytes);
            return true;
        }
        return false;
    }

    void set_bandwidth(double bps) { rate_ = bps; }

private:
    double rate_;        // bytes per second
    double max_tokens_;
    double tokens_;
    std::chrono::steady_clock::time_point last_refill_;

    void refill() {
        auto now = std::chrono::steady_clock::now();
        double elapsed = std::chrono::duration<double>(now - last_refill_).count();
        tokens_ = std::min(max_tokens_, tokens_ + elapsed * rate_);
        last_refill_ = now;
    }
};
