=== Listing 10-2: Heartbeat Monitor Thread (1 Hz) ===

// monitor/heartbeat_thread.cpp
#include "monitor/link_health.h"
#include "transport/serial_transport.h"
#include "transport/udp_transport.h"
#include "comms/comms_context.h"
#include "util/realtime.h"

void heartbeat_monitor_thread(CommsContext& ctx,
                               SerialTransport& fc,
                               UDPTransport& gcs_udp,
                               std::atomic<bool>& running) {
    set_thread_params("HBMon", 3, SCHED_FIFO, 40);

    LinkHealth fc_link("FC", 5.0);
    LinkHealth gcs_link("GCS", 10.0);  // GCS may be on slow radio

    // Register heartbeat handler for FC
    fc.on_message(MAVLINK_MSG_ID_HEARTBEAT, [&](const mavlink_message_t& msg) {
        if (msg.sysid == 1)  // Flight controller
            fc_link.heartbeat_received();
    });

    // GCS heartbeats come via UDP
    gcs_udp.on_message([&](const mavlink_message_t& msg, const auto&) {
        if (msg.msgid == MAVLINK_MSG_ID_HEARTBEAT && msg.sysid >= 200)
            gcs_link.heartbeat_received();
    });

    const auto period = std::chrono::seconds(1);  // 1 Hz

    while (running.load(std::memory_order_relaxed)) {
        auto start = std::chrono::steady_clock::now();

        // Send our heartbeat to FC and GCS
        auto hb = ctx.mavlink().pack_heartbeat();
        fc.send(hb);
        gcs_udp.send_to_gcs(hb);

        // Check link health
        auto fc_status = fc_link.status();
        auto gcs_status = gcs_link.status();

        if (fc_status == LinkStatus::LOST) {
            spdlog::error("FC LINK LOST — no heartbeat for {:.1f}s",
                           fc_link.seconds_since_last());
            ctx.set_fc_link_status(false);
            // Trigger failsafe via Process 4
            ctx.gcs_cmd_writer().write_command(GCSCommand::EMERGENCY_STOP);
        } else {
            ctx.set_fc_link_status(true);
        }

        if (gcs_status == LinkStatus::LOST) {
            spdlog::warn("GCS LINK LOST — no heartbeat for {:.1f}s",
                          gcs_link.seconds_since_last());
            ctx.set_gcs_link_status(false);
            // Configurable: continue mission or RTL
            if (ctx.config().gcs_loss_action == "rtl") {
                ctx.gcs_cmd_writer().write_command(GCSCommand::RTL);
            }
        } else {
            ctx.set_gcs_link_status(true);
        }

        // Publish link stats to mission status
        ctx.update_link_quality(
            fc_link.avg_interval(), gcs_link.avg_interval(),
            fc_status, gcs_status);

        spdlog::debug("Links — FC: {} ({:.0f}ms avg), GCS: {} ({:.0f}ms avg)",
                       link_status_str(fc_status),
                       fc_link.avg_interval() * 1000,
                       link_status_str(gcs_status),
                       gcs_link.avg_interval() * 1000);

        std::this_thread::sleep_until(start + period);
    }
}
