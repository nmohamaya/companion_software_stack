=== Listing 14-1: GCS Command Types ===

// data/gcs_command.h
#pragma once
#include <cstdint>

enum class GCSCommand : uint8_t {
    NONE,
    ARM,
    DISARM,
    TAKEOFF,
    LAND,
    RTL,
    LOITER,
    RESUME,
    EMERGENCY_STOP,
    SET_MODE,
    MISSION_START,
    MISSION_PAUSE,
    MISSION_CLEAR,
};

struct GCSCommandMsg {
    GCSCommand command;
    float      param1;     // e.g., takeoff altitude
    float      param2;
    uint64_t   timestamp_ns;
};
