=== Listing 6-2: Async Serial Transport with Message Dispatch ===

// transport/serial_transport.h
#pragma once
#include "transport/serial_port.h"
#include "mavlink/mavlink_wrapper.h"
#include <functional>
#include <array>
#include <deque>
#include <mutex>

using MessageHandler = std::function<void(const mavlink_message_t&)>;

class SerialTransport {
public:
    SerialTransport(asio::io_context& io, const std::string& device,
                     unsigned baud, uint8_t chan_id = MAVLINK_COMM_0)
        : io_(io), serial_(io, device, baud), channel_(chan_id) {}

    // Register handler for specific message IDs
    void on_message(uint32_t msg_id, MessageHandler handler) {
        handlers_[msg_id] = std::move(handler);
    }

    // Register catch-all handler
    void on_any_message(MessageHandler handler) {
        any_handler_ = std::move(handler);
    }

    // Start async read loop
    void start() {
        do_read();
    }

    // Queue a message for sending (thread-safe)
    void send(const mavlink_message_t& msg) {
        auto bytes = channel_.serialise(msg);
        {
            std::lock_guard lock(tx_mtx_);
            tx_queue_.push_back(std::move(bytes));
        }
        // Trigger write if not already in progress
        asio::post(io_, [this]() { do_write(); });
    }

private:
    asio::io_context& io_;
    SerialPort serial_;
    MAVLinkChannel channel_;

    std::array<uint8_t, 512> rx_buf_;
    std::mutex tx_mtx_;
    std::deque<std::vector<uint8_t>> tx_queue_;
    bool writing_ = false;

    std::unordered_map<uint32_t, MessageHandler> handlers_;
    MessageHandler any_handler_;

    void do_read() {
        serial_.native().async_read_some(
            asio::buffer(rx_buf_),
            [this](boost::system::error_code ec, std::size_t n) {
                if (ec) {
                    spdlog::error("Serial read error: {}", ec.message());
                    reconnect();
                    return;
                }

                for (std::size_t i = 0; i < n; ++i) {
                    if (auto msg = channel_.parse_byte(rx_buf_[i])) {
                        dispatch(*msg);
                    }
                }

                do_read();  // continue reading
            });
    }

    void dispatch(const mavlink_message_t& msg) {
        if (any_handler_) any_handler_(msg);
        auto it = handlers_.find(msg.msgid);
        if (it != handlers_.end()) {
            it->second(msg);
        }
    }

    void do_write() {
        if (writing_) return;

        std::vector<uint8_t> data;
        {
            std::lock_guard lock(tx_mtx_);
            if (tx_queue_.empty()) return;
            data = std::move(tx_queue_.front());
            tx_queue_.pop_front();
        }

        writing_ = true;
        asio::async_write(serial_.native(), asio::buffer(data),
            [this](boost::system::error_code ec, std::size_t) {
                writing_ = false;
                if (ec) {
                    spdlog::error("Serial write error: {}", ec.message());
                    return;
                }
                do_write();  // drain queue
            });
    }

    void reconnect();  // attempt to re-open serial port after error
};
