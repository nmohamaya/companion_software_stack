// src/main.cpp — Process 5: Comms
#include "comms/comms_context.h"
#include "transport/serial_transport.h"
#include "transport/udp_transport.h"
#include "relay/trajectory_relay_thread.h"
#include "telemetry/downlink_thread.h"
#include "telemetry/uplink_thread.h"
#include "monitor/heartbeat_thread.h"
#include "config/config_loader.h"
#include "util/realtime.h"
#include <csignal>
#include <thread>
#include <atomic>
#include <spdlog/spdlog.h>

static std::atomic<bool> g_running{true};

void signal_handler(int sig) {
    spdlog::info("Signal {} received, shutting down...", sig);
    g_running.store(false);
}

int main(int argc, char* argv[]) {
    std::signal(SIGTERM, signal_handler);
    std::signal(SIGINT, signal_handler);

    // Parse args
    std::string config_path = "/opt/drone/etc/comms_config.json";
    for (int i = 1; i < argc - 1; ++i) {
        if (std::string(argv[i]) == "--config") config_path = argv[i + 1];
    }

    // Load config
    CommsConfig cfg;
    load_comms_config(config_path, cfg);
    spdlog::set_level(spdlog::level::from_str(cfg.log_level));

    spdlog::info("=== Process 5: Comms starting ===");

    // Create shared context
    CommsContext ctx(cfg);

    // Open IPC
    if (!ctx.init_ipc()) {
        spdlog::critical("Failed to initialise SHM segments");
        return 1;
    }

    // Create transports
    asio::io_context fc_io;
    SerialTransport fc_serial(fc_io, cfg.fc_device, cfg.fc_baud,
                               MAVLINK_COMM_0);

    asio::io_context gcs_io;
    UDPTransport gcs_udp(gcs_io, cfg.gcs_udp_port, MAVLINK_COMM_1);

    // Launch threads
    std::vector<std::thread> threads;

    threads.emplace_back(serial_transport_thread,
                          std::ref(cfg), std::ref(fc_serial), std::ref(g_running));
    threads.emplace_back(telemetry_uplink_thread,
                          std::ref(ctx), std::ref(gcs_udp), std::ref(g_running));
    threads.emplace_back(telemetry_downlink_thread,
                          std::ref(ctx), std::ref(gcs_udp), std::ref(g_running));
    threads.emplace_back(trajectory_relay_thread,
                          std::ref(ctx.traj_reader()), std::ref(fc_serial),
                          std::ref(ctx.mavlink()), std::ref(g_running));
    threads.emplace_back(heartbeat_monitor_thread,
                          std::ref(ctx), std::ref(fc_serial), std::ref(gcs_udp),
                          std::ref(g_running));

    spdlog::info("All 5 threads launched");

    // Wait for shutdown
    for (auto& t : threads)
        t.join();

    spdlog::info("=== Process 5: Comms shutdown complete ===");
    return 0;
}
