=== Listing 17-1: GCS Command Writer (SeqLock SHM) ===

// ipc/gcs_command_writer.h
#pragma once
#include "data/gcs_command.h"
#include <atomic>
#include <sys/mman.h>
#include <fcntl.h>
#include <unistd.h>

struct ShmGCSCommand {
    std::atomic<uint64_t> seq;
    GCSCommandMsg cmd;
};

class GCSCommandWriter {
public:
    bool create(const char* name = "/gcs_commands") {
        fd_ = shm_open(name, O_CREAT | O_RDWR, 0666);
        if (fd_ < 0) return false;
        ftruncate(fd_, sizeof(ShmGCSCommand));
        ptr_ = static_cast<ShmGCSCommand*>(
            mmap(nullptr, sizeof(ShmGCSCommand),
                 PROT_READ | PROT_WRITE, MAP_SHARED, fd_, 0));
        if (ptr_ == MAP_FAILED) return false;
        ptr_->seq.store(0);
        return true;
    }

    void write_command(GCSCommand cmd, float p1 = 0.0f, float p2 = 0.0f) {
        if (!ptr_) return;
        uint64_t s = ptr_->seq.load(std::memory_order_relaxed);
        ptr_->seq.store(s + 1, std::memory_order_release);

        ptr_->cmd.command = cmd;
        ptr_->cmd.param1 = p1;
        ptr_->cmd.param2 = p2;
        ptr_->cmd.timestamp_ns = current_time_ns();

        ptr_->seq.store(s + 2, std::memory_order_release);
    }

    ~GCSCommandWriter() {
        if (ptr_ && ptr_ != MAP_FAILED) munmap(ptr_, sizeof(ShmGCSCommand));
        if (fd_ >= 0) { shm_unlink("/gcs_commands"); close(fd_); }
    }

private:
    int fd_ = -1;
    ShmGCSCommand* ptr_ = nullptr;
    uint64_t current_time_ns() const;
};
