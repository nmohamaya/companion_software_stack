=== Listing 13-1: MAVLink Message Router ===

// routing/message_router.h
#pragma once
#include <mavlink/v2.0/ardupilotmega/mavlink.h>
#include <unordered_set>
#include <functional>
#include <vector>

class MessageRouter {
public:
    enum class Direction { FC_TO_GCS, GCS_TO_FC, COMPANION_TO_FC, COMPANION_TO_GCS };

    using RouteHandler = std::function<void(const mavlink_message_t&, Direction)>;

    void add_route(RouteHandler handler) {
        routes_.push_back(std::move(handler));
    }

    // message IDs to NOT forward (high-bandwidth, companion handles internally)
    void add_block_filter(Direction dir, uint32_t msg_id) {
        if (dir == Direction::FC_TO_GCS)
            fc_to_gcs_block_.insert(msg_id);
        else if (dir == Direction::GCS_TO_FC)
            gcs_to_fc_block_.insert(msg_id);
    }

    void route_fc_to_gcs(const mavlink_message_t& msg) {
        if (fc_to_gcs_block_.count(msg.msgid)) return;
        for (auto& handler : routes_)
            handler(msg, Direction::FC_TO_GCS);
    }

    void route_gcs_to_fc(const mavlink_message_t& msg) {
        if (gcs_to_fc_block_.count(msg.msgid)) return;
        for (auto& handler : routes_)
            handler(msg, Direction::GCS_TO_FC);
    }

    void inject_companion_to_fc(const mavlink_message_t& msg) {
        for (auto& handler : routes_)
            handler(msg, Direction::COMPANION_TO_FC);
    }

    void inject_companion_to_gcs(const mavlink_message_t& msg) {
        for (auto& handler : routes_)
            handler(msg, Direction::COMPANION_TO_GCS);
    }

private:
    std::vector<RouteHandler> routes_;
    std::unordered_set<uint32_t> fc_to_gcs_block_;
    std::unordered_set<uint32_t> gcs_to_fc_block_;
};
