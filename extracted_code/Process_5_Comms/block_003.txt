=== Listing 4-1: Trajectory Command SHM Reader ===

// ipc/trajectory_reader.h — reads setpoints from Process 4
#pragma once
#include <atomic>
#include <cstring>
#include <sys/mman.h>
#include <fcntl.h>
#include <unistd.h>
#include <Eigen/Core>

struct ShmTrajectoryCmd {
    std::atomic<uint64_t> seq;
    uint64_t timestamp_ns;
    double position[3];
    double velocity[3];
    double acceleration[3];
    double yaw;
    double yaw_rate;
    uint16_t type_mask;
    uint8_t  coordinate_frame;
    uint8_t  padding[5];
};

class TrajectoryReader {
public:
    bool open(const char* name = "/trajectory_cmd") {
        fd_ = shm_open(name, O_RDONLY, 0);
        if (fd_ < 0) return false;
        ptr_ = static_cast<ShmTrajectoryCmd*>(
            mmap(nullptr, sizeof(ShmTrajectoryCmd), PROT_READ, MAP_SHARED, fd_, 0));
        return ptr_ != MAP_FAILED;
    }

    struct SetpointData {
        uint64_t timestamp_ns;
        double position[3];
        double velocity[3];
        double acceleration[3];
        double yaw;
        double yaw_rate;
        uint16_t type_mask;
        uint8_t  coordinate_frame;
    };

    bool read(SetpointData& out) const {
        if (!ptr_) return false;
        for (int attempt = 0; attempt < 4; ++attempt) {
            uint64_t s1 = ptr_->seq.load(std::memory_order_acquire);
            if (s1 & 1) continue;
            out.timestamp_ns = ptr_->timestamp_ns;
            std::memcpy(out.position, ptr_->position, sizeof(out.position));
            std::memcpy(out.velocity, ptr_->velocity, sizeof(out.velocity));
            std::memcpy(out.acceleration, ptr_->acceleration, sizeof(out.acceleration));
            out.yaw = ptr_->yaw;
            out.yaw_rate = ptr_->yaw_rate;
            out.type_mask = ptr_->type_mask;
            out.coordinate_frame = ptr_->coordinate_frame;
            std::atomic_thread_fence(std::memory_order_acquire);
            uint64_t s2 = ptr_->seq.load(std::memory_order_relaxed);
            if (s1 == s2) return true;
        }
        return false;
    }

    ~TrajectoryReader() {
        if (ptr_ && ptr_ != MAP_FAILED) munmap(ptr_, sizeof(ShmTrajectoryCmd));
        if (fd_ >= 0) close(fd_);
    }

private:
    int fd_ = -1;
    ShmTrajectoryCmd* ptr_ = nullptr;
};
