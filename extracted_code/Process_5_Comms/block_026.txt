=== Listing 11-1: Lock-Free MPSC Bounded Queue (CAS-based) ===

// util/mpsc_queue.h
#pragma once
#include <atomic>
#include <optional>
#include <new>  // std::hardware_destructive_interference_size

template <typename T, std::size_t Capacity = 256>
class MPSCBoundedQueue {
    static_assert((Capacity & (Capacity - 1)) == 0,
                  "Capacity must be power of 2");

public:
    MPSCBoundedQueue() : head_(0), tail_(0) {
        for (auto& slot : slots_)
            slot.seq.store(0, std::memory_order_relaxed);
    }

    // Thread-safe for multiple producers
    bool try_push(const T& value) {
        std::size_t pos = tail_.load(std::memory_order_relaxed);

        for (;;) {
            auto& slot = slots_[pos & (Capacity - 1)];
            std::size_t seq = slot.seq.load(std::memory_order_acquire);
            intptr_t diff = static_cast<intptr_t>(seq) -
                            static_cast<intptr_t>(pos);

            if (diff == 0) {
                // Slot is available — try to claim it
                if (tail_.compare_exchange_weak(pos, pos + 1,
                        std::memory_order_relaxed)) {
                    slot.data = value;
                    slot.seq.store(pos + 1, std::memory_order_release);
                    return true;
                }
            } else if (diff < 0) {
                return false;  // queue full
            } else {
                pos = tail_.load(std::memory_order_relaxed);
            }
        }
    }

    // Single consumer only
    std::optional<T> try_pop() {
        auto& slot = slots_[head_ & (Capacity - 1)];
        std::size_t seq = slot.seq.load(std::memory_order_acquire);
        intptr_t diff = static_cast<intptr_t>(seq) -
                        static_cast<intptr_t>(head_ + 1);

        if (diff < 0) return std::nullopt;  // queue empty

        T value = std::move(slot.data);
        slot.seq.store(head_ + Capacity, std::memory_order_release);
        ++head_;
        return value;
    }

private:
    struct Slot {
        std::atomic<std::size_t> seq;
        T data;
    };

    // Pad to avoid false sharing between head and tail
    alignas(64) std::size_t head_;
    alignas(64) std::atomic<std::size_t> tail_;
    std::array<Slot, Capacity> slots_;
};
