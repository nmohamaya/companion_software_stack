// transport/transport_factory.cpp
#include "transport/transport_interface.h"
#include "transport/udp_transport_impl.h"
#include "transport/serial_radio_impl.h"

std::unique_ptr<Transport> create_transport(const nlohmann::json& config) {
    std::string type = config.value("type", "udp");

    if (type == "udp") {
        auto port = config.value("port", 14550);
        auto bind_ip = config.value("bind", "0.0.0.0");
        return std::make_unique<UDPTransportImpl>(bind_ip, port);
    }
    else if (type == "serial") {
        auto device = config.at("device").get<std::string>();
        auto baud = config.value("baud", 57600);
        return std::make_unique<SerialRadioImpl>(device, baud);
    }
    else {
        throw std::runtime_error("Unknown transport type: " + type);
    }
}
