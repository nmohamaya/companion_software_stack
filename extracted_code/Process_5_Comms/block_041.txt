# CMakeLists.txt — Process 5: Comms
cmake_minimum_required(VERSION 3.18)
project(comms VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# ── Compiler flags ──────────────────────────────────────────────
add_compile_options(
    -Wall -Wextra -Wpedantic -Werror
    -O2
    -march=armv8.2-a+fp16
)

# ── Dependencies ────────────────────────────────────────────────
find_package(Boost 1.74 REQUIRED COMPONENTS system)
find_package(Eigen3 3.4 REQUIRED NO_MODULE)
find_package(spdlog REQUIRED)
find_package(nlohmann_json 3.11 REQUIRED)
find_package(Threads REQUIRED)

# MAVLink headers (header-only, installed to /usr/local/include/mavlink)
find_path(MAVLINK_INCLUDE_DIR mavlink/v2.0/ardupilotmega/mavlink.h
          HINTS /usr/local/include /opt/drone/include)
if(NOT MAVLINK_INCLUDE_DIR)
    message(FATAL_ERROR "MAVLink headers not found")
endif()

# ── Sources ─────────────────────────────────────────────────────
set(COMMS_SOURCES
    src/main.cpp
    src/comms/comms_context.cpp
    src/mavlink/mavlink_wrapper.cpp
    src/transport/serial_port.cpp
    src/transport/serial_transport.cpp
    src/transport/serial_thread.cpp
    src/transport/udp_transport.cpp
    src/transport/transport_factory.cpp
    src/transport/rate_limiter.cpp
    src/telemetry/telemetry_scheduler.cpp
    src/telemetry/downlink_thread.cpp
    src/telemetry/uplink_thread.cpp
    src/relay/trajectory_relay_thread.cpp
    src/monitor/link_health.cpp
    src/monitor/heartbeat_thread.cpp
    src/routing/message_router.cpp
    src/comms/gcs_command_parser.cpp
    src/ipc/trajectory_reader.cpp
    src/ipc/mission_status_reader.cpp
    src/ipc/pose_reader.cpp
    src/ipc/gcs_command_writer.cpp
    src/ipc/fc_state_writer.cpp
    src/config/config_loader.cpp
    src/util/realtime.cpp
    src/util/reconnect.cpp
)

# ── Executable ──────────────────────────────────────────────────
add_executable(comms ${COMMS_SOURCES})

target_include_directories(comms PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/include
    ${MAVLINK_INCLUDE_DIR}
)

target_link_libraries(comms PRIVATE
    Boost::system
    Eigen3::Eigen
    spdlog::spdlog
    nlohmann_json::nlohmann_json
    Threads::Threads
    rt          # POSIX shared memory
    pthread     # POSIX threads
)

# ── Install ─────────────────────────────────────────────────────
install(TARGETS comms DESTINATION /opt/drone/bin)
install(FILES config/comms_config.json DESTINATION /opt/drone/etc)

# ── Tests ───────────────────────────────────────────────────────
option(BUILD_TESTS "Build unit tests" ON)
if(BUILD_TESTS)
    enable_testing()
    find_package(GTest REQUIRED)

    add_executable(test_mavlink_wrapper
        tests/test_mavlink_wrapper.cpp
        src/mavlink/mavlink_wrapper.cpp
    )
    target_include_directories(test_mavlink_wrapper PRIVATE include ${MAVLINK_INCLUDE_DIR})
    target_link_libraries(test_mavlink_wrapper GTest::gtest_main)
    add_test(NAME MAVLinkWrapperTest COMMAND test_mavlink_wrapper)

    add_executable(test_rate_limiter
        tests/test_rate_limiter.cpp
        src/transport/rate_limiter.cpp
    )
    target_include_directories(test_rate_limiter PRIVATE include)
    target_link_libraries(test_rate_limiter GTest::gtest_main)
    add_test(NAME RateLimiterTest COMMAND test_rate_limiter)

    add_executable(test_mpsc_queue tests/test_mpsc_queue.cpp)
    target_include_directories(test_mpsc_queue PRIVATE include)
    target_link_libraries(test_mpsc_queue GTest::gtest_main Threads::Threads)
    add_test(NAME MPSCQueueTest COMMAND test_mpsc_queue)

    add_executable(test_message_router
        tests/test_message_router.cpp
        src/routing/message_router.cpp
    )
    target_include_directories(test_message_router PRIVATE include ${MAVLINK_INCLUDE_DIR})
    target_link_libraries(test_message_router GTest::gtest_main)
    add_test(NAME MessageRouterTest COMMAND test_message_router)

    add_executable(test_gcs_parser
        tests/test_gcs_parser.cpp
        src/comms/gcs_command_parser.cpp
    )
    target_include_directories(test_gcs_parser PRIVATE include ${MAVLINK_INCLUDE_DIR})
    target_link_libraries(test_gcs_parser GTest::gtest_main)
    add_test(NAME GCSParserTest COMMAND test_gcs_parser)

    add_executable(test_link_health
        tests/test_link_health.cpp
        src/monitor/link_health.cpp
    )
    target_include_directories(test_link_health PRIVATE include)
    target_link_libraries(test_link_health GTest::gtest_main)
    add_test(NAME LinkHealthTest COMMAND test_link_health)
endif()
