=== Listing 6-3: Serial Transport Thread with io_context ===

// transport/serial_thread.cpp
#include "transport/serial_transport.h"
#include "util/realtime.h"

void serial_transport_thread(const CommsConfig& cfg,
                              SerialTransport& transport,
                              std::atomic<bool>& running) {
    set_thread_params("SerialIO", 3, SCHED_FIFO, 75);

    // The io_context::run() blocks until all async operations complete
    // or the context is stopped.
    asio::io_context& io = transport.io_context();

    transport.start();  // begin async reading

    while (running.load(std::memory_order_relaxed)) {
        try {
            io.run_for(std::chrono::milliseconds(100));
        } catch (const std::exception& e) {
            spdlog::error("Serial thread exception: {}", e.what());
            std::this_thread::sleep_for(std::chrono::seconds(1));
            io.restart();
        }
    }

    transport.stop();
}
