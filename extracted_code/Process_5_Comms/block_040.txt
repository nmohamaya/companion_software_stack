=== Listing 17-2: FC State Writer (SeqLock SHM) ===

// ipc/fc_state_writer.h
#pragma once
#include "data/fc_state.h"
#include <sys/mman.h>
#include <fcntl.h>
#include <unistd.h>

class FCStateWriter {
public:
    bool create(const char* name = "/fc_state") {
        fd_ = shm_open(name, O_CREAT | O_RDWR, 0666);
        if (fd_ < 0) return false;
        ftruncate(fd_, sizeof(ShmFCState));
        ptr_ = static_cast<ShmFCState*>(
            mmap(nullptr, sizeof(ShmFCState),
                 PROT_READ | PROT_WRITE, MAP_SHARED, fd_, 0));
        if (ptr_ == MAP_FAILED) return false;
        ptr_->seq.store(0);
        return true;
    }

    void write(const FCState& state) {
        if (!ptr_) return;
        uint64_t s = ptr_->seq.load(std::memory_order_relaxed);
        ptr_->seq.store(s + 1, std::memory_order_release);
        ptr_->timestamp_ns = current_time_ns();
        ptr_->state = state;
        ptr_->seq.store(s + 2, std::memory_order_release);
    }

    ~FCStateWriter() {
        if (ptr_ && ptr_ != MAP_FAILED) munmap(ptr_, sizeof(ShmFCState));
        if (fd_ >= 0) { shm_unlink("/fc_state"); close(fd_); }
    }

private:
    int fd_ = -1;
    ShmFCState* ptr_ = nullptr;
    uint64_t current_time_ns() const;
};
