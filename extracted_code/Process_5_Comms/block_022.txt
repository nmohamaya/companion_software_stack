=== Listing 10-1: Link Health Tracker ===

// monitor/link_health.h
#pragma once
#include <chrono>
#include <atomic>
#include <string>
#include <spdlog/spdlog.h>

enum class LinkStatus : uint8_t { HEALTHY, DEGRADED, LOST };

class LinkHealth {
public:
    explicit LinkHealth(std::string name, double timeout_s = 5.0)
        : name_(std::move(name)), timeout_s_(timeout_s) {}

    void heartbeat_received() {
        auto now = std::chrono::steady_clock::now();
        auto prev = last_heartbeat_.load(std::memory_order_relaxed);
        auto tp = now.time_since_epoch().count();
        last_heartbeat_.store(tp, std::memory_order_release);

        ++total_heartbeats_;

        // Calculate one-way latency estimate (heartbeat round-trip / 2)
        if (prev > 0) {
            double dt = std::chrono::duration<double>(
                now - std::chrono::steady_clock::time_point(
                    std::chrono::steady_clock::duration(prev))).count();
            // Exponential moving average
            avg_interval_s_ = 0.9 * avg_interval_s_ + 0.1 * dt;
        }
    }

    LinkStatus status() const {
        auto now = std::chrono::steady_clock::now();
        auto last = std::chrono::steady_clock::time_point(
            std::chrono::steady_clock::duration(
                last_heartbeat_.load(std::memory_order_acquire)));
        double elapsed = std::chrono::duration<double>(now - last).count();

        if (elapsed < timeout_s_ * 0.5) return LinkStatus::HEALTHY;
        if (elapsed < timeout_s_)        return LinkStatus::DEGRADED;
        return LinkStatus::LOST;
    }

    double seconds_since_last() const {
        auto now = std::chrono::steady_clock::now();
        auto last = std::chrono::steady_clock::time_point(
            std::chrono::steady_clock::duration(
                last_heartbeat_.load(std::memory_order_acquire)));
        return std::chrono::duration<double>(now - last).count();
    }

    double avg_interval() const { return avg_interval_s_; }
    uint64_t total() const { return total_heartbeats_; }
    const std::string& name() const { return name_; }

private:
    std::string name_;
    double timeout_s_;
    std::atomic<int64_t> last_heartbeat_{0};
    std::atomic<uint64_t> total_heartbeats_{0};
    double avg_interval_s_ = 1.0;
};
