=== Listing 16-1: Exponential Backoff with Jitter ===

// util/reconnect.h
#pragma once
#include <chrono>
#include <random>
#include <algorithm>
#include <thread>
#include <spdlog/spdlog.h>

class ExponentialBackoff {
public:
    ExponentialBackoff(double init_s = 1.0, double max_s = 30.0,
                       double multiplier = 2.0)
        : init_(init_s), max_(max_s), mult_(multiplier),
          current_(init_s), rng_(std::random_device{}()) {}

    void wait() {
        // Add jitter: ±25%
        double jitter = std::uniform_real_distribution<>(0.75, 1.25)(rng_);
        double wait_s = current_ * jitter;

        spdlog::info("Backoff: waiting {:.1f}s before retry", wait_s);
        std::this_thread::sleep_for(
            std::chrono::duration<double>(wait_s));

        current_ = std::min(current_ * mult_, max_);
    }

    void reset() { current_ = init_; }

private:
    double init_, max_, mult_, current_;
    std::mt19937 rng_;
};
