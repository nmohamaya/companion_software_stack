#pragma once
#include <atomic>
#include <array>
#include <optional>
#include <cassert>

namespace drone::video {

/// Lock-free SPSC ring buffer for intra-process thread communication.
/// T must be trivially copyable (e.g., FrameBuffer with pixel data).
/// N must be a power of 2.
template <typename T, size_t N>
class SPSCRing {
    static_assert((N & (N - 1)) == 0, "N must be power of 2");

public:
    /// Producer calls this. Returns false if ring is full (consumer too slow).
    bool try_push(const T& item) {
        const uint64_t w = write_idx_.load(std::memory_order_relaxed);
        const uint64_t r = read_idx_.load(std::memory_order_acquire);

        if (w - r >= N) {
            return false;  // ring full — drop frame or block
        }

        slots_[w & (N - 1)] = item;
        write_idx_.store(w + 1, std::memory_order_release);
        return true;
    }

    /// Consumer calls this. Returns std::nullopt if ring is empty.
    std::optional<T> try_pop() {
        const uint64_t r = read_idx_.load(std::memory_order_relaxed);
        const uint64_t w = write_idx_.load(std::memory_order_acquire);

        if (r >= w) {
            return std::nullopt;  // ring empty — no new frame
        }

        T item = slots_[r & (N - 1)];
        read_idx_.store(r + 1, std::memory_order_release);
        return item;
    }

    /// How many items are available to consume.
    uint64_t available() const {
        return write_idx_.load(std::memory_order_acquire)
             - read_idx_.load(std::memory_order_relaxed);
    }

private:
    alignas(64) std::atomic<uint64_t> write_idx_{0};  // own cache line
    alignas(64) std::atomic<uint64_t> read_idx_{0};   // own cache line
    std::array<T, N> slots_;
};

} // namespace drone::video

#pragma once
#include "frame.h"
#include <atomic>
#include <cstddef>
#include <sys/mman.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <unistd.h>
#include <cstring>
#include <stdexcept>

namespace drone::video {

/// Layout of the shared memory region.
/// This struct IS the shared memory — both producer and consumer map it.
struct ShmRingHeader {
    alignas(64) std::atomic<uint64_t> write_idx;   // producer increments
    alignas(64) std::atomic<uint64_t> read_idx;    // consumer increments
    uint32_t num_slots;
    uint32_t slot_size;    // sizeof(ShmFrameSlot) + pixel_data_size
    uint32_t pixel_data_size;
    uint32_t magic;        // 0xDEADCAFE — validates correct mapping
    // Slot data follows immediately after this header
};

struct ShmFrameSlot {
    FrameMetadata meta;
    uint8_t pixel_data[];  // flexible array — actual pixels live here
};

/// Producer side — creates and writes to shared memory ring.
class ShmRingProducer {
public:
    ShmRingProducer(const char* name, uint32_t num_slots,
                    uint32_t width, uint32_t height, uint32_t channels)
    {
        pixel_size_ = width * height * channels;
        slot_size_  = sizeof(ShmFrameSlot) + pixel_size_;
        size_t total = sizeof(ShmRingHeader) + num_slots * slot_size_;

        // Create shared memory object
        fd_ = shm_open(name, O_CREAT | O_RDWR, 0666);
        if (fd_ < 0) throw std::runtime_error("shm_open failed");

        if (ftruncate(fd_, total) < 0) throw std::runtime_error("ftruncate failed");

        base_ = mmap(nullptr, total, PROT_READ | PROT_WRITE, MAP_SHARED, fd_, 0);
        if (base_ == MAP_FAILED) throw std::runtime_error("mmap failed");

        total_size_ = total;

        // Initialize header
        auto* hdr = header();
        hdr->write_idx.store(0);
        hdr->read_idx.store(0);
        hdr->num_slots = num_slots;
        hdr->slot_size = slot_size_;
        hdr->pixel_data_size = pixel_size_;
        hdr->magic = 0xDEADCAFE;
    }

    ~ShmRingProducer() {
        if (base_ != MAP_FAILED) munmap(base_, total_size_);
        if (fd_ >= 0) close(fd_);
    }

    /// Write a frame into the next slot. Returns false if ring is full.
    bool publish(const FrameMetadata& meta, const uint8_t* pixels) {
        auto* hdr = header();
        uint64_t w = hdr->write_idx.load(std::memory_order_relaxed);
        uint64_t r = hdr->read_idx.load(std::memory_order_acquire);

        if (w - r >= hdr->num_slots) {
            return false;  // consumer too slow — drop frame
        }

        auto* slot = get_slot(w);
        slot->meta = meta;
        std::memcpy(slot->pixel_data, pixels, pixel_size_);

        hdr->write_idx.store(w + 1, std::memory_order_release);
        return true;
    }

private:
    ShmRingHeader* header() { return static_cast<ShmRingHeader*>(base_); }

    ShmFrameSlot* get_slot(uint64_t idx) {
        auto* hdr = header();
        char* data_start = static_cast<char*>(base_) + sizeof(ShmRingHeader);
        return reinterpret_cast<ShmFrameSlot*>(
            data_start + (idx % hdr->num_slots) * slot_size_
        );
    }

    int fd_ = -1;
    void* base_ = MAP_FAILED;
    size_t total_size_ = 0;
    uint32_t pixel_size_ = 0;
    uint32_t slot_size_ = 0;
};

/// Consumer side — opens and reads from shared memory ring (read-only).
class ShmRingConsumer {
public:
    explicit ShmRingConsumer(const char* name) {
        fd_ = shm_open(name, O_RDONLY, 0);
        if (fd_ < 0) throw std::runtime_error("shm_open failed");

        struct stat st;
        fstat(fd_, &st);
        total_size_ = st.st_size;

        base_ = mmap(nullptr, total_size_, PROT_READ, MAP_SHARED, fd_, 0);
        if (base_ == MAP_FAILED) throw std::runtime_error("mmap failed");

        if (header()->magic != 0xDEADCAFE)
            throw std::runtime_error("Invalid shared memory — magic mismatch");
    }

    ~ShmRingConsumer() {
        if (base_ != MAP_FAILED) munmap(base_, total_size_);
        if (fd_ >= 0) close(fd_);
    }

    /// Returns pointer to latest unread frame, or nullptr if no new frame.
    /// Caller must finish reading before calling advance().
    const ShmFrameSlot* peek() const {
        auto* hdr = header();
        uint64_t r = hdr->read_idx.load(std::memory_order_relaxed);
        uint64_t w = hdr->write_idx.load(std::memory_order_acquire);
        if (r >= w) return nullptr;
        return get_slot(r);
    }

    /// Advance read pointer after consuming a frame.
    void advance() {
        auto* hdr = const_cast<ShmRingHeader*>(header());
        uint64_t r = hdr->read_idx.load(std::memory_order_relaxed);
        hdr->read_idx.store(r + 1, std::memory_order_release);
    }

private:
    const ShmRingHeader* header() const {
        return static_cast<const ShmRingHeader*>(base_);
    }

    const ShmFrameSlot* get_slot(uint64_t idx) const {
        auto* hdr = header();
        const char* data_start = static_cast<const char*>(base_)
                                 + sizeof(ShmRingHeader);
        return reinterpret_cast<const ShmFrameSlot*>(
            data_start + (idx % hdr->num_slots) * hdr->slot_size
        );
    }

    int fd_ = -1;
    void* base_ = MAP_FAILED;
    size_t total_size_ = 0;
};

} // namespace drone::video

#include "video/frame.h"
#include "video/spsc_ring.h"
#include <linux/videodev2.h>
#include <sys/ioctl.h>
#include <sys/mman.h>
#include <fcntl.h>
#include <unistd.h>
#include <poll.h>
#include <cstring>
#include <stdexcept>
#include <chrono>
#include <spdlog/spdlog.h>

namespace drone::video {

static constexpr int NUM_BUFFERS = 4;  // V4L2 kernel ring size

struct V4L2Buffer {
    void*  start;
    size_t length;
};

/// Wraps V4L2 device lifecycle: open, configure, capture, close.
class V4L2Camera {
public:
    V4L2Camera(const char* device, uint32_t w, uint32_t h,
               uint32_t pixfmt = V4L2_PIX_FMT_SRGGB10)
        : width_(w), height_(h)
    {
        fd_ = open(device, O_RDWR | O_NONBLOCK);
        if (fd_ < 0)
            throw std::runtime_error(std::string("Cannot open ") + device);

        // 1. Verify capability
        v4l2_capability cap{};
        xioctl(VIDIOC_QUERYCAP, &cap);
        if (!(cap.capabilities & V4L2_CAP_VIDEO_CAPTURE))
            throw std::runtime_error("Device does not support capture");

        // 2. Set format
        v4l2_format fmt{};
        fmt.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
        fmt.fmt.pix.width       = w;
        fmt.fmt.pix.height      = h;
        fmt.fmt.pix.pixelformat = pixfmt;
        fmt.fmt.pix.field       = V4L2_FIELD_NONE;
        xioctl(VIDIOC_S_FMT, &fmt);

        // Driver may adjust — read back actual values
        width_  = fmt.fmt.pix.width;
        height_ = fmt.fmt.pix.height;
        stride_ = fmt.fmt.pix.bytesperline;

        spdlog::info("Camera {}: {}x{} stride={} fmt=0x{:X}",
                     device, width_, height_, stride_, fmt.fmt.pix.pixelformat);

        // 3. Request MMAP buffers
        v4l2_requestbuffers req{};
        req.count  = NUM_BUFFERS;
        req.type   = V4L2_BUF_TYPE_VIDEO_CAPTURE;
        req.memory = V4L2_MEMORY_MMAP;
        xioctl(VIDIOC_REQBUFS, &req);

        // 4. Map each buffer
        for (uint32_t i = 0; i < req.count; ++i) {
            v4l2_buffer buf{};
            buf.type   = V4L2_BUF_TYPE_VIDEO_CAPTURE;
            buf.memory = V4L2_MEMORY_MMAP;
            buf.index  = i;
            xioctl(VIDIOC_QUERYBUF, &buf);

            void* ptr = mmap(nullptr, buf.length, PROT_READ | PROT_WRITE,
                             MAP_SHARED, fd_, buf.m.offset);
            if (ptr == MAP_FAILED)
                throw std::runtime_error("mmap failed for buffer " + std::to_string(i));

            buffers_[i] = {ptr, buf.length};
        }

        // 5. Enqueue all buffers
        for (uint32_t i = 0; i < req.count; ++i) {
            v4l2_buffer buf{};
            buf.type   = V4L2_BUF_TYPE_VIDEO_CAPTURE;
            buf.memory = V4L2_MEMORY_MMAP;
            buf.index  = i;
            xioctl(VIDIOC_QBUF, &buf);
        }

        // 6. Start streaming
        int type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
        xioctl(VIDIOC_STREAMON, &type);
    }

    ~V4L2Camera() {
        int type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
        ioctl(fd_, VIDIOC_STREAMOFF, &type);
        for (auto& b : buffers_) {
            if (b.start) munmap(b.start, b.length);
        }
        if (fd_ >= 0) close(fd_);
    }

    struct CapturedFrame {
        const uint8_t* data;
        size_t         size;
        uint64_t       timestamp_ns;
        uint32_t       sequence;
        uint32_t       buffer_index;  // needed for re-enqueue
    };

    /// Block until a frame is ready (poll + dequeue). Returns the frame.
    CapturedFrame capture() {
        // Poll with 1-second timeout (detect hardware failure)
        pollfd pfd{fd_, POLLIN, 0};
        int ret = poll(&pfd, 1, 1000);
        if (ret <= 0)
            throw std::runtime_error("Camera poll timeout — hardware failure?");

        v4l2_buffer buf{};
        buf.type   = V4L2_BUF_TYPE_VIDEO_CAPTURE;
        buf.memory = V4L2_MEMORY_MMAP;
        xioctl(VIDIOC_DQBUF, &buf);

        // Convert kernel timestamp to CLOCK_MONOTONIC nanoseconds
        uint64_t ts = buf.timestamp.tv_sec * 1'000'000'000ULL
                    + buf.timestamp.tv_usec * 1'000ULL;

        return {
            static_cast<const uint8_t*>(buffers_[buf.index].start),
            buf.bytesused,
            ts,
            buf.sequence,
            buf.index
        };
    }

    /// Re-enqueue a buffer after processing.
    void release(uint32_t buffer_index) {
        v4l2_buffer buf{};
        buf.type   = V4L2_BUF_TYPE_VIDEO_CAPTURE;
        buf.memory = V4L2_MEMORY_MMAP;
        buf.index  = buffer_index;
        xioctl(VIDIOC_QBUF, &buf);
    }

    uint32_t width()  const { return width_; }
    uint32_t height() const { return height_; }
    uint32_t stride() const { return stride_; }

private:
    void xioctl(unsigned long request, void* arg) {
        int ret;
        do { ret = ioctl(fd_, request, arg); }
        while (ret == -1 && errno == EINTR);  // retry on signal
        if (ret < 0)
            throw std::runtime_error("ioctl failed: " + std::to_string(errno));
    }

    int fd_ = -1;
    uint32_t width_, height_, stride_;
    V4L2Buffer buffers_[NUM_BUFFERS]{};
};

} // namespace drone::video

#include <thread>
#include <atomic>

namespace drone::video {

/// This is the thread entry point. Runs until stop_flag is set.
void mission_cam_thread(
    SPSCRing<CapturedFrame, 4>& output_queue,   // → consumed by ISP thread
    std::atomic<bool>& stop_flag,
    const char* device = "/dev/video0")
{
    try {
        V4L2Camera cam(device, 1920, 1080, V4L2_PIX_FMT_SRGGB10);
        spdlog::info("Mission camera started: {}x{}", cam.width(), cam.height());

        uint64_t frame_count = 0;
        uint64_t drop_count  = 0;

        while (!stop_flag.load(std::memory_order_relaxed)) {
            auto frame = cam.capture();  // blocks on poll()

            if (!output_queue.try_push(frame)) {
                // ISP thread can't keep up — drop this frame
                ++drop_count;
                if (drop_count % 100 == 0) {
                    spdlog::warn("Mission cam: dropped {} frames (ISP too slow)",
                                 drop_count);
                }
            }

            cam.release(frame.buffer_index);  // re-enqueue to kernel
            ++frame_count;
        }

        spdlog::info("Mission cam stopped. {} frames captured, {} dropped",
                     frame_count, drop_count);
    }
    catch (const std::exception& e) {
        spdlog::error("Mission cam thread FATAL: {}", e.what());
        // Process-level error handler will decide: restart thread or whole process
    }
}

} // namespace drone::video

// file: src/video/stereo_cam_thread.cpp
#include "video/frame.h"
#include "video/spsc_ring.h"
#include <spdlog/spdlog.h>
#include <atomic>

namespace drone::video {

struct StereoFrame {
    FrameMetadata meta_left;
    FrameMetadata meta_right;
    const uint8_t* left_data;
    const uint8_t* right_data;
    size_t data_size;
    uint32_t left_buf_idx;
    uint32_t right_buf_idx;
};

/// For hardware-synced stereo: both cameras trigger simultaneously.
/// We open two V4L2 devices and dequeue from both per iteration.
void stereo_cam_thread(
    SPSCRing<StereoFrame, 4>& output_queue,  // → consumed by ISP thread
    std::atomic<bool>& stop_flag,
    const char* left_device  = "/dev/video2",
    const char* right_device = "/dev/video3")
{
    try {
        V4L2Camera cam_left(left_device,  640, 480, V4L2_PIX_FMT_GREY);
        V4L2Camera cam_right(right_device, 640, 480, V4L2_PIX_FMT_GREY);

        spdlog::info("Stereo cameras started: {}x{} (L) + {}x{} (R)",
                     cam_left.width(), cam_left.height(),
                     cam_right.width(), cam_right.height());

        while (!stop_flag.load(std::memory_order_relaxed)) {
            // With hardware sync, both frames are ready at the same time
            auto left  = cam_left.capture();
            auto right = cam_right.capture();

            // Verify sync — timestamps should be within 1ms
            int64_t dt = static_cast<int64_t>(left.timestamp_ns)
                       - static_cast<int64_t>(right.timestamp_ns);
            if (std::abs(dt) > 1'000'000) {  // > 1 ms
                spdlog::warn("Stereo sync drift: {} µs", dt / 1000);
                // Could try to re-sync or drop this pair
            }

            StereoFrame sf{};
            sf.meta_left = {left.timestamp_ns, left.sequence,
                           CameraId::STEREO_L, PixelFormat::GRAY8,
                           cam_left.width(), cam_left.height(),
                           cam_left.stride(), 0, 0.0f, 0.0f};

/// ISP configuration — loaded from YAML config file at startup.
struct ISPConfig {
    // Lens calibration (from cv::calibrateCamera)
    cv::Mat camera_matrix;       // 3×3 intrinsic matrix
    cv::Mat dist_coeffs;         // distortion coefficients (k1,k2,p1,p2,k3)
    cv::Mat rectify_map1;        // precomputed for cv::remap (faster than undistort)
    cv::Mat rectify_map2;

    // White balance gains
    float wb_red   = 1.0f;
    float wb_green = 1.0f;
    float wb_blue  = 1.0f;

    // Denoise strength (0 = off, 10 = aggressive)
    int denoise_strength = 3;

    // Whether to produce NV12 for encoder (vs RGB)
    bool produce_nv12 = true;
};

/// Precompute undistortion remap tables (call once at startup).
void init_undistort_maps(ISPConfig& cfg, int width, int height) {
    cv::Mat new_camera_matrix = cv::getOptimalNewCameraMatrix(
        cfg.camera_matrix, cfg.dist_coeffs,
        cv::Size(width, height), 0.0  // alpha=0: crop to valid pixels only
    );
    cv::initUndistortRectifyMap(
        cfg.camera_matrix, cfg.dist_coeffs,
        cv::Mat(),  // R (identity — no stereo rectification for mission cam)
        new_camera_matrix,
        cv::Size(width, height),
        CV_16SC2,   // fixed-point for speed
        cfg.rectify_map1, cfg.rectify_map2
    );
    spdlog::info("ISP: undistortion maps precomputed for {}x{}", width, height);
}

/// Apply ISP pipeline to a single mission camera frame.
cv::Mat process_mission_frame(const uint8_t* raw_data,
                               uint32_t width, uint32_t height,
                               uint32_t stride,
                               const ISPConfig& cfg)
{
    // 1. Debayer (Bayer RGGB → RGB)
    cv::Mat bayer(height, width, CV_8UC1, const_cast<uint8_t*>(raw_data), stride);
    cv::Mat rgb;
    cv::cvtColor(bayer, rgb, cv::COLOR_BayerRG2RGB);

    // 2. White balance
    if (cfg.wb_red != 1.0f || cfg.wb_blue != 1.0f) {
        std::vector<cv::Mat> channels;
        cv::split(rgb, channels);
        channels[0] *= cfg.wb_blue;   // B
        channels[2] *= cfg.wb_red;    // R
        cv::merge(channels, rgb);
    }

    // 3. Denoise (bilateral filter — preserves edges)
    if (cfg.denoise_strength > 0) {
        cv::Mat denoised;
        cv::bilateralFilter(rgb, denoised,
                           cfg.denoise_strength,   // d (neighborhood diameter)
                           cfg.denoise_strength * 10.0,  // sigmaColor
                           cfg.denoise_strength * 10.0); // sigmaSpace
        rgb = denoised;
    }

    // 4. Lens undistortion (using precomputed remap — ~2ms vs ~8ms for undistort)
    cv::Mat undistorted;
    cv::remap(rgb, undistorted, cfg.rectify_map1, cfg.rectify_map2,
              cv::INTER_LINEAR);

    return undistorted;
}

struct EncoderFrame {
    cv::Mat image;          // NV12 or RGB
    FrameMetadata meta;
};

/// ISP thread entry point — consumes from both camera queues,
/// distributes to all four outputs.
void isp_thread(
    // Inputs
    SPSCRing<V4L2Camera::CapturedFrame, 4>& mission_queue,
    SPSCRing<StereoFrame, 4>& stereo_queue,
    // Outputs — inter-process
    ShmRingProducer& shm_perception,  // → Perception process
    ShmRingProducer& shm_slam,        // → SLAM process
    // Outputs — intra-process
    SPSCRing<EncoderFrame, 4>& encoder_queue,  // → Thread 4
    SPSCRing<EncoderFrame, 4>& recorder_queue, // → Thread 5
    // Control
    std::atomic<bool>& stop_flag,
    const ISPConfig& cfg)
{
    spdlog::info("ISP thread started");

    while (!stop_flag.load(std::memory_order_relaxed)) {

        // ── Process mission camera frames ──
        if (auto frame = mission_queue.try_pop()) {
            auto rgb = process_mission_frame(
                frame->data, 1920, 1080, 1920, cfg);

            // Output 1: To Perception (via shared memory)
            FrameMetadata meta{
                frame->timestamp_ns, frame->sequence,
                CameraId::MISSION, PixelFormat::RGB24,
                static_cast<uint32_t>(rgb.cols),
                static_cast<uint32_t>(rgb.rows),
                static_cast<uint32_t>(rgb.step),
                0, 0.0f, 0.0f
            };
            shm_perception.publish(meta, rgb.data);

            // Output 3: To encoder
            EncoderFrame ef;
            if (cfg.produce_nv12) {
                cv::cvtColor(rgb, ef.image, cv::COLOR_RGB2YUV_I420);
            } else {
                ef.image = rgb;
            }
            ef.meta = meta;
            encoder_queue.try_push(ef);

            // Output 4: To SSD recorder
            EncoderFrame rf;
            rf.image = rgb.clone();  // recorder may be slower — needs own copy
            rf.meta  = meta;
            recorder_queue.try_push(rf);
        }

        // ── Process stereo frames ──
        if (auto stereo = stereo_queue.try_pop()) {
            // For SLAM: rectify both images using stereo calibration
            // (Stereo rectification aligns epipolar lines — required for
            //  dense stereo matching and VIO)
            // Simplified: just undistort each. Full stereo rectification
            // needs R, T from stereo calibration.

            // Output 2: To SLAM (left image via shared memory)
            // SLAM typically only needs the left image + IMU data
            FrameMetadata meta_l = stereo->meta_left;
            meta_l.format = PixelFormat::GRAY8;
            shm_slam.publish(meta_l, stereo->left_data);
        }

        // Prevent busy-spin if both queues are empty
        // (In practice, at 30fps each camera produces a frame every 33ms
        //  so we only spin briefly between frames)
        if (mission_queue.available() == 0 && stereo_queue.available() == 0) {
            std::this_thread::sleep_for(std::chrono::microseconds(100));
        }
    }
    spdlog::info("ISP thread stopped");
}

/// Simple UDP sender for RTP packets
class RTPSender {
public:
    RTPSender(const char* dest_ip, uint16_t dest_port)
    {
        sock_ = socket(AF_INET, SOCK_DGRAM | SOCK_NONBLOCK, 0);
        if (sock_ < 0) throw std::runtime_error("socket() failed");

        memset(&dest_, 0, sizeof(dest_));
        dest_.sin_family = AF_INET;
        dest_.sin_port   = htons(dest_port);
        inet_pton(AF_INET, dest_ip, &dest_.sin_addr);

        // Increase send buffer to handle bursts
        int buf_size = 2 * 1024 * 1024;  // 2MB
        setsockopt(sock_, SOL_SOCKET, SO_SNDBUF, &buf_size, sizeof(buf_size));
    }

    ~RTPSender() { if (sock_ >= 0) close(sock_); }

    void send_nal(const uint8_t* nal_data, size_t nal_size, uint32_t rtp_ts) {
        // Fragment NAL into RTP packets (max 1400 bytes payload)
        constexpr size_t MAX_PAYLOAD = 1400;

        if (nal_size <= MAX_PAYLOAD) {
            // Single NAL unit packet
            send_rtp_packet(nal_data, nal_size, rtp_ts, true);
        } else {
            // Fragmentation Unit (FU) — RFC 7798 Section 4.4.3
            size_t offset = 2;  // skip NAL header (2 bytes for H.265)
            bool first = true;
            while (offset < nal_size) {
                size_t chunk = std::min(MAX_PAYLOAD - 3, nal_size - offset);
                bool last = (offset + chunk >= nal_size);

                // Build FU header: FU indicator + FU header
                uint8_t fu_packet[MAX_PAYLOAD + 3];
                // ... construct FU packet per RFC 7798 ...

                send_rtp_packet(fu_packet, chunk + 3, rtp_ts, last);
                offset += chunk;
                first = false;
            }
        }
    }

private:
    void send_rtp_packet(const uint8_t* payload, size_t len,
                         uint32_t timestamp, bool marker) {
        uint8_t packet[1500];
        RTPHeader hdr{};
        hdr.version_flags = 0x80;
        hdr.payload_type  = 96 | (marker ? 0x80 : 0);
        hdr.sequence      = htons(seq_++);
        hdr.timestamp     = htonl(timestamp);
        hdr.ssrc          = htonl(0x44524F4E);  // "DRON"

        memcpy(packet, &hdr, sizeof(hdr));
        memcpy(packet + sizeof(hdr), payload, len);

        sendto(sock_, packet, sizeof(hdr) + len, 0,
               (sockaddr*)&dest_, sizeof(dest_));
    }

    int sock_ = -1;
    sockaddr_in dest_{};
    uint16_t seq_ = 0;
};

/// Encoder thread entry point
void encoder_thread(
    SPSCRing<EncoderFrame, 4>& input_queue,   // from ISP thread
    std::atomic<bool>& stop_flag,
    std::atomic<uint32_t>& target_bitrate_kbps,  // set by adaptive controller
    const char* gcs_ip   = "192.168.1.100",
    uint16_t   gcs_port  = 5600)
{
    H265Encoder encoder(H265Encoder::Config{
        1920, 1080, 30, 8000, 30, true  // NVENC
    });
    RTPSender sender(gcs_ip, gcs_port);

    auto last_bitrate_check = std::chrono::steady_clock::now();
    uint64_t frames_encoded = 0;

    while (!stop_flag.load(std::memory_order_relaxed)) {
        auto frame_opt = input_queue.try_pop();
        if (!frame_opt) {
            std::this_thread::sleep_for(std::chrono::microseconds(500));
            continue;
        }

        auto& ef = *frame_opt;

        // Encode frame
        auto nals = encoder.encode(ef.image.data);

        // RTP timestamp: 90kHz clock = (frame_ns / 1000000) * 90
        uint32_t rtp_ts = static_cast<uint32_t>(
            (ef.meta.timestamp_ns / 1'000'000) * 90);

        // Send each NAL unit
        for (auto& nal : nals) {
            sender.send_nal(nal.data, nal.size, rtp_ts);
        }

        ++frames_encoded;

        // Check if bitrate needs adjusting (every 500ms)
        auto now = std::chrono::steady_clock::now();
        if (now - last_bitrate_check > std::chrono::milliseconds(500)) {
            uint32_t target = target_bitrate_kbps.load(std::memory_order_relaxed);
            encoder.set_bitrate(target);
            last_bitrate_check = now;
        }
    }

    spdlog::info("Encoder stopped. {} frames encoded", frames_encoded);
}

ShmRingConsumer* connect_to_video(const char* shm_name, int max_retries = 50) {
    for (int i = 0; i < max_retries; ++i) {
        try {
            return new ShmRingConsumer(shm_name);
        } catch (const std::exception& e) {
            spdlog::warn("Waiting for {} (attempt {}/{}): {}",
                         shm_name, i+1, max_retries, e.what());
            std::this_thread::sleep_for(
                std::chrono::milliseconds(100 * std::min(i + 1, 10)));
        }
    }
    throw std::runtime_error(std::string("Cannot connect to ") + shm_name);
}

// Simplified pub-sub within data bus
// Real implementation uses lock-free MPSC (multiple-producer single-consumer)
// or the shared data bus abstraction

template <typename T>
class Topic {
public:
    using Callback = std::function<void(const T&)>;

    void subscribe(Callback cb) {
        std::lock_guard<std::mutex> lock(mu_);
        subscribers_.push_back(std::move(cb));
    }

    void publish(const T& msg) {
        std::lock_guard<std::mutex> lock(mu_);
        for (auto& cb : subscribers_) {
            cb(msg);  // In production, push to subscriber's queue instead
        }
    }

private:
    std::mutex mu_;
    std::vector<Callback> subscribers_;
};

// Why this is lock-free:
//
// Thread A (producer):                Thread B (consumer):
//   data[idx] = frame;                  if (write > read) {
//   write.store(idx+1, RELEASE);          frame = data[read];
//       ↑                                  read.store(read+1, RELEASE);
//       │                                }
//       └── RELEASE ensures the data        ↑
//           write is visible BEFORE         │
//           the index increment.            └── ACQUIRE on write ensures
//                                               we see the data that was
//                                               written before the index.
//
// No mutex needed because:
// 1. Only ONE thread writes to write_idx (producer)
// 2. Only ONE thread writes to read_idx (consumer)
// 3. Memory ordering (acquire/release) ensures visibility
//
// This is O(1) — no CAS loop, no retry, no contention.
// Worst case: consumer sees stale write_idx and misses one frame
// (it catches up on the next iteration — totally fine for perception).

// CRTP base: the filter logic is generic, the motion model is specialized
template <typename Derived, int StateSize>
class KalmanFilterBase {
public:
    using StateVec = Eigen::Matrix<float, StateSize, 1>;
    using StateMat = Eigen::Matrix<float, StateSize, StateSize>;

    void predict(float dt) {
        // Calls the derived class's motion model — resolved at COMPILE TIME
        static_cast<Derived*>(this)->predict_impl(dt);
        state_ = F_ * state_;
        P_ = F_ * P_ * F_.transpose() + Q_;
    }

protected:
    StateVec state_;
    StateMat F_, P_, Q_;
};

/// UDP receiver for LiDAR packets.
/// Most LiDARs send packets to a fixed UDP port (e.g., 2368 for Velodyne).
class LiDARReceiver {
public:
    explicit LiDARReceiver(uint16_t port = 2368) {
        sock_ = socket(AF_INET, SOCK_DGRAM, 0);
        if (sock_ < 0) throw std::runtime_error("LiDAR socket failed");

        sockaddr_in addr{};
        addr.sin_family = AF_INET;
        addr.sin_port = htons(port);
        addr.sin_addr.s_addr = INADDR_ANY;

        if (bind(sock_, (sockaddr*)&addr, sizeof(addr)) < 0)
            throw std::runtime_error("LiDAR bind failed");

        // Set receive buffer large enough for burst
        int buf = 4 * 1024 * 1024;
        setsockopt(sock_, SOL_SOCKET, SO_RCVBUF, &buf, sizeof(buf));
    }

    ~LiDARReceiver() { if (sock_ >= 0) close(sock_); }

    ssize_t receive(uint8_t* buffer, size_t max_size) {
        return recv(sock_, buffer, max_size, 0);
    }

private:
    int sock_ = -1;
};

/// Simplified ground removal: remove points below a height threshold
/// relative to the drone's current altitude.
/// Production: use RANSAC plane fitting for robustness.
void remove_ground(PointCloud& cloud, float ground_height = -0.3f) {
    auto it = std::remove_if(cloud.points.begin(), cloud.points.end(),
        [ground_height](const Point3D& p) { return p.z < ground_height; });
    cloud.points.erase(it, cloud.points.end());
}

/// Voxel grid downsample — keep one point per voxel cell.
/// This reduces 100K+ points to ~20K, making clustering tractable.
void voxel_downsample(PointCloud& cloud, float voxel_size = 0.1f) {
    // Simple hash-based approach (production: use PCL VoxelGrid)
    struct VoxelKey {
        int32_t x, y, z;
        bool operator==(const VoxelKey& o) const {
            return x == o.x && y == o.y && z == o.z;
        }
    };
    struct VoxelHash {
        size_t operator()(const VoxelKey& k) const {
            return std::hash<int32_t>()(k.x) ^
                   (std::hash<int32_t>()(k.y) << 16) ^
                   (std::hash<int32_t>()(k.z) << 32);
        }
    };

    std::unordered_map<VoxelKey, Point3D, VoxelHash> grid;
    for (auto& p : cloud.points) {
        VoxelKey key{
            static_cast<int32_t>(std::floor(p.x / voxel_size)),
            static_cast<int32_t>(std::floor(p.y / voxel_size)),
            static_cast<int32_t>(std::floor(p.z / voxel_size))
        };
        grid.emplace(key, p);  // first point in each voxel wins
    }

    cloud.points.clear();
    cloud.points.reserve(grid.size());
    for (auto& [key, point] : grid) {
        cloud.points.push_back(point);
    }
}

/// RAII wrapper for TensorRT inference engine
class TensorRTEngine {
public:
    explicit TensorRTEngine(const std::string& engine_path) {
        // Load serialized engine from file
        std::ifstream file(engine_path, std::ios::binary);
        if (!file) throw std::runtime_error("Cannot open engine: " + engine_path);

        file.seekg(0, std::ios::end);
        size_t size = file.tellg();
        file.seekg(0, std::ios::beg);
        std::vector<char> engine_data(size);
        file.read(engine_data.data(), size);

        // Create runtime + engine + context
        runtime_.reset(nvinfer1::createInferRuntime(logger_));
        engine_.reset(runtime_->deserializeCudaEngine(
            engine_data.data(), size));
        context_.reset(engine_->createExecutionContext());

        if (!context_)
            throw std::runtime_error("Failed to create TensorRT context");

        // Allocate GPU I/O buffers
        cudaMalloc(&input_gpu_,  INPUT_SIZE);
        cudaMalloc(&output_gpu_, OUTPUT_SIZE);
        cudaStreamCreate(&stream_);

        spdlog::info("TensorRT engine loaded: {}", engine_path);
    }

    ~TensorRTEngine() {
        cudaFree(input_gpu_);
        cudaFree(output_gpu_);
        cudaStreamDestroy(stream_);
    }

    // Non-copyable, movable
    TensorRTEngine(const TensorRTEngine&) = delete;
    TensorRTEngine& operator=(const TensorRTEngine&) = delete;

    /// Run inference on a preprocessed image (CHW, FP32, normalized)
    /// Returns raw output tensor — caller must post-process
    std::vector<float> infer(const float* input_data) {
        // Copy input to GPU
        cudaMemcpyAsync(input_gpu_, input_data, INPUT_SIZE,
                        cudaMemcpyHostToDevice, stream_);

        // Execute
        void* bindings[] = {input_gpu_, output_gpu_};
        context_->enqueueV2(bindings, stream_, nullptr);

        // Copy output from GPU
        std::vector<float> output(OUTPUT_ELEMENTS);
        cudaMemcpyAsync(output.data(), output_gpu_, OUTPUT_SIZE,
                        cudaMemcpyDeviceToHost, stream_);

        cudaStreamSynchronize(stream_);
        return output;
    }

private:
    static constexpr int INPUT_C = 3, INPUT_H = 640, INPUT_W = 640;
    static constexpr size_t INPUT_SIZE = INPUT_C * INPUT_H * INPUT_W * sizeof(float);
    static constexpr int OUTPUT_ELEMENTS = 8400 * 85;  // YOLOv8 output
    static constexpr size_t OUTPUT_SIZE = OUTPUT_ELEMENTS * sizeof(float);

    TRTLogger logger_;
    // unique_ptr with custom deleter for TensorRT objects
    struct TRTDeleter {
        template <typename T>
        void operator()(T* p) { if (p) p->destroy(); }
    };
    std::unique_ptr<nvinfer1::IRuntime, TRTDeleter> runtime_;
    std::unique_ptr<nvinfer1::ICudaEngine, TRTDeleter> engine_;
    std::unique_ptr<nvinfer1::IExecutionContext, TRTDeleter> context_;

    void* input_gpu_  = nullptr;
    void* output_gpu_ = nullptr;
    cudaStream_t stream_ = nullptr;
};

// Pre-processing: resize + normalize + HWC→CHW

/// Letterbox resize — maintains aspect ratio, pads with gray
cv::Mat letterbox(const cv::Mat& src, int target_w, int target_h,
                  float& scale, int& pad_x, int& pad_y) {
    float r = std::min(float(target_w) / src.cols,
                       float(target_h) / src.rows);
    scale = r;
    int new_w = static_cast<int>(src.cols * r);
    int new_h = static_cast<int>(src.rows * r);
    pad_x = (target_w - new_w) / 2;
    pad_y = (target_h - new_h) / 2;

    cv::Mat resized;
    cv::resize(src, resized, cv::Size(new_w, new_h));
    cv::Mat padded(target_h, target_w, CV_8UC3, cv::Scalar(114, 114, 114));
    resized.copyTo(padded(cv::Rect(pad_x, pad_y, new_w, new_h)));
    return padded;
}

/// HWC uint8 → CHW float32 normalized [0,1]
std::vector<float> preprocess(const cv::Mat& img) {
    std::vector<float> blob(3 * img.rows * img.cols);
    for (int c = 0; c < 3; ++c) {
        for (int y = 0; y < img.rows; ++y) {
            for (int x = 0; x < img.cols; ++x) {
                blob[c * img.rows * img.cols + y * img.cols + x] =
                    img.at<cv::Vec3b>(y, x)[c] / 255.0f;
            }
        }
    }
    return blob;
}

/// Non-Maximum Suppression — remove overlapping detections
float iou(const Detection2D& a, const Detection2D& b) {
    float x1 = std::max(a.x, b.x);
    float y1 = std::max(a.y, b.y);
    float x2 = std::min(a.x + a.w, b.x + b.w);
    float y2 = std::min(a.y + a.h, b.y + b.h);
    float inter = std::max(0.f, x2 - x1) * std::max(0.f, y2 - y1);
    float un = a.w * a.h + b.w * b.h - inter;
    return un > 0 ? inter / un : 0;
}

std::vector<Detection2D> nms(std::vector<Detection2D>& dets,
                              float iou_threshold = 0.45f) {
    // Sort by confidence (descending)
    std::sort(dets.begin(), dets.end(),
              [](const Detection2D& a, const Detection2D& b) {
                  return a.confidence > b.confidence;
              });

    std::vector<Detection2D> result;
    std::vector<bool> suppressed(dets.size(), false);

    for (size_t i = 0; i < dets.size(); ++i) {
        if (suppressed[i]) continue;
        result.push_back(dets[i]);
        for (size_t j = i + 1; j < dets.size(); ++j) {
            if (!suppressed[j] && iou(dets[i], dets[j]) > iou_threshold) {
                suppressed[j] = true;
            }
        }
    }
    return result;
}

// file: src/perception/kalman_tracker.h
#pragma once
#include <Eigen/Dense>

namespace drone::perception {

/// 2D Kalman filter for a single tracked object.
/// State: [cx, cy, w, h, vx, vy, vw, vh] — center, size, velocities
/// Measurement: [cx, cy, w, h] — from bounding box detection
class KalmanBoxTracker {
public:
    static constexpr int STATE_DIM = 8;
    static constexpr int MEAS_DIM  = 4;

    using StateVec = Eigen::Matrix<float, STATE_DIM, 1>;
    using StateMat = Eigen::Matrix<float, STATE_DIM, STATE_DIM>;
    using MeasVec  = Eigen::Matrix<float, MEAS_DIM, 1>;
    using MeasMat  = Eigen::Matrix<float, MEAS_DIM, STATE_DIM>;

    explicit KalmanBoxTracker(const Detection2D& initial_det, uint32_t id)
        : track_id(id)
    {
        // Initialize state from first detection
        x_ << initial_det.x + initial_det.w/2,  // cx
               initial_det.y + initial_det.h/2,  // cy
               initial_det.w,                     // w
               initial_det.h,                     // h
               0, 0, 0, 0;                        // velocities = 0

        // State transition: constant velocity model
        F_ = StateMat::Identity();
        // F(0,4)=dt, F(1,5)=dt, etc. — set in predict()

        // Measurement matrix: we observe [cx, cy, w, h]
        H_ = MeasMat::Zero();
        H_(0,0) = H_(1,1) = H_(2,2) = H_(3,3) = 1.0f;

        // Covariances (tuned for pixel coordinates)
        P_ = StateMat::Identity() * 10.0f;    // initial uncertainty
        Q_ = StateMat::Identity() * 0.01f;    // process noise
        Q_(4,4) = Q_(5,5) = Q_(6,6) = Q_(7,7) = 0.1f;  // velocity noise higher
        R_ = Eigen::Matrix<float, MEAS_DIM, MEAS_DIM>::Identity() * 1.0f;
    }

    /// Predict state forward by dt seconds
    void predict(float dt = 1.0f/30.0f) {
        F_ = StateMat::Identity();
        F_(0, 4) = dt;  // cx += vx * dt
        F_(1, 5) = dt;  // cy += vy * dt
        F_(2, 6) = dt;  // w  += vw * dt
        F_(3, 7) = dt;  // h  += vh * dt

        x_ = F_ * x_;
        P_ = F_ * P_ * F_.transpose() + Q_;
        age++;
    }

    /// Update state with a matched detection
    void update(const Detection2D& det) {
        MeasVec z;
        z << det.x + det.w/2, det.y + det.h/2, det.w, det.h;

        // Kalman gain
        auto S = H_ * P_ * H_.transpose() + R_;
        auto K = P_ * H_.transpose() * S.inverse();

        // State update
        x_ = x_ + K * (z - H_ * x_);
        P_ = (StateMat::Identity() - K * H_) * P_;

        hits++;
        consecutive_misses = 0;
        confidence = det.confidence;
    }

    /// Get current bounding box prediction
    Detection2D predicted_bbox() const {
        return {x_(0) - x_(2)/2, x_(1) - x_(3)/2,
                x_(2), x_(3), confidence, class_id, 0, 0};
    }

    /// Get estimated velocity (pixels/sec)
    Eigen::Vector2f velocity() const { return {x_(4), x_(5)}; }

    uint32_t track_id;
    ObjectClass class_id = ObjectClass::UNKNOWN;
    float confidence = 0;
    uint32_t age = 0;
    uint32_t hits = 0;
    uint32_t consecutive_misses = 0;

    bool is_confirmed() const { return hits >= 3; }
    bool is_stale() const { return consecutive_misses > 10; }

private:
    StateVec x_;              // state vector
    StateMat F_, P_, Q_;      // transition, covariance, process noise
    MeasMat H_;               // measurement matrix
    Eigen::Matrix<float, MEAS_DIM, MEAS_DIM> R_;  // measurement noise
};

} // namespace drone::perception

// file: src/perception/tracker_thread.cpp

void tracker_thread(
    SPSCRing<Detection2DList, 4>& input_queue,      // from inference thread
    SPSCRing<TrackedObjectList, 4>& output_queue,    // to fusion thread
    std::atomic<bool>& stop_flag)
{
    std::vector<std::unique_ptr<KalmanBoxTracker>> tracks;
    uint32_t next_id = 1;

    spdlog::info("Tracker thread started");

    while (!stop_flag.load(std::memory_order_relaxed)) {
        auto dets_opt = input_queue.try_pop();
        if (!dets_opt) {
            std::this_thread::sleep_for(std::chrono::microseconds(100));
            continue;
        }

        auto& dets = *dets_opt;
        float dt = 1.0f / 30.0f;  // TODO: compute actual dt from timestamps

        // Step 1: Predict all tracks forward
        for (auto& t : tracks) t->predict(dt);

        // Step 2: Build cost matrix (1 - IoU, so lower = better match)
        size_t num_dets = dets.detections.size();
        size_t num_tracks = tracks.size();
        std::vector<std::vector<double>> cost(num_dets,
            std::vector<double>(num_tracks, 1.0));
        for (size_t i = 0; i < num_dets; ++i) {
            for (size_t j = 0; j < num_tracks; ++j) {
                auto pred = tracks[j]->predicted_bbox();
                cost[i][j] = 1.0 - iou(dets.detections[i], pred);
            }
        }

        // Step 3: Hungarian assignment
        auto result = HungarianSolver::solve(cost, 0.7);

        // Step 4a: Update matched tracks
        for (size_t i = 0; i < result.assignment.size(); ++i) {
            int j = result.assignment[i];
            if (j >= 0) {
                tracks[j]->update(dets.detections[i]);
                tracks[j]->class_id = dets.detections[i].class_id;
            }
        }

        // Step 4b: Create new tracks for unmatched detections
        for (int i : result.unmatched_rows) {
            tracks.push_back(
                std::make_unique<KalmanBoxTracker>(
                    dets.detections[i], next_id++));
        }

        // Step 4c: Mark unmatched tracks
        for (int j : result.unmatched_cols) {
            tracks[j]->consecutive_misses++;
        }

        // Step 4d: Remove stale tracks
        tracks.erase(
            std::remove_if(tracks.begin(), tracks.end(),
                [](const auto& t) { return t->is_stale(); }),
            tracks.end());

        // Build output: only confirmed tracks
        TrackedObjectList output;
        output.timestamp_ns = dets.timestamp_ns;
        output.frame_sequence = dets.frame_sequence;
        for (auto& t : tracks) {
            if (t->is_confirmed()) {
                TrackedObject obj;
                obj.track_id = t->track_id;
                obj.class_id = t->class_id;
                obj.confidence = t->confidence;
                obj.position_2d = {t->predicted_bbox().x + t->predicted_bbox().w/2,
                                   t->predicted_bbox().y + t->predicted_bbox().h/2};
                obj.velocity_2d = t->velocity();
                obj.age = t->age;
                obj.hits = t->hits;
                obj.misses = t->consecutive_misses;
                obj.state = TrackedObject::State::CONFIRMED;
                obj.timestamp_ns = dets.timestamp_ns;
                output.objects.push_back(obj);
            }
        }

        output_queue.try_push(std::move(output));
    }
    spdlog::info("Tracker thread stopped");
}

#include "perception/types.h"
#include <Eigen/Dense>
#include <opencv2/calib3d.hpp>

namespace drone::perception {

/// Camera-LiDAR calibration (from extrinsic calibration procedure)
struct CalibrationData {
    Eigen::Matrix3f K;          // camera intrinsic matrix (3×3)
    Eigen::Matrix3f R_lidar2cam;  // rotation: LiDAR → camera frame
    Eigen::Vector3f t_lidar2cam;  // translation: LiDAR → camera frame
    uint32_t image_width;
    uint32_t image_height;
};

/// Project a 3D point (LiDAR frame) to 2D pixel coordinates (camera frame)
Eigen::Vector2f project_lidar_to_camera(
    const Eigen::Vector3f& point_lidar,
    const CalibrationData& calib)
{
    // Transform to camera frame
    Eigen::Vector3f p_cam = calib.R_lidar2cam * point_lidar + calib.t_lidar2cam;

    // Check if point is in front of camera
    if (p_cam.z() <= 0) return {-1, -1};  // behind camera

    // Project to image plane
    Eigen::Vector3f p_img = calib.K * p_cam;
    float u = p_img.x() / p_img.z();
    float v = p_img.y() / p_img.z();

    // Check bounds
    if (u < 0 || u >= calib.image_width || v < 0 || v >= calib.image_height)
        return {-1, -1};  // outside image

    return {u, v};
}

/// Check if a projected point falls inside a bounding box
bool point_in_bbox(const Eigen::Vector2f& pt, const Detection2D& bbox) {
    return pt.x() >= bbox.x && pt.x() <= bbox.x + bbox.w &&
           pt.y() >= bbox.y && pt.y() <= bbox.y + bbox.h;
}

} // namespace drone::perception

/// Sensor fusion thread entry point
void fusion_thread(
    // Inputs
    SPSCRing<TrackedObjectList, 4>& track_input,        // from tracker
    SPSCRing<std::vector<LiDARCluster>, 4>& lidar_input,  // from LiDAR thread
    SPSCRing<RadarDetectionList, 4>& radar_input,       // from radar thread
    // Output
    // In production: publish to shared data bus
    std::atomic<bool>& stop_flag,
    const CalibrationData& calib)
{
    // Latest sensor data (updated asynchronously)
    std::vector<LiDARCluster> latest_lidar;
    RadarDetectionList latest_radar;

    spdlog::info("Fusion thread started");

    while (!stop_flag.load(std::memory_order_relaxed)) {
        // Drain sensor queues (non-blocking — take latest available)
        if (auto lc = lidar_input.try_pop()) latest_lidar = std::move(*lc);
        if (auto rd = radar_input.try_pop()) latest_radar = std::move(*rd);

        // Wait for tracked objects (this is the primary cadence — 30Hz)
        auto tracks_opt = track_input.try_pop();
        if (!tracks_opt) {
            std::this_thread::sleep_for(std::chrono::microseconds(100));
            continue;
        }

        auto& tracks = *tracks_opt;
        FusedObjectList fused;
        fused.timestamp_ns = tracks.timestamp_ns;
        fused.frame_sequence = tracks.frame_sequence;

        // STEP 1+2: Associate tracks with LiDAR clusters
        std::vector<bool> lidar_matched(latest_lidar.size(), false);

        for (auto& track : tracks.objects) {
            FusedObject obj;
            obj.track_id = track.track_id;
            obj.class_id = track.class_id;
            obj.confidence = track.confidence;
            obj.has_camera = true;
            obj.has_lidar = false;
            obj.has_radar = false;
            obj.timestamp_ns = track.timestamp_ns;

            // Build a pseudo Detection2D from tracked position for bbox
            Detection2D track_bbox;
            track_bbox.x = track.position_2d.x() - 50;  // approximate bbox
            track_bbox.y = track.position_2d.y() - 50;
            track_bbox.w = 100;
            track_bbox.h = 100;

            // Find best matching LiDAR cluster
            float best_dist = std::numeric_limits<float>::max();
            int best_lidar = -1;

            for (size_t i = 0; i < latest_lidar.size(); ++i) {
                if (lidar_matched[i]) continue;

                auto proj = project_lidar_to_camera(
                    latest_lidar[i].centroid, calib);
                if (proj.x() < 0) continue;  // not visible

                float dist = (proj - track.position_2d).norm();
                if (dist < best_dist && dist < 100.0f) {  // within 100px
                    best_dist = dist;
                    best_lidar = i;
                }
            }

            if (best_lidar >= 0) {
                obj.has_lidar = true;
                obj.position_3d = latest_lidar[best_lidar].centroid;
                lidar_matched[best_lidar] = true;
                obj.confidence = std::min(1.0f, obj.confidence + 0.1f);
            }

            // STEP 3: Cross-check with radar
            for (auto& rd : latest_radar.detections) {
                if (!obj.has_lidar) break;  // need 3D position to compare

                float range_diff = std::abs(
                    obj.position_3d.norm() - rd.range);
                float azimuth_diff = std::abs(
                    std::atan2(obj.position_3d.y(), obj.position_3d.x())
                    - rd.azimuth);

                if (range_diff < 2.0f && azimuth_diff < 0.26f) {  // ±2m, ±15°
                    obj.has_radar = true;
                    obj.velocity_3d.x() = rd.velocity_radial *
                        std::cos(rd.azimuth);
                    obj.velocity_3d.y() = rd.velocity_radial *
                        std::sin(rd.azimuth);
                    obj.confidence = std::min(1.0f, obj.confidence + 0.15f);
                    break;
                }
            }

            fused.objects.push_back(obj);
        }

        // Publish to data bus
        // data_bus.publish("/perception/objects", fused);
        spdlog::debug("Fused: {} objects ({} with LiDAR, {} with radar)",
            fused.objects.size(),
            std::count_if(fused.objects.begin(), fused.objects.end(),
                         [](const auto& o) { return o.has_lidar; }),
            std::count_if(fused.objects.begin(), fused.objects.end(),
                         [](const auto& o) { return o.has_radar; }));
    }
    spdlog::info("Fusion thread stopped");
}

class IFeatureDetector {

    struct Result {
        std::vector<cv::KeyPoint> keypoints;
        cv::Mat descriptors;           // CV_8U for ORB, CV_32F for SuperPoint
    };

    virtual Result detect(const cv::Mat& grey_frame) = 0;
    virtual std::string name() const = 0;

// ── ORB concrete strategy ──
class OrbDetector final : public IFeatureDetector {
    cv::Ptr<cv::ORB> orb_;
    int grid_rows_ = 6, grid_cols_ = 8;
public:
    explicit OrbDetector(int max_features = 1500)
        : orb_(cv::ORB::create(max_features)) {}

    Result detect(const cv::Mat& grey) override {
        Result res;
        int cell_h = grey.rows / grid_rows_;
        int cell_w = grey.cols / grid_cols_;
        int per_cell = orb_->getMaxFeatures() / (grid_rows_ * grid_cols_);

        for (int r = 0; r < grid_rows_; ++r) {
            for (int c = 0; c < grid_cols_; ++c) {
                cv::Rect roi(c * cell_w, r * cell_h, cell_w, cell_h);
                std::vector<cv::KeyPoint> kps;
                cv::Mat desc;
                orb_->detectAndCompute(grey(roi), cv::noArray(), kps, desc);
                // Offset keypoints back to full-image coordinates
                for (auto& kp : kps) {
                    kp.pt.x += roi.x;
                    kp.pt.y += roi.y;
                }
                // Keep top-N per cell
                if ((int)kps.size() > per_cell) {
                    std::partial_sort(kps.begin(), kps.begin() + per_cell,
                                      kps.end(), [](auto& a, auto& b) {
                                          return a.response > b.response;
                                      });
                    kps.resize(per_cell);
                    desc = desc.rowRange(0, per_cell).clone();
                }
                res.keypoints.insert(res.keypoints.end(), kps.begin(), kps.end());
                if (!desc.empty())
                    res.descriptors.push_back(desc);
            }
        }
        return res;
    }
    std::string name() const override { return "ORB-Grid"; }
};

struct KeyframePolicy {
    double min_parallax_px    = 15.0;
    double min_tracked_ratio  = 0.6;
    double max_time_sec       = 0.5;

    bool should_create_keyframe(
        double median_parallax,
        int tracked_count,
        int initial_count,
        double dt_since_last_kf,
        bool backend_idle) const
    {
        if (median_parallax > min_parallax_px) return true;
        double ratio = static_cast<double>(tracked_count) / initial_count;
        if (ratio < min_tracked_ratio) return true;
        if (dt_since_last_kf > max_time_sec) return true;
        if (backend_idle && median_parallax > min_parallax_px * 0.5) return true;
        return false;
    }
};

void VisualFrontend::run() {
    while (!stop_flag_.load(std::memory_order_relaxed)) {
        // 1. Wait for new frame from shared memory
        FrameHandle frame = shm_reader_.wait_and_acquire();  // blocks on semaphore
        if (!frame.valid()) continue;

        cv::Mat grey(frame.height(), frame.width(), CV_8UC1,
                     frame.data());  // zero-copy wrap

        // 2. Detect & match features
        auto result = detector_->detect(grey);
        auto matches = matcher_.match(prev_result_, result);

        // 3. Geometric verification (Essential matrix via RANSAC)
        auto [inlier_matches, E] = geometric_verify(matches, camera_intrinsics_);

        // 4. Compute relative pose R|t from Essential matrix
        Sophus::SE3d T_cur_prev = decompose_essential(E, inlier_matches, camera_intrinsics_);

        // 5. Accumulate relative motion since last keyframe
        T_cur_kf_ = T_cur_prev * T_cur_kf_;

        double parallax = compute_median_parallax(inlier_matches);
        double dt = frame.timestamp() - last_kf_time_;

        // 6. Keyframe decision
        if (kf_policy_.should_create_keyframe(
                parallax, inlier_matches.size(),
                prev_result_.keypoints.size(), dt,
                backend_queue_.was_empty())) {
            KeyframeData kf{frame.timestamp(), grey.clone(), result,
                            inlier_matches, T_cur_kf_};
            backend_queue_.push(std::move(kf));     // SPSC → Thread 3
            T_cur_kf_ = Sophus::SE3d{};             // reset
            last_kf_time_ = frame.timestamp();
        }

        // 7. Always publish fast odometry pose
        pose_queue_.push(PoseStamped{frame.timestamp(), T_world_cur_});

        prev_result_ = std::move(result);
        shm_reader_.release(frame);
    }
}

template <typename Derived>
class ImuPreintegratorBase {
public:
    void integrate(const ImuSample& sample) {
        // Common bookkeeping
        dt_total_ += sample.dt;
        n_samples_++;

        // Delegate noise model to derived (no virtual call!)
        static_cast<Derived*>(this)->propagate_impl(sample);
    }

    PreintegratedImuFactor get_factor() const {
        return {delta_R_, delta_v_, delta_p_,
                jacobian_bias_gyro_, jacobian_bias_accel_,
                covariance_, dt_total_};
    }

protected:
    Sophus::SO3d delta_R_ = Sophus::SO3d{};
    Eigen::Vector3d delta_v_ = Eigen::Vector3d::Zero();
    Eigen::Vector3d delta_p_ = Eigen::Vector3d::Zero();
    Eigen::Matrix<double,9,6> jacobian_bias_gyro_;
    Eigen::Matrix<double,9,6> jacobian_bias_accel_;
    Eigen::Matrix<double,9,9> covariance_ = Eigen::Matrix<double,9,9>::Zero();
    double dt_total_ = 0.0;
    int n_samples_ = 0;
};

class ContinuousNoiseModel
    : public ImuPreintegratorBase<ContinuousNoiseModel> {
    friend class ImuPreintegratorBase<ContinuousNoiseModel>;
    void propagate_impl(const ImuSample& s) {
        // Continuous-time noise propagation
        Eigen::Vector3d a_corrected = s.accel - bias_accel_;
        Eigen::Vector3d w_corrected = s.gyro  - bias_gyro_;
        Sophus::SO3d dR = Sophus::SO3d::exp(w_corrected * s.dt);
        delta_p_ += delta_v_ * s.dt + 0.5 * delta_R_ * a_corrected * s.dt * s.dt;
        delta_v_ += delta_R_ * a_corrected * s.dt;
        delta_R_ = delta_R_ * dR;
        // Update Jacobians & covariance (omitted for brevity)
    }
    Eigen::Vector3d bias_accel_, bias_gyro_;
};

class SpiImuDriver {
    int fd_;
public:
    explicit SpiImuDriver(const std::string& device = "/dev/spidev0.0",
                          uint32_t speed_hz = 8'000'000)
    {
        fd_ = ::open(device.c_str(), O_RDWR);
        if (fd_ < 0) throw std::system_error(errno, std::system_category(), "SPI open");
        ::ioctl(fd_, SPI_IOC_WR_MAX_SPEED_HZ, &speed_hz);
        uint8_t mode = SPI_MODE_3;
        ::ioctl(fd_, SPI_IOC_WR_MODE, &mode);
    }

    ImuSample read_blocking() {
        uint8_t tx[14] = {0x80 | REG_ACCEL_XOUT_H};  // read burst
        uint8_t rx[14] = {};
        spi_ioc_transfer xfer{};
        xfer.tx_buf = reinterpret_cast<__u64>(tx);
        xfer.rx_buf = reinterpret_cast<__u64>(rx);
        xfer.len = sizeof(tx);
        xfer.speed_hz = 8'000'000;
        ::ioctl(fd_, SPI_IOC_MESSAGE(1), &xfer);

        ImuSample s;
        s.timestamp = steady_clock_now();
        s.accel = parse_accel(&rx[1]);   // 3 × int16 → m/s²
        s.gyro  = parse_gyro(&rx[7]);    // 3 × int16 → rad/s
        return s;
    }

    ~SpiImuDriver() { if (fd_ >= 0) ::close(fd_); }
};

    BackendOptimiser() {
        gtsam::ISAM2Params params;
        params.relinearizeThreshold = 0.01;
        params.relinearizeSkip = 1;
        params.enableDetailedResults = false;
        isam2_ = gtsam::ISAM2(params);

        // Pose initial estimate
        gtsam::Pose3 pose_init = sophus_to_gtsam(kf.T_world_kf);
        new_values_.insert(X(kf_id_), pose_init);

        auto imu = boost::make_shared<gtsam::CombinedImuFactor>(

            B(prev), B(kf_id_), imu_factor.gtsam_pim);
        new_factors_.add(imu);

        // Visual re-projection factors (SmartFactor per landmark)
        for (auto& [lm_id, obs] : kf.observations) {
            smart_factors_[lm_id]->add(obs.uv, X(kf_id_));
        }

        // Incremental update
        auto result = isam2_.update(new_factors_, new_values_);
        new_factors_.resize(0);
        new_values_.clear();

        // Extract optimised pose
        gtsam::Values estimate = isam2_.calculateEstimate();
        publish_optimised_poses(estimate);
    }

// Shared-memory layout for pose publication
struct alignas(64) ShmPoseSlot {
    std::atomic<uint64_t> sequence;     // monotonic, odd = writing
    double timestamp;
    double translation[3];              // x, y, z in world frame
    double quaternion[4];               // w, x, y, z
    double covariance[36];              // 6×6 upper-triangle
    uint8_t padding[64 - (8+8+24+32+288) % 64];
};

class PosePublisher {
    ShmPoseSlot* slot_;  // mmap'd shared memory

public:
    void publish(double ts, const Sophus::SE3d& T, const Eigen::Matrix6d& cov) {
        uint64_t seq = slot_->sequence.load(std::memory_order_relaxed);
        slot_->sequence.store(seq + 1, std::memory_order_release);  // odd = writing

        slot_->timestamp = ts;
        Eigen::Map<Eigen::Vector3d>(slot_->translation) = T.translation();
        Eigen::Quaterniond q(T.rotationMatrix());
        slot_->quaternion[0] = q.w();
        slot_->quaternion[1] = q.x();
        slot_->quaternion[2] = q.y();
        slot_->quaternion[3] = q.z();
        Eigen::Map<Eigen::Matrix<double,6,6>>(slot_->covariance) = cov;

        slot_->sequence.store(seq + 2, std::memory_order_release);  // even = valid
    }
};

// Reader side (in Process 4 or Process 5):
bool read_pose(const ShmPoseSlot* slot, PoseStamped& out) {
    for (int attempt = 0; attempt < 3; ++attempt) {
        uint64_t seq1 = slot->sequence.load(std::memory_order_acquire);
        if (seq1 & 1) continue;  // writer is mid-update, retry

        // Copy data
        out.timestamp = slot->timestamp;
        out.position  = Eigen::Map<const Eigen::Vector3d>(slot->translation);
        out.orientation = Eigen::Quaterniond(
            slot->quaternion[0], slot->quaternion[1],
            slot->quaternion[2], slot->quaternion[3]);

        std::atomic_thread_fence(std::memory_order_acquire);
        uint64_t seq2 = slot->sequence.load(std::memory_order_relaxed);
        if (seq1 == seq2) return true;  // consistent read
    }
    return false;  // failed after retries (write was in progress)
}

template <typename T, size_t Capacity>
class SpscQueue {
    static_assert((Capacity & (Capacity - 1)) == 0, "Capacity must be power of 2");

    struct alignas(64) {
        std::atomic<size_t> head{0};
    };
    struct alignas(64) {
        std::atomic<size_t> tail{0};
    };
    std::array<T, Capacity> buffer_;

public:
    bool push(T&& item) {
        size_t h = head.load(std::memory_order_relaxed);
        size_t next = (h + 1) & (Capacity - 1);
        if (next == tail.load(std::memory_order_acquire)) return false; // full
        buffer_[h] = std::move(item);
        head.store(next, std::memory_order_release);
        return true;
    }

    bool pop(T& item) {
        size_t t = tail.load(std::memory_order_relaxed);
        if (t == head.load(std::memory_order_acquire)) return false; // empty
        item = std::move(buffer_[t]);
        tail.store((t + 1) & (Capacity - 1), std::memory_order_release);
        return true;
    }
};

class ShmFrameReader {
    int shm_fd_;
    void* base_ptr_;
    size_t ring_size_;
    sem_t* sem_;
    size_t read_idx_ = 0;

    struct FrameHeader {
        uint64_t sequence;
        uint64_t timestamp_ns;
        uint32_t width, height, stride;
        uint32_t format;  // GREY8
    };

public:
    ShmFrameReader(const std::string& shm_name, const std::string& sem_name,
                   size_t num_slots)
    {
        shm_fd_ = shm_open(shm_name.c_str(), O_RDONLY, 0);
        if (shm_fd_ < 0) throw std::system_error(errno, std::system_category());

        struct stat sb;
        fstat(shm_fd_, &sb);
        ring_size_ = sb.st_size;
        base_ptr_ = mmap(nullptr, ring_size_, PROT_READ, MAP_SHARED, shm_fd_, 0);

        sem_ = sem_open(sem_name.c_str(), 0);
    }

    FrameHandle wait_and_acquire() {
        sem_wait(sem_);  // block until new frame
        auto* header = reinterpret_cast<const FrameHeader*>(
            static_cast<char*>(base_ptr_) + read_idx_ * slot_size_);
        read_idx_ = (read_idx_ + 1) % num_slots_;
        return FrameHandle{header};
    }

    ~ShmFrameReader() {
        munmap(base_ptr_, ring_size_);
        close(shm_fd_);
        sem_close(sem_);
    }
};
