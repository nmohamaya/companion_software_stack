=== Listing 6-1: Thermal Manager with Fan Control ===

// thermal/thermal_manager.h
#pragma once
#include <fstream>
#include <string>
#include <vector>
#include <algorithm>
#include <spdlog/spdlog.h>

enum class ThermalZone : uint8_t {
    NORMAL, WARNING, CRITICAL, EMERGENCY
};

struct ThermalSnapshot {
    double cpu_temp_c     = 0.0;
    double gpu_temp_c     = 0.0;
    double board_temp_c   = 0.0;
    double max_temp_c     = 0.0;
    ThermalZone zone      = ThermalZone::NORMAL;
    double fan_duty       = 0.0;  // 0.0 - 1.0
};

class ThermalManager {
public:
    bool init(const std::string& fan_pwm_path = "/sys/devices/pwm-fan/target_pwm") {
        fan_pwm_path_ = fan_pwm_path;
        // Discover thermal zones
        for (int i = 0; i < 10; ++i) {
            std::string path = "/sys/class/thermal/thermal_zone" +
                               std::to_string(i) + "/temp";
            std::ifstream f(path);
            if (f.good()) zone_paths_.push_back(path);
        }

        // Try to read zone type names for labelling
        for (int i = 0; i < static_cast<int>(zone_paths_.size()); ++i) {
            std::string type_path = "/sys/class/thermal/thermal_zone" +
                                     std::to_string(i) + "/type";
            std::ifstream f(type_path);
            std::string type_name;
            if (f >> type_name) zone_names_.push_back(type_name);
            else zone_names_.push_back("zone" + std::to_string(i));
        }

        spdlog::info("ThermalManager: found {} thermal zones", zone_paths_.size());
        return !zone_paths_.empty();
    }

    ThermalSnapshot update() {
        ThermalSnapshot snap;
        std::vector<double> temps;

        for (size_t i = 0; i < zone_paths_.size(); ++i) {
            double t = read_temp(zone_paths_[i]);
            temps.push_back(t);

            // Label known zones (Jetson Orin Nano specific)
            if (zone_names_[i].find("CPU") != std::string::npos ||
                zone_names_[i].find("cpu") != std::string::npos)
                snap.cpu_temp_c = t;
            else if (zone_names_[i].find("GPU") != std::string::npos ||
                     zone_names_[i].find("gpu") != std::string::npos)
                snap.gpu_temp_c = t;
        }

        snap.max_temp_c = *std::max_element(temps.begin(), temps.end());
        snap.board_temp_c = snap.max_temp_c;  // fallback

        // Classify thermal zone
        if (snap.max_temp_c >= emergency_threshold_)
            snap.zone = ThermalZone::EMERGENCY;
        else if (snap.max_temp_c >= critical_threshold_)
            snap.zone = ThermalZone::CRITICAL;
        else if (snap.max_temp_c >= warning_threshold_)
            snap.zone = ThermalZone::WARNING;
        else
            snap.zone = ThermalZone::NORMAL;

        // Compute fan duty cycle (linear ramp between min and max temp)
        snap.fan_duty = compute_fan_duty(snap.max_temp_c);
        set_fan_pwm(snap.fan_duty);

        return snap;
    }

private:
    std::vector<std::string> zone_paths_;
    std::vector<std::string> zone_names_;
    std::string fan_pwm_path_;

    double warning_threshold_   = 65.0;
    double critical_threshold_  = 80.0;
    double emergency_threshold_ = 90.0;

    double fan_min_temp_ = 45.0;   // fan starts ramping
    double fan_max_temp_ = 75.0;   // fan at 100%

    double read_temp(const std::string& path) {
        std::ifstream f(path);
        int millideg = 0;
        f >> millideg;
        return millideg / 1000.0;
    }

    double compute_fan_duty(double temp_c) {
        if (temp_c <= fan_min_temp_) return 0.0;
        if (temp_c >= fan_max_temp_) return 1.0;
        return (temp_c - fan_min_temp_) / (fan_max_temp_ - fan_min_temp_);
    }

    void set_fan_pwm(double duty) {
        int pwm_val = static_cast<int>(duty * 255);
        std::ofstream f(fan_pwm_path_);
        if (f) f << pwm_val;
    }
};
