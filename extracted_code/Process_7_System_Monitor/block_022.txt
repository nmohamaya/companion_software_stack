// logging/log_config.h
#pragma once
#include <spdlog/spdlog.h>
#include <spdlog/sinks/rotating_file_sink.h>
#include <spdlog/sinks/stdout_color_sinks.h>
#include <spdlog/sinks/daily_file_sink.h>
#include <spdlog/pattern_formatter.h>
#include <string>
#include <memory>

class LogConfig {
public:
    static void init(const std::string& process_name,
                     const std::string& log_dir = "/opt/drone/logs",
                     const std::string& level = "info") {
        // Console sink (for systemd journal)
        auto console = std::make_shared<spdlog::sinks::stdout_color_sink_mt>();
        console->set_pattern("[%Y-%m-%d %H:%M:%S.%f] [%n] [%^%l%$] %v");

        // Rotating file sink (10 MB per file, 5 rotated files)
        std::string log_path = log_dir + "/" + process_name + ".log";
        auto rotating = std::make_shared<spdlog::sinks::rotating_file_sink_mt>(
            log_path, 10 * 1024 * 1024, 5);
        rotating->set_pattern("[%Y-%m-%d %H:%M:%S.%f] [%l] [thread %t] %v");

        // Create multi-sink logger
        auto logger = std::make_shared<spdlog::logger>(
            process_name,
            spdlog::sinks_init_list{console, rotating});

        logger->set_level(spdlog::level::from_str(level));
        logger->flush_on(spdlog::level::warn);  // auto-flush on warnings

        // Register as default
        spdlog::set_default_logger(logger);
        spdlog::info("Logging initialised: level={}, path={}", level, log_path);
    }
};
