=== Listing 14-1: Black-Box Flight Recorder (mmap Ring Buffer) ===

// blackbox/flight_recorder.h
#pragma once
#include "store/health_store.h"
#include <sys/mman.h>
#include <fcntl.h>
#include <unistd.h>
#include <cstring>
#include <string>
#include <spdlog/spdlog.h>

#pragma pack(push, 1)
struct FlightRecord {
    uint64_t timestamp_ns;
    // Compact system state (fixed-size for binary format)
    float    cpu_percent;
    float    memory_percent;
    float    cpu_temp;
    float    gpu_temp;
    float    voltage;
    float    current;
    float    disk_percent;
    uint8_t  thermal_zone;       // ThermalZone enum
    uint8_t  processes_healthy;
    uint8_t  processes_dead;
    uint8_t  alert_count;
    uint32_t images_captured;
    float    gimbal_pitch;
    float    gimbal_roll;
    float    gimbal_yaw;
};
#pragma pack(pop)

static_assert(sizeof(FlightRecord) == 56, "FlightRecord must be 56 bytes");

class FlightRecorder {
public:
    bool open(const std::string& path, size_t max_records = 360000) {
        // 360000 records × 56 bytes = ~20 MB
        // At 10 Hz, this is 10 hours of recording
        capacity_ = max_records;
        file_size_ = sizeof(FlightRecord) * capacity_ + sizeof(Header);

        fd_ = ::open(path.c_str(), O_CREAT | O_RDWR, 0644);
        if (fd_ < 0) return false;
        ftruncate(fd_, file_size_);

        base_ = static_cast<uint8_t*>(
            mmap(nullptr, file_size_, PROT_READ | PROT_WRITE,
                 MAP_SHARED, fd_, 0));
        if (base_ == MAP_FAILED) return false;

        header_ = reinterpret_cast<Header*>(base_);
        records_ = reinterpret_cast<FlightRecord*>(base_ + sizeof(Header));

        // If existing file, continue from stored write position
        if (header_->magic == MAGIC) {
            write_pos_ = header_->write_pos;
            spdlog::info("FlightRecorder: resuming at record {}",
                          write_pos_);
        } else {
            header_->magic = MAGIC;
            header_->version = 1;
            header_->capacity = capacity_;
            header_->write_pos = 0;
            write_pos_ = 0;
        }

        spdlog::info("FlightRecorder: opened {} ({:.1f} MB, {} records)",
                      path, file_size_ / 1e6, capacity_);
        return true;
    }

    void record(const HealthSnapshot& snap) {
        if (!records_) return;

        size_t idx = write_pos_ % capacity_;
        FlightRecord& rec = records_[idx];

        rec.timestamp_ns     = snap.timestamp_ns;
        rec.cpu_percent      = static_cast<float>(snap.cpu_usage_percent);
        rec.memory_percent   = static_cast<float>(snap.memory.usage_percent());
        rec.cpu_temp         = static_cast<float>(snap.thermal.cpu_temp_c);
        rec.gpu_temp         = static_cast<float>(snap.thermal.gpu_temp_c);
        rec.voltage          = static_cast<float>(snap.power.voltage_v);
        rec.current          = static_cast<float>(snap.power.current_a);
        rec.disk_percent     = static_cast<float>(snap.disk.usage_percent);
        rec.thermal_zone     = static_cast<uint8_t>(snap.thermal.zone);
        rec.processes_healthy = snap.watchdog.total_healthy;
        rec.processes_dead    = snap.watchdog.total_dead;
        rec.alert_count      = 0;  // filled by caller if needed

        ++write_pos_;
        header_->write_pos = write_pos_;

        // Periodic sync (every 100 records = 10 seconds at 10 Hz)
        if (write_pos_ % 100 == 0) {
            msync(base_, file_size_, MS_ASYNC);
        }
    }

    ~FlightRecorder() {
        if (base_ && base_ != MAP_FAILED) {
            msync(base_, file_size_, MS_SYNC);
            munmap(base_, file_size_);
        }
        if (fd_ >= 0) ::close(fd_);
    }

private:
    struct Header {
        uint64_t magic;
        uint32_t version;
        uint32_t capacity;
        uint64_t write_pos;
    };

    static constexpr uint64_t MAGIC = 0x424C41434B424F58ULL; // "BLACKBOX"

    int fd_ = -1;
    uint8_t* base_ = nullptr;
    Header* header_ = nullptr;
    FlightRecord* records_ = nullptr;
    size_t capacity_ = 0;
    size_t file_size_ = 0;
    size_t write_pos_ = 0;
};
