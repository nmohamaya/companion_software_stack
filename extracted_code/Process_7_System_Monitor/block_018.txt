=== Listing 11-1: Alert Engine with Token Bucket Rate Limiting ===

// alert/alert_engine.h
#pragma once
#include <string>
#include <vector>
#include <deque>
#include <chrono>
#include <functional>
#include <mutex>
#include <spdlog/spdlog.h>

enum class AlertSeverity : uint8_t {
    INFO, WARNING, CRITICAL, EMERGENCY
};

struct Alert {
    AlertSeverity severity;
    std::string source;
    std::string message;
    uint64_t timestamp_ns;
    uint32_t id;
};

class TokenBucket {
public:
    TokenBucket(int max_tokens, double refill_rate)
        : max_tokens_(max_tokens), tokens_(max_tokens),
          refill_rate_(refill_rate),
          last_refill_(std::chrono::steady_clock::now()) {}

    bool try_consume() {
        refill();
        if (tokens_ > 0) {
            --tokens_;
            return true;
        }
        return false;
    }

private:
    int max_tokens_;
    double tokens_;
    double refill_rate_;
    std::chrono::steady_clock::time_point last_refill_;

    void refill() {
        auto now = std::chrono::steady_clock::now();
        double elapsed = std::chrono::duration<double>(now - last_refill_).count();
        tokens_ = std::min(double(max_tokens_), tokens_ + elapsed * refill_rate_);
        last_refill_ = now;
    }
};

class AlertEngine {
public:
    using AlertCallback = std::function<void(const Alert&)>;

    AlertEngine() : rate_limiter_(5, 0.5) {}  // 5 burst, 0.5 per sec

    void set_callback(AlertCallback cb) { callback_ = std::move(cb); }

    void emit(AlertSeverity severity, const std::string& source,
              const std::string& message) {
        // Rate limit non-EMERGENCY alerts
        if (severity != AlertSeverity::EMERGENCY) {
            if (!rate_limiter_.try_consume()) {
                ++suppressed_count_;
                return;
            }
        }

        Alert alert;
        alert.severity = severity;
        alert.source = source;
        alert.message = message;
        alert.timestamp_ns = current_time_ns();
        alert.id = ++alert_counter_;

        // Log
        switch (severity) {
            case AlertSeverity::EMERGENCY:
                spdlog::critical("[ALERT] {}: {}", source, message); break;
            case AlertSeverity::CRITICAL:
                spdlog::error("[ALERT] {}: {}", source, message); break;
            case AlertSeverity::WARNING:
                spdlog::warn("[ALERT] {}: {}", source, message); break;
            default:
                spdlog::info("[ALERT] {}: {}", source, message); break;
        }

        // Callback to telemetry
        if (callback_) callback_(alert);

        // Store in history ring buffer
        std::lock_guard lock(history_mtx_);
        history_.push_back(alert);
        if (history_.size() > max_history_) history_.pop_front();
    }

    std::vector<Alert> recent_alerts(int n = 10) {
        std::lock_guard lock(history_mtx_);
        std::vector<Alert> recent;
        auto it = history_.rbegin();
        for (int i = 0; i < n && it != history_.rend(); ++it, ++i)
            recent.push_back(*it);
        return recent;
    }

private:
    AlertCallback callback_;
    TokenBucket rate_limiter_;
    uint32_t alert_counter_ = 0;
    uint32_t suppressed_count_ = 0;
    std::deque<Alert> history_;
    std::mutex history_mtx_;
    size_t max_history_ = 1000;

    uint64_t current_time_ns() const;
};
