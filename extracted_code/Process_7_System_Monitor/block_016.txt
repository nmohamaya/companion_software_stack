=== Listing 10-1: Network Monitor (reads /sys/class/net) ===

// monitor/network_monitor.h
#pragma once
#include <fstream>
#include <string>

struct NetworkSnapshot {
    std::string interface;
    uint64_t rx_bytes       = 0;
    uint64_t tx_bytes       = 0;
    uint64_t rx_packets     = 0;
    uint64_t tx_packets     = 0;
    uint64_t rx_errors      = 0;
    uint64_t tx_errors      = 0;
    uint64_t rx_dropped     = 0;
    bool     link_up        = false;

    // Computed rates (bytes/sec)
    double rx_rate_bps      = 0.0;
    double tx_rate_bps      = 0.0;
};

class NetworkMonitor {
public:
    explicit NetworkMonitor(const std::string& iface = "wlan0")
        : iface_(iface) {}

    NetworkSnapshot update() {
        NetworkSnapshot snap;
        snap.interface = iface_;

        std::string base = "/sys/class/net/" + iface_ + "/statistics/";
        snap.rx_bytes   = read_stat(base + "rx_bytes");
        snap.tx_bytes   = read_stat(base + "tx_bytes");
        snap.rx_packets = read_stat(base + "rx_packets");
        snap.tx_packets = read_stat(base + "tx_packets");
        snap.rx_errors  = read_stat(base + "rx_errors");
        snap.tx_errors  = read_stat(base + "tx_errors");
        snap.rx_dropped = read_stat(base + "rx_dropped");

        // Link state
        std::ifstream carrier("/sys/class/net/" + iface_ + "/carrier");
        int c = 0;
        if (carrier >> c) snap.link_up = (c == 1);

        // Compute rates
        snap.rx_rate_bps = static_cast<double>(snap.rx_bytes - prev_rx_) * 8.0;
        snap.tx_rate_bps = static_cast<double>(snap.tx_bytes - prev_tx_) * 8.0;
        prev_rx_ = snap.rx_bytes;
        prev_tx_ = snap.tx_bytes;

        return snap;
    }

private:
    std::string iface_;
    uint64_t prev_rx_ = 0, prev_tx_ = 0;

    uint64_t read_stat(const std::string& path) {
        std::ifstream f(path);
        uint64_t val = 0;
        f >> val;
        return val;
    }
};
