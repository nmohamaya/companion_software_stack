# CMakeLists.txt — Process 7: System Monitor
cmake_minimum_required(VERSION 3.18)
project(system_monitor VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# ── Compiler flags ──────────────────────────────────────────────
add_compile_options(
    -Wall -Wextra -Wpedantic -Werror
    -O2
    -march=armv8.2-a+fp16
)

# ── Dependencies ────────────────────────────────────────────────
find_package(spdlog REQUIRED)
find_package(nlohmann_json 3.11 REQUIRED)
find_package(Threads REQUIRED)

# libsystemd for sd_notify
find_library(SYSTEMD_LIB systemd)
if(NOT SYSTEMD_LIB)
    message(WARNING "libsystemd not found — watchdog integration disabled")
endif()

# ── Sources ─────────────────────────────────────────────────────
set(SYSMON_SOURCES
    src/main.cpp
    src/monitor/cpu_monitor.cpp
    src/monitor/memory_monitor.cpp
    src/monitor/process_monitor.cpp
    src/monitor/disk_monitor.cpp
    src/monitor/power_monitor.cpp
    src/monitor/network_monitor.cpp
    src/thermal/thermal_manager.cpp
    src/watchdog/process_watchdog.cpp
    src/watchdog/watchdog_thread.cpp
    src/alert/alert_engine.cpp
    src/store/health_store.cpp
    src/aggregate/telemetry_thread.cpp
    src/aggregate/resource_thread.cpp
    src/blackbox/flight_recorder.cpp
    src/blackbox/recorder_thread.cpp
    src/ipc/shm_health_reader.cpp
    src/ipc/system_health_writer.cpp
    src/config/config_loader.cpp
    src/logging/log_config.cpp
    src/util/realtime.cpp
)

# ── Executable ──────────────────────────────────────────────────
add_executable(system_monitor ${SYSMON_SOURCES})

target_include_directories(system_monitor PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/include
)

target_link_libraries(system_monitor PRIVATE
    spdlog::spdlog
    nlohmann_json::nlohmann_json
    Threads::Threads
    rt          # POSIX shared memory
    pthread     # POSIX threads
)

if(SYSTEMD_LIB)
    target_link_libraries(system_monitor PRIVATE ${SYSTEMD_LIB})
    target_compile_definitions(system_monitor PRIVATE HAS_SYSTEMD=1)
endif()

# ── Install ─────────────────────────────────────────────────────
install(TARGETS system_monitor DESTINATION /opt/drone/bin)
install(FILES config/system_monitor_config.json DESTINATION /opt/drone/etc)

# ── Tests ───────────────────────────────────────────────────────
option(BUILD_TESTS "Build unit tests" ON)
if(BUILD_TESTS)
    enable_testing()
    find_package(GTest REQUIRED)

    add_executable(test_cpu_monitor
        tests/test_cpu_monitor.cpp
        src/monitor/cpu_monitor.cpp
    )
    target_include_directories(test_cpu_monitor PRIVATE include)
    target_link_libraries(test_cpu_monitor GTest::gtest_main)
    add_test(NAME CPUMonitorTest COMMAND test_cpu_monitor)

    add_executable(test_thermal_manager
        tests/test_thermal_manager.cpp
        src/thermal/thermal_manager.cpp
    )
    target_include_directories(test_thermal_manager PRIVATE include)
    target_link_libraries(test_thermal_manager GTest::gtest_main spdlog::spdlog)
    add_test(NAME ThermalManagerTest COMMAND test_thermal_manager)

    add_executable(test_alert_engine
        tests/test_alert_engine.cpp
        src/alert/alert_engine.cpp
    )
    target_include_directories(test_alert_engine PRIVATE include)
    target_link_libraries(test_alert_engine GTest::gtest_main spdlog::spdlog)
    add_test(NAME AlertEngineTest COMMAND test_alert_engine)

    add_executable(test_flight_recorder
        tests/test_flight_recorder.cpp
        src/blackbox/flight_recorder.cpp
    )
    target_include_directories(test_flight_recorder PRIVATE include)
    target_link_libraries(test_flight_recorder GTest::gtest_main spdlog::spdlog)
    add_test(NAME FlightRecorderTest COMMAND test_flight_recorder)

    add_executable(test_watchdog
        tests/test_watchdog.cpp
        src/watchdog/process_watchdog.cpp
        src/ipc/shm_health_reader.cpp
    )
    target_include_directories(test_watchdog PRIVATE include)
    target_link_libraries(test_watchdog GTest::gtest_main spdlog::spdlog rt)
    add_test(NAME WatchdogTest COMMAND test_watchdog)
endif()
