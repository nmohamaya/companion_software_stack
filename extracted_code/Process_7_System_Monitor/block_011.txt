=== Listing 7-1: Process Watchdog with SHM Freshness Check ===

// watchdog/process_watchdog.h
#pragma once
#include "ipc/shm_health_reader.h"
#include <string>
#include <vector>
#include <unordered_map>
#include <chrono>
#include <fstream>
#include <spdlog/spdlog.h>

enum class ProcessHealth : uint8_t {
    HEALTHY,        // running & SHM fresh
    DEGRADED,       // running but SHM stale
    DEAD,           // process not found
    RESTARTING,     // systemd restarting it
    UNKNOWN         // initial state
};

struct WatchdogEntry {
    std::string process_name;
    std::string shm_name;        // SHM to check (empty if none)
    double shm_timeout_ms;       // max SHM age before "degraded"
    int pid = -1;
    ProcessHealth health = ProcessHealth::UNKNOWN;
    int restart_count = 0;
    std::chrono::steady_clock::time_point last_healthy;
};

struct WatchdogReport {
    std::vector<WatchdogEntry> entries;
    int total_healthy = 0;
    int total_degraded = 0;
    int total_dead = 0;
    bool all_healthy = false;
};

class ProcessWatchdog {
public:
    void add_watch(const std::string& process_name,
                   const std::string& shm_name = "",
                   double shm_timeout_ms = 2000.0) {
        WatchdogEntry entry;
        entry.process_name = process_name;
        entry.shm_name = shm_name;
        entry.shm_timeout_ms = shm_timeout_ms;

        if (!shm_name.empty()) {
            auto checker = std::make_unique<ShmFreshnessChecker>();
            checker->set_timeout_ms(shm_timeout_ms);
            checker->open(shm_name);
            shm_checkers_[process_name] = std::move(checker);
        }

        entries_.push_back(std::move(entry));
    }

    WatchdogReport check() {
        WatchdogReport report;

        for (auto& entry : entries_) {
            // Step 1: Is the process alive?
            entry.pid = find_pid(entry.process_name);
            if (entry.pid < 0) {
                entry.health = ProcessHealth::DEAD;
                spdlog::warn("WATCHDOG: {} is DEAD", entry.process_name);
            }
            else if (!entry.shm_name.empty()) {
                // Step 2: Is its SHM output fresh?
                auto it = shm_checkers_.find(entry.process_name);
                if (it != shm_checkers_.end()) {
                    auto freshness = it->second->check();
                    if (freshness.is_fresh) {
                        entry.health = ProcessHealth::HEALTHY;
                        entry.last_healthy = std::chrono::steady_clock::now();
                    } else {
                        entry.health = ProcessHealth::DEGRADED;
                        spdlog::warn("WATCHDOG: {} SHM stale ({:.0f} ms)",
                                      entry.process_name, freshness.age_ms);
                    }
                }
            } else {
                // No SHM to check — alive is enough
                entry.health = ProcessHealth::HEALTHY;
                entry.last_healthy = std::chrono::steady_clock::now();
            }

            // Tally
            switch (entry.health) {
                case ProcessHealth::HEALTHY:  ++report.total_healthy; break;
                case ProcessHealth::DEGRADED: ++report.total_degraded; break;
                case ProcessHealth::DEAD:     ++report.total_dead; break;
                default: break;
            }
        }

        report.entries = entries_;
        report.all_healthy = (report.total_dead == 0 && report.total_degraded == 0);
        return report;
    }

private:
    std::vector<WatchdogEntry> entries_;
    std::unordered_map<std::string, std::unique_ptr<ShmFreshnessChecker>> shm_checkers_;

    int find_pid(const std::string& name) {
        // Same as ProcessMonitor::find_pid_by_name
        std::ifstream cmd("/proc/self/comm");  // simplified
        DIR* dir = opendir("/proc");
        if (!dir) return -1;
        struct dirent* entry;
        while ((entry = readdir(dir))) {
            if (entry->d_type != DT_DIR) continue;
            int pid = atoi(entry->d_name);
            if (pid <= 0) continue;
            std::ifstream comm("/proc/" + std::string(entry->d_name) + "/comm");
            std::string pname;
            if (comm >> pname && pname == name) {
                closedir(dir);
                return pid;
            }
        }
        closedir(dir);
        return -1;
    }
};
