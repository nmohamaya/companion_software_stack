// config/monitor_config.h
#pragma once
#include <string>
#include <vector>
#include <nlohmann/json.hpp>

struct MonitorConfig {
    // General
    std::string log_level        = "info";
    std::string log_dir          = "/opt/drone/logs";
    std::string blackbox_path    = "/opt/drone/data/blackbox.bin";
    size_t blackbox_max_records  = 360000;

    // Thermal thresholds
    double thermal_warning_c     = 65.0;
    double thermal_critical_c    = 80.0;
    double thermal_emergency_c   = 90.0;
    double fan_min_temp_c        = 45.0;
    double fan_max_temp_c        = 75.0;
    std::string fan_pwm_path     = "/sys/devices/pwm-fan/target_pwm";

    // Memory thresholds
    double memory_warning_pct    = 80.0;
    double memory_critical_pct   = 90.0;

    // Disk thresholds
    std::string disk_path        = "/opt/drone";
    double disk_warning_pct      = 85.0;
    double disk_cleanup_free_mb  = 500.0;

    // Network
    std::string network_iface    = "wlan0";

    // Watchdog
    struct WatchEntry {
        std::string process_name;
        std::string shm_name;
        double shm_timeout_ms = 2000.0;
    };
    std::vector<WatchEntry> watched_processes = {
        {"video_capture",   "/video_frames",      1000.0},
        {"perception",      "/detected_objects",   2000.0},
        {"slam_vio_nav",    "/slam_pose",          500.0},
        {"mission_planner", "/mission_status",     2000.0},
        {"comms",           "/fc_state",           1000.0},
        {"payload_manager", "/payload_status",     3000.0},
    };

    // Alert rate limiting
    int alert_burst_limit        = 5;
    double alert_refill_rate     = 0.5;
};
