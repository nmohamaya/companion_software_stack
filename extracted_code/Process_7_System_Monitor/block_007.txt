=== Listing 5-3: Per-Process Monitor (reads /proc/<pid>/stat) ===

// monitor/process_monitor.h
#pragma once
#include <fstream>
#include <string>
#include <sstream>
#include <unistd.h>
#include <dirent.h>

struct ProcessStats {
    std::string name;
    int pid = -1;
    bool alive = false;
    double cpu_percent = 0.0;
    uint64_t rss_kb = 0;
    uint64_t vsize_kb = 0;
    int threads = 0;
    char state = '?';
    uint64_t utime = 0;   // user jiffies
    uint64_t stime = 0;   // system jiffies
};

class ProcessMonitor {
public:
    ProcessStats update(const std::string& process_name) {
        ProcessStats ps;
        ps.name = process_name;
        ps.pid = find_pid_by_name(process_name);
        if (ps.pid < 0) { ps.alive = false; return ps; }
        ps.alive = true;

        // Read /proc/<pid>/stat
        std::ifstream f("/proc/" + std::to_string(ps.pid) + "/stat");
        if (!f) { ps.alive = false; return ps; }

        std::string line;
        std::getline(f, line);

        // Parse fields (skip past comm which may contain spaces)
        auto close_paren = line.rfind(')');
        if (close_paren == std::string::npos) return ps;

        std::istringstream rest(line.substr(close_paren + 2));
        std::string field;
        int idx = 0;
        while (rest >> field) {
            switch (idx) {
                case 0: ps.state = field[0]; break;
                case 11: ps.utime = std::stoull(field); break;
                case 12: ps.stime = std::stoull(field); break;
                case 17: ps.threads = std::stoi(field); break;
                case 20: ps.vsize_kb = std::stoull(field) / 1024; break;
                case 21: ps.rss_kb = std::stoull(field) *
                         (sysconf(_SC_PAGESIZE) / 1024); break;
            }
            ++idx;
        }

        // Compute CPU percentage
        uint64_t total_time = ps.utime + ps.stime;
        uint64_t prev_total = prev_times_[ps.pid];
        long hz = sysconf(_SC_CLK_TCK);
        ps.cpu_percent = 100.0 * double(total_time - prev_total) / double(hz);
        prev_times_[ps.pid] = total_time;

        return ps;
    }

private:
    std::unordered_map<int, uint64_t> prev_times_;

    int find_pid_by_name(const std::string& name) {
        DIR* dir = opendir("/proc");
        if (!dir) return -1;
        struct dirent* entry;
        while ((entry = readdir(dir))) {
            if (entry->d_type != DT_DIR) continue;
            int pid = atoi(entry->d_name);
            if (pid <= 0) continue;

            std::ifstream comm("/proc/" + std::string(entry->d_name) + "/comm");
            std::string pname;
            if (comm >> pname && pname == name) {
                closedir(dir);
                return pid;
            }
        }
        closedir(dir);
        return -1;
    }
};
