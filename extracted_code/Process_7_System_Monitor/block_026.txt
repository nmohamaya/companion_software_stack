=== Listing 15-1: System Health Summary (published to SHM) ===

// data/system_health.h
#pragma once
#include <cstdint>
#include <atomic>

struct SystemHealthSummary {
    // CPU
    float cpu_usage_percent;
    float cpu_core_usage[6];

    // Memory
    float memory_usage_percent;
    uint32_t memory_used_mb;
    uint32_t memory_total_mb;

    // Thermal
    float cpu_temp_c;
    float gpu_temp_c;
    float board_temp_c;
    uint8_t thermal_zone;
    float fan_duty;

    // Watchdog
    uint8_t processes_healthy;
    uint8_t processes_degraded;
    uint8_t processes_dead;
    uint8_t watchdog_ok;  // 1 = all healthy

    // Disk
    float disk_usage_percent;
    uint32_t disk_free_mb;

    // Power
    float voltage_v;
    float current_a;
    float power_w;
    float battery_percent;

    // Network
    float rx_rate_kbps;
    float tx_rate_kbps;
    uint32_t rx_errors;
    bool link_up;

    // Alerts
    uint32_t alert_count;
    uint8_t highest_alert_severity;
};

struct ShmSystemHealth {
    std::atomic<uint64_t> seq;
    uint64_t timestamp_ns;
    SystemHealthSummary health;
};
