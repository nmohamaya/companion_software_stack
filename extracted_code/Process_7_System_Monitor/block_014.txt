// monitor/disk_monitor.h
#pragma once
#include <sys/statvfs.h>
#include <string>
#include <fstream>
#include <spdlog/spdlog.h>

struct DiskSnapshot {
    std::string mount_point;
    uint64_t total_bytes       = 0;
    uint64_t available_bytes   = 0;
    uint64_t used_bytes        = 0;
    double   usage_percent     = 0.0;
    bool     low_space_warning = false;
};

class DiskMonitor {
public:
    explicit DiskMonitor(const std::string& path = "/opt/drone",
                         double warning_percent = 90.0)
        : path_(path), warning_pct_(warning_percent) {}

    DiskSnapshot update() {
        DiskSnapshot snap;
        snap.mount_point = path_;

        struct statvfs vfs;
        if (statvfs(path_.c_str(), &vfs) != 0) {
            spdlog::error("statvfs failed for {}", path_);
            return snap;
        }

        snap.total_bytes     = vfs.f_blocks * vfs.f_frsize;
        snap.available_bytes = vfs.f_bavail * vfs.f_frsize;
        snap.used_bytes      = snap.total_bytes - snap.available_bytes;

        if (snap.total_bytes > 0) {
            snap.usage_percent = 100.0 *
                double(snap.used_bytes) / double(snap.total_bytes);
        }

        snap.low_space_warning = (snap.usage_percent >= warning_pct_);

        if (snap.low_space_warning) {
            spdlog::warn("DISK WARNING: {} at {:.1f}% ({:.0f} MB free)",
                          path_, snap.usage_percent,
                          double(snap.available_bytes) / 1e6);
        }

        return snap;
    }

    // Automatic log rotation: delete oldest logs when space is low
    void cleanup_old_logs(const std::string& log_dir,
                          uint64_t target_free_bytes) {
        DiskSnapshot snap = update();
        if (snap.available_bytes >= target_free_bytes) return;

        spdlog::info("DiskMonitor: cleaning old logs in {} "
                      "(need {} MB free)", log_dir,
                      target_free_bytes / 1000000);
        // Implementation: list files sorted by mtime, delete oldest first
        // until target free space is reached

private:
    std::string path_;
    double warning_pct_;
};
