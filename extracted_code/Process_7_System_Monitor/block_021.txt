=== Listing 12-2: Telemetry Aggregator Thread (1 Hz) ===

// aggregate/telemetry_thread.cpp
#include "store/health_store.h"
#include "ipc/system_health_writer.h"
#include "alert/alert_engine.h"
#include "util/realtime.h"

void telemetry_aggregator_thread(HealthStore& store,
                                  SystemHealthWriter& writer,
                                  AlertEngine& alerts,
                                  std::atomic<bool>& running) {
    set_thread_params("TelAggr", 5, SCHED_OTHER, 0);
    const auto period = std::chrono::seconds(1);  // 1 Hz

    while (running.load(std::memory_order_relaxed)) {
        auto start = std::chrono::steady_clock::now();

        HealthSnapshot snap = store.get_snapshot();

        // Publish to SHM
        writer.write(snap);

        // Evaluate alert conditions
        if (snap.thermal.zone >= ThermalZone::CRITICAL)
            alerts.emit(AlertSeverity::CRITICAL, "thermal",
                        "Temperature critical: " +
                        std::to_string(int(snap.thermal.max_temp_c)) + "°C");

        if (snap.memory.usage_percent() > 90.0)
            alerts.emit(AlertSeverity::WARNING, "memory",
                        "RAM usage at " +
                        std::to_string(int(snap.memory.usage_percent())) + "%");

        if (snap.watchdog.total_dead > 0)
            alerts.emit(AlertSeverity::CRITICAL, "watchdog",
                        std::to_string(snap.watchdog.total_dead) +
                        " process(es) dead");

        if (snap.disk.low_space_warning)
            alerts.emit(AlertSeverity::WARNING, "disk",
                        "Disk space at " +
                        std::to_string(int(snap.disk.usage_percent)) + "%");

        if (snap.power.critical_voltage)
            alerts.emit(AlertSeverity::EMERGENCY, "power",
                        "Voltage critical: " +
                        std::to_string(snap.power.voltage_v) + "V");

        std::this_thread::sleep_until(start + period);
    }
}
