// file: include/video/frame.h
#pragma once
#include <cstdint>
#include <cstring>

namespace drone::video {

enum class PixelFormat : uint32_t {
    BAYER_RGGB8  = 0,
    BAYER_GRBG8  = 1,
    YUV422       = 2,
    RGB24        = 3,
    NV12         = 4,   // for HW encoder input
    GRAY8        = 5,   // mono for SLAM
};

enum class CameraId : uint8_t {
    MISSION   = 0,
    STEREO_L  = 1,
    STEREO_R  = 2,
};

struct FrameMetadata {
    uint64_t timestamp_ns;      // CLOCK_MONOTONIC — consistent across threads
    uint64_t sequence_number;   // monotonic frame counter
    CameraId camera_id;
    PixelFormat format;
    uint32_t width;
    uint32_t height;
    uint32_t stride;            // bytes per row (includes padding)
    uint32_t exposure_us;       // actual exposure time — useful for SLAM
    float    gain_db;           // actual analog gain
    float    temperature_c;     // sensor temperature (for thermal drift compensation)
};

} // namespace drone::video
