#include "video/frame.h"
#include "video/spsc_ring.h"
#include <spdlog/spdlog.h>
#include <atomic>
#include <fstream>
#include <filesystem>
#include <chrono>

namespace drone::video {

/// Simple recording session — writes encoded frames to MKV container.
/// In production, use libavformat (FFmpeg) for proper MKV muxing.
class MissionRecorder {
public:
    explicit MissionRecorder(const std::string& output_dir) {
        namespace fs = std::filesystem;
        fs::create_directories(output_dir);

        // Generate filename from timestamp
        auto now = std::chrono::system_clock::now();
        auto t   = std::chrono::system_clock::to_time_t(now);
        char buf[64];
        std::strftime(buf, sizeof(buf), "%Y%m%d_%H%M%S", std::localtime(&t));

        video_path_    = output_dir + "/" + buf + "_mission.mkv";
        metadata_path_ = output_dir + "/" + buf + "_metadata.csv";

        // Open metadata sidecar
        metadata_file_.open(metadata_path_);
        metadata_file_ << "frame_seq,timestamp_ns,exposure_us,gain_db\n";

        // In production: open MKV via avformat_alloc_output_context2()
        spdlog::info("Recording to: {}", video_path_);
    }

    ~MissionRecorder() {
        // Flush and finalize container
        metadata_file_.close();
        spdlog::info("Recording finalized: {} frames, {:.1f} MB",
                     frame_count_, bytes_written_ / (1024.0 * 1024.0));
    }

    void write_frame(const EncoderFrame& ef) {
        // In production: use av_write_frame() to mux into MKV
        // Here showing the metadata sidecar write:
        metadata_file_ << ef.meta.sequence_number << ","
                       << ef.meta.timestamp_ns << ","
                       << ef.meta.exposure_us << ","
                       << ef.meta.gain_db << "\n";

        // Track stats
        bytes_written_ += ef.image.total() * ef.image.elemSize();
        ++frame_count_;

        // Flush metadata every 100 frames (not every frame — too slow)
        if (frame_count_ % 100 == 0) {
            metadata_file_.flush();
        }
    }

    /// Check if disk space is running low
    bool disk_space_ok() const {
        namespace fs = std::filesystem;
        auto info = fs::space("/data");
        size_t free_mb = info.available / (1024 * 1024);
        return free_mb > 500;  // keep at least 500MB free
    }

private:
    std::string video_path_;
    std::string metadata_path_;
    std::ofstream metadata_file_;
    uint64_t frame_count_ = 0;
    uint64_t bytes_written_ = 0;
};

/// Recorder thread entry point
void recorder_thread(
    SPSCRing<EncoderFrame, 4>& input_queue,  // from ISP thread
    std::atomic<bool>& stop_flag,
    const std::string& recording_dir = "/data/recordings")
{
    MissionRecorder recorder(recording_dir);

    while (!stop_flag.load(std::memory_order_relaxed)) {
        auto frame_opt = input_queue.try_pop();
        if (!frame_opt) {
            std::this_thread::sleep_for(std::chrono::milliseconds(1));
            continue;
        }

        // Check disk space periodically
        if (!recorder.disk_space_ok()) {
            spdlog::warn("SSD low on space — dropping recording frames");
            // Could also implement circular buffer: delete oldest segment
            continue;
        }

        recorder.write_frame(*frame_opt);
    }

    // Destructor will finalize the recording
    spdlog::info("Recorder thread stopped");
}

} // namespace drone::video
