# /etc/systemd/system/video-capture.service

[Unit]
Description=Drone Video Capture & Streaming
After=network.target
# Start after hardware is ready
After=dev-video0.device
Wants=dev-video0.device

[Service]
Type=notify                         # Process calls sd_notify(READY=1)
ExecStart=/opt/drone/bin/video-capture
User=drone
Group=video                         # Access to /dev/videoX without root

# Resource limits
CPUAffinity=2 3                     # Pin to cores 2-3
MemoryMax=512M
MemoryHigh=384M
CPUQuota=200%                       # Max 2 cores worth of CPU

# Restart policy
Restart=always
RestartSec=200ms                    # Restart within 200ms
StartLimitIntervalSec=60
StartLimitBurst=10                  # Max 10 restarts per minute

# Watchdog: process must send WATCHDOG=1 every 5s or it's killed
WatchdogSec=5

# Security hardening
ProtectSystem=strict
PrivateTmp=true
ReadWritePaths=/data/recordings     # Only writable path (for SSD recording)
NoNewPrivileges=true

# Shared memory permissions
ReadWritePaths=/dev/shm             # For shm_open()

# Namespace for device access
PrivateDevices=false                # Need access to /dev/videoX and /dev/nvhost-*
DeviceAllow=/dev/video0 rw
DeviceAllow=/dev/video2 rw
DeviceAllow=/dev/video3 rw

[Install]
WantedBy=multi-user.target
