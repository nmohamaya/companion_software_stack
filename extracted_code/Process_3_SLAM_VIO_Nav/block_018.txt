Context: The ShmPoseSlot uses a SeqLock – a sequence number that is odd during writes and even when stable. Readers spin-read the sequence, copy the data, then verify the sequence hasn't changed. This is lock-free, wait-free for writers, and allows multiple readers (Process 4 and Process 5) without any mutex. It works because there is exactly one writer (Map Manager thread).

// Reader side (in Process 4 or Process 5):
bool read_pose(const ShmPoseSlot* slot, PoseStamped& out) {
    for (int attempt = 0; attempt < 3; ++attempt) {
        uint64_t seq1 = slot->sequence.load(std::memory_order_acquire);
        if (seq1 & 1) continue;  // writer is mid-update, retry

        // Copy data
        out.timestamp = slot->timestamp;
        out.position  = Eigen::Map<const Eigen::Vector3d>(slot->translation);
        out.orientation = Eigen::Quaterniond(
            slot->quaternion[0], slot->quaternion[1],
            slot->quaternion[2], slot->quaternion[3]);

        std::atomic_thread_fence(std::memory_order_acquire);
        uint64_t seq2 = slot->sequence.load(std::memory_order_relaxed);
        if (seq1 == seq2) return true;  // consistent read
    }
    return false;  // failed after retries (write was in progress)
}
