// /dev/shm/slam_pose – created by Process 3 at startup
// Layout: single ShmPoseSlot (see Section 7.3)

void MapManager::create_pose_shm() {
    int fd = shm_open("/slam_pose", O_CREAT | O_RDWR, 0666);
    ftruncate(fd, sizeof(ShmPoseSlot));
    auto* slot = static_cast<ShmPoseSlot*>(
        mmap(nullptr, sizeof(ShmPoseSlot), PROT_READ | PROT_WRITE,
             MAP_SHARED, fd, 0));
    slot->sequence.store(0, std::memory_order_relaxed);
    pose_publisher_ = std::make_unique<PosePublisher>(slot);
    close(fd);  // mapping persists after close
}
