Context: Eigen::Map wraps raw memory as an Eigen vector/matrix without copying data. This is used extensively when reading from shared memory (Section 7.3) and when converting between C arrays and Eigen types for GTSAM/Ceres.
Source: code_paragraphs
---
// Wrap raw double[3] in shared memory as Vector3d (zero-copy)
double raw[3] = {1.0, 2.0, 3.0};
Eigen::Map<Eigen::Vector3d> v(raw);
v.normalize();  // modifies raw[] directly

// Const map for read-only access:
const double* pose_data = shm_slot->translation;
Eigen::Map<const Eigen::Vector3d> pos(pose_data);

// Map a contiguous buffer as a matrix:
double cov_data[36];
Eigen::Map<Eigen::Matrix<double,6,6,Eigen::RowMajor>> cov(cov_data);
