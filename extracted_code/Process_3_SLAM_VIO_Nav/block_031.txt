// In main()
int main(int argc, char* argv[]) {
    auto config = load_config(argv);

    // Open inputs
    auto shm_reader = ShmFrameReader("/video_capture_ring", "/video_sem", 4);
    auto imu_driver = SpiImuDriver("/dev/spidev0.0", 8'000'000);

    // Create output pose shared memory
    create_pose_shm("/slam_pose");

    // Load calibration
    auto cam_params = load_camera_params(config.camera_yaml);
    auto imu_params = load_imu_params(config.imu_yaml);
    auto T_BC = load_extrinsics(config.extrinsics_yaml);

    // Optionally load saved map
    std::optional<Map> prior_map;
    if (std::filesystem::exists(config.map_path))
        prior_map = load_map(config.map_path);

    // Create and start threads
    SlamSystem slam(shm_reader, imu_driver, cam_params, imu_params, T_BC, prior_map);
    slam.start();

    // Signal systemd: we are ready
    sd_notify(0, "READY=1");

    // Wait for shutdown signal
    sigset_t sigs;
    sigemptyset(&sigs);
    sigaddset(&sigs, SIGTERM);
    sigaddset(&sigs, SIGINT);
    sigprocmask(SIG_BLOCK, &sigs, nullptr);
    int sig;
    sigwait(&sigs, &sig);

    slam.stop();
    save_map(config.map_path, slam.get_map());
    sd_notify(0, "STOPPING=1");
    return 0;
}
