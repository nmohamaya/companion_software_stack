Context: When the SLAM system accepts measurements from different sensor types (camera, IMU, barometer, GPS), std::variant provides a type-safe union. std::visit with a visitor struct ensures exhaustive handling of all types at compile time – if you add a new sensor type but forget to handle it, the code won't compile.

struct CameraMeasurement { double timestamp; std::vector<cv::KeyPoint> keypoints; };
struct ImuMeasurement     { double timestamp; Eigen::Vector3d accel, gyro; };
struct BaroMeasurement    { double timestamp; double altitude_m; };
struct GpsMeasurement     { double timestamp; double lat, lon, alt; double hdop; };

using SensorMeasurement = std::variant<
    CameraMeasurement, ImuMeasurement, BaroMeasurement, GpsMeasurement>;

// Visitor pattern with std::visit
struct MeasurementHandler {
    BackendOptimiser& backend;

    void operator()(const CameraMeasurement& m) { backend.add_visual(m); }
    void operator()(const ImuMeasurement& m)    { backend.add_imu(m); }
    void operator()(const BaroMeasurement& m)   { backend.add_baro(m); }
    void operator()(const GpsMeasurement& m)    { backend.add_gps(m); }
};

// Usage:
SensorMeasurement meas = imu_queue.pop();
std::visit(MeasurementHandler{backend}, meas);
