cmake_minimum_required(VERSION 3.18)
project(slam_vio_nav LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_FLAGS_RELEASE "-O3 -march=armv8.2-a+fp16+dotprod -flto -DNDEBUG")

# ── Dependencies ──
find_package(OpenCV 4.5 REQUIRED COMPONENTS core imgproc features2d calib3d)
find_package(Eigen3 3.3 REQUIRED NO_MODULE)
find_package(GTSAM REQUIRED)
find_package(Sophus REQUIRED)
find_package(spdlog REQUIRED)

# DBoW2 (build from source or find_package)
find_package(DBoW2 QUIET)
if(NOT DBoW2_FOUND)
    add_subdirectory(third_party/DBoW2)
endif()

# cereal (header-only)
find_package(cereal REQUIRED)

# systemd for sd_notify
find_package(PkgConfig REQUIRED)
pkg_check_modules(SYSTEMD REQUIRED libsystemd)

# ── Sources ──
set(SOURCES
    src/main.cpp
    src/visual_frontend.cpp
    src/imu_reader.cpp
    src/backend_optimiser.cpp
    src/loop_closure.cpp
    src/map_manager.cpp
    src/feature_detectors/orb_detector.cpp
    src/feature_detectors/klt_tracker.cpp
    src/ipc/shm_frame_reader.cpp
    src/ipc/pose_publisher.cpp
    src/calibration/calib_loader.cpp
    src/utils/spsc_queue.hpp
)

add_executable(${PROJECT_NAME} ${SOURCES})

target_include_directories(${PROJECT_NAME} PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/include
    ${SYSTEMD_INCLUDE_DIRS}
)

target_link_libraries(${PROJECT_NAME} PRIVATE
    OpenCV::core OpenCV::imgproc OpenCV::features2d OpenCV::calib3d
    Eigen3::Eigen
    gtsam
    Sophus::Sophus
    spdlog::spdlog
    DBoW2
    cereal::cereal
    ${SYSTEMD_LIBRARIES}
    pthread rt
)

# ── Install ──
install(TARGETS ${PROJECT_NAME} DESTINATION /opt/drone/bin)
install(DIRECTORY config/ DESTINATION /opt/drone/config
        FILES_MATCHING PATTERN "*.yaml")
