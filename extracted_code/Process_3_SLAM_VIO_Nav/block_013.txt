void BackendOptimiser::run() {
    while (!stop_flag_.load(std::memory_order_relaxed)) {
        // Block until a keyframe arrives
        KeyframeData kf;
        if (!keyframe_queue_.pop_wait(kf, std::chrono::milliseconds(100)))
            continue;

        // Grab the corresponding IMU pre-integration factor
        PreintegratedImuFactor imu_factor;
        imu_factor_queue_.pop_wait(imu_factor, std::chrono::milliseconds(50));

        add_keyframe(kf, imu_factor);

        // Notify loop closure thread (new keyframe to check)
        loop_closure_event_.signal();

        // Update IMU thread with latest bias estimates
        auto bias = isam2_.calculateEstimate<gtsam::imuBias::ConstantBias>(
            B(kf_id_));
        bias_update_.store({gtsam_to_eigen(bias.accelerometer()),
                            gtsam_to_eigen(bias.gyroscope())});

        sd_notify(0, "WATCHDOG=1");  // heartbeat
    }
}
