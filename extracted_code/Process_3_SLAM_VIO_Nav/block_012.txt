Context: GTSAM uses Symbol keys like X(i), V(i), B(i) to identify pose, velocity, and bias nodes in the graph. This pattern prevents accidentally connecting a pose factor to a bias node. In your own code, you can achieve similar type-safety using strong typedefs or enum class indices.

// GTSAM's approach – character + index:
gtsam::Key pose_key  = gtsam::Symbol('x', 42);  // X(42)
gtsam::Key vel_key   = gtsam::Symbol('v', 42);  // V(42)
gtsam::Key bias_key  = gtsam::Symbol('b', 42);  // B(42)

// Custom approach using strong typedefs:
enum class NodeType : char { Pose = 'x', Velocity = 'v', Bias = 'b' };

struct GraphKey {
    NodeType type;
    uint64_t index;
    bool operator==(const GraphKey& o) const = default;
};

struct GraphKeyHash {
    size_t operator()(const GraphKey& k) const {
        return std::hash<uint64_t>{}(
            (static_cast<uint64_t>(k.type) << 56) | k.index);
    }
};
