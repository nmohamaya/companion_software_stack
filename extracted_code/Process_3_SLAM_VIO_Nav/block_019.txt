#include <cereal/archives/binary.hpp>
#include <cereal/types/vector.hpp>
#include <cereal/types/map.hpp>

struct SerializableMap {
    std::vector<KeyframePose> keyframes;
    std::vector<Landmark3D> landmarks;
    std::vector<BowVector> bow_vectors;
    std::string vocabulary_hash;  // to verify vocabulary compatibility

    template <class Archive>
    void serialize(Archive& ar) {
        ar(keyframes, landmarks, bow_vectors, vocabulary_hash);
    }
};

void MapManager::save_map(const std::string& path) {
    SerializableMap smap = build_serializable();
    std::ofstream ofs(path, std::ios::binary);
    cereal::BinaryOutputArchive archive(ofs);
    archive(smap);
    spdlog::info("Map saved: {} KFs, {} landmarks → {}",
                 smap.keyframes.size(), smap.landmarks.size(), path);
}
