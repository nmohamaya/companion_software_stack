// planning/astar_planner.h
#pragma once
#include "planning/planning_strategy.h"
#include <queue>
#include <unordered_map>

class AStarPlanner : public PlanningStrategy {
public:
    std::optional<PathResult> plan(
        const Eigen::Vector3d& start,
        const Eigen::Vector3d& goal,
        const OccupancyGrid& grid,
        double timeout_ms = 200.0) override
    {
        auto t0 = std::chrono::steady_clock::now();
        PathResult result;

        // Convert to grid coordinates
        auto si = grid.to_index(start);
        auto gi = grid.to_index(goal);
        if (!si || !gi) return std::nullopt;

        // Priority queue: (f-cost, node_index)
        using Node = std::pair<double, GridIndex>;
        std::priority_queue<Node, std::vector<Node>, std::greater<>> open;
        std::unordered_map<GridIndex, double, GridIndexHash> g_cost;
        std::unordered_map<GridIndex, GridIndex, GridIndexHash> came_from;

        open.push({heuristic(*si, *gi), *si});
        g_cost[*si] = 0.0;

        while (!open.empty()) {
            // Timeout check
            auto elapsed = std::chrono::steady_clock::now() - t0;
            if (std::chrono::duration<double, std::milli>(elapsed).count() > timeout_ms) {
                result.is_partial = true;
                break;
            }

            auto [f, current] = open.top();
            open.pop();

            if (current == *gi) {
                result.waypoints = reconstruct_path(came_from, *gi, grid);
                result.total_cost = g_cost[*gi];
                result.is_partial = false;
                result.compute_time_ms =
                    std::chrono::duration<double, std::milli>(
                        std::chrono::steady_clock::now() - t0).count();
                return result;
            }

            for (const auto& nb : grid.neighbors_26(current)) {
                if (grid.is_occupied(nb)) continue;

                double tentative_g = g_cost[current] + grid.edge_cost(current, nb);
                if (g_cost.find(nb) == g_cost.end() || tentative_g < g_cost[nb]) {
                    g_cost[nb] = tentative_g;
                    came_from[nb] = current;
                    open.push({tentative_g + heuristic(nb, *gi), nb});
                }
            }
        }

        return result.is_partial ? std::optional(result) : std::nullopt;
    }

    std::string name() const override { return "A*"; }

private:
    double heuristic(const GridIndex& a, const GridIndex& b) const {
        // Octile distance (3D)
        int dx = std::abs(a.x - b.x);
        int dy = std::abs(a.y - b.y);
        int dz = std::abs(a.z - b.z);
        // Sort dx >= dy >= dz
        if (dx < dy) std::swap(dx, dy);
        if (dx < dz) std::swap(dx, dz);
        if (dy < dz) std::swap(dy, dz);
        return dx + (std::sqrt(2.0) - 1) * dy + (std::sqrt(3.0) - std::sqrt(2.0)) * dz;
    }

    std::vector<Eigen::Vector3d> reconstruct_path(
        const std::unordered_map<GridIndex, GridIndex, GridIndexHash>& came_from,
        GridIndex goal, const OccupancyGrid& grid);
};
