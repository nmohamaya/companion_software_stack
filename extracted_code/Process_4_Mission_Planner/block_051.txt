=== Listing 15-1: Trajectory Command SHM Writer (SeqLock) ===

// ipc/trajectory_writer.h
#pragma once
#include "trajectory/trajectory_generator.h"
#include <atomic>
#include <sys/mman.h>
#include <fcntl.h>
#include <unistd.h>

struct ShmTrajectoryCmd {
    std::atomic<uint64_t> seq;       // SeqLock
    uint64_t timestamp_ns;
    double position[3];              // NED metres
    double velocity[3];              // m/s
    double acceleration[3];          // m/s²
    double yaw;                      // radians
    double yaw_rate;                 // rad/s
    uint16_t type_mask;              // MAVLink type_mask for SET_POSITION_TARGET
    uint8_t  coordinate_frame;       // MAV_FRAME_LOCAL_NED
    uint8_t  padding[5];
};

class TrajectoryWriter {
public:
    bool create(const char* name = "/trajectory_cmd") {
        fd_ = shm_open(name, O_CREAT | O_RDWR, 0666);
        if (fd_ < 0) return false;
        ftruncate(fd_, sizeof(ShmTrajectoryCmd));
        ptr_ = static_cast<ShmTrajectoryCmd*>(
            mmap(nullptr, sizeof(ShmTrajectoryCmd),
                 PROT_READ | PROT_WRITE, MAP_SHARED, fd_, 0));
        if (ptr_ == MAP_FAILED) return false;
        ptr_->seq.store(0);
        return true;
    }

    void write(const TrajectoryPoint& pt) {
        if (!ptr_) return;
        // SeqLock: increment to odd (write in progress)
        uint64_t s = ptr_->seq.load(std::memory_order_relaxed);
        ptr_->seq.store(s + 1, std::memory_order_release);

        ptr_->timestamp_ns = current_time_ns();
        Eigen::Map<Eigen::Vector3d>(ptr_->position) = pt.position;
        Eigen::Map<Eigen::Vector3d>(ptr_->velocity) = pt.velocity;
        Eigen::Map<Eigen::Vector3d>(ptr_->acceleration) = pt.acceleration;
        ptr_->yaw = pt.yaw;
        ptr_->yaw_rate = pt.yaw_rate;
        ptr_->type_mask = 0b0000'0000'0000'0000;  // use all fields
        ptr_->coordinate_frame = 1;  // MAV_FRAME_LOCAL_NED

        // Increment to even (write complete)
        ptr_->seq.store(s + 2, std::memory_order_release);
    }

    ~TrajectoryWriter() {
        if (ptr_ && ptr_ != MAP_FAILED) munmap(ptr_, sizeof(ShmTrajectoryCmd));
        if (fd_ >= 0) { shm_unlink("/trajectory_cmd"); close(fd_); }
    }

private:
    int fd_ = -1;
    ShmTrajectoryCmd* ptr_ = nullptr;
    uint64_t current_time_ns() const;
};
