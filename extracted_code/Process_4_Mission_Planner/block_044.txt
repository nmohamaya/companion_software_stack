// data/mission_status.h
#pragma once
#include <cstdint>
#include <atomic>

enum class MissionPhase : uint8_t {
    IDLE, ARMED, TAKEOFF, CRUISE, LOITER, RTL, LANDING, LANDED, EMERGENCY
};

struct MissionStatus {
    MissionPhase  phase;
    uint32_t      current_wp_index;
    uint32_t      total_wp_count;
    float         distance_to_wp;       // metres
    float         eta_seconds;          // estimated time of arrival
    float         battery_pct;
    float         ground_speed;         // m/s
    float         altitude_agl;         // metres
    uint64_t      mission_elapsed_ms;   // since takeoff
    uint8_t       gps_fix;              // 0=none, 3=3D, 5=RTK
    bool          geofence_ok;
    bool          obstacle_detected;
    float         closest_obstacle_m;
};
