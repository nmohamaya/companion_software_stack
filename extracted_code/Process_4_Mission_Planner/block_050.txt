// util/scoped_timer.h
#pragma once
#include <chrono>
#include <string>
#include <spdlog/spdlog.h>

class ScopedTimer {
public:
    explicit ScopedTimer(std::string label, double warn_ms = 0.0)
        : label_(std::move(label)), warn_ms_(warn_ms),
          start_(std::chrono::steady_clock::now()) {}

    ~ScopedTimer() {
        auto elapsed = std::chrono::steady_clock::now() - start_;
        double ms = std::chrono::duration<double, std::milli>(elapsed).count();
        if (warn_ms_ > 0 && ms > warn_ms_) {
            spdlog::warn("{}: {:.2f} ms (threshold: {:.1f} ms)", label_, ms, warn_ms_);
        } else {
            spdlog::trace("{}: {:.2f} ms", label_, ms);
        }
    }

private:
    std::string label_;
    double warn_ms_;
    std::chrono::steady_clock::time_point start_;
};
