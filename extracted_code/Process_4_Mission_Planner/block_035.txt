=== Listing 9-2: Trapezoidal Velocity Profile Generator ===

// trajectory/trapezoidal_profile.h
#pragma once
#include "trajectory/trajectory_generator.h"

class TrapezoidalProfile
    : public TrajectoryGeneratorBase<TrapezoidalProfile> {
public:
    void do_generate(const std::vector<Eigen::Vector3d>& waypoints,
                     double max_vel, double max_accel) {
        segments_.clear();
        total_time_ = 0.0;

        for (size_t i = 0; i + 1 < waypoints.size(); ++i) {
            Segment seg;
            seg.start = waypoints[i];
            seg.end   = waypoints[i + 1];
            Eigen::Vector3d diff = seg.end - seg.start;
            seg.distance = diff.norm();
            seg.direction = diff.normalized();

            // Trapezoidal velocity profile: accel → cruise → decel
            double t_accel = max_vel / max_accel;       // time to accelerate
            double d_accel = 0.5 * max_accel * t_accel * t_accel;

            if (2 * d_accel >= seg.distance) {
                // Triangular profile (never reaches max_vel)
                t_accel = std::sqrt(seg.distance / max_accel);
                seg.t_accel = t_accel;
                seg.t_cruise = 0.0;
                seg.t_decel = t_accel;
                seg.peak_vel = max_accel * t_accel;
            } else {
                double d_cruise = seg.distance - 2 * d_accel;
                double t_cruise = d_cruise / max_vel;
                seg.t_accel = t_accel;
                seg.t_cruise = t_cruise;
                seg.t_decel = t_accel;
                seg.peak_vel = max_vel;
            }

            seg.t_start = total_time_;
            seg.total_time = seg.t_accel + seg.t_cruise + seg.t_decel;
            total_time_ += seg.total_time;
            segments_.push_back(seg);
        }
    }

    TrajectoryPoint do_evaluate(double t) const {
        TrajectoryPoint pt;
        pt.timestamp = t;

        // Find active segment
        for (const auto& seg : segments_) {
            double local_t = t - seg.t_start;
            if (local_t < 0) continue;
            if (local_t > seg.total_time && &seg != &segments_.back()) continue;

            local_t = std::clamp(local_t, 0.0, seg.total_time);

            double s, v, a;  // scalar position, velocity, acceleration
            if (local_t < seg.t_accel) {
                // Acceleration phase
                a = seg.peak_vel / seg.t_accel;
                v = a * local_t;
                s = 0.5 * a * local_t * local_t;
            } else if (local_t < seg.t_accel + seg.t_cruise) {
                // Cruise phase
                double dt = local_t - seg.t_accel;
                a = 0.0;
                v = seg.peak_vel;
                s = 0.5 * seg.peak_vel * seg.t_accel + seg.peak_vel * dt;
            } else {
                // Deceleration phase
                double dt = local_t - seg.t_accel - seg.t_cruise;
                a = -seg.peak_vel / seg.t_decel;
                v = seg.peak_vel + a * dt;
                s = 0.5 * seg.peak_vel * seg.t_accel
                  + seg.peak_vel * seg.t_cruise
                  + seg.peak_vel * dt + 0.5 * a * dt * dt;
            }

            pt.position = seg.start + seg.direction * s;
            pt.velocity = seg.direction * v;
            pt.acceleration = seg.direction * a;
            pt.yaw = std::atan2(seg.direction.y(), seg.direction.x());
            pt.yaw_rate = 0.0;
            return pt;
        }

        // Past end — hold last position
        pt.position = segments_.back().end;
        pt.velocity = Eigen::Vector3d::Zero();
        pt.acceleration = Eigen::Vector3d::Zero();
        return pt;
    }

    double do_duration() const { return total_time_; }

private:
    struct Segment {
        Eigen::Vector3d start, end, direction;
        double distance, peak_vel;
        double t_accel, t_cruise, t_decel;
        double t_start, total_time;
    };
    std::vector<Segment> segments_;
    double total_time_ = 0.0;
};
