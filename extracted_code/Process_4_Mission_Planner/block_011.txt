// mission/states/cruise_state.h
#pragma once
#include "mission/mission_state.h"
#include "mission/mission_context.h"

class CruiseState : public MissionState {
public:
    void on_enter(MissionContext& ctx) override {
        spdlog::info("Entering CRUISE state");
        ctx.set_planning_active(true);
    }

    void update(MissionContext& ctx) override {
        // Check waypoint completion
        if (ctx.waypoint_manager().current_reached()) {
            if (ctx.waypoint_manager().advance()) {
                spdlog::info("Advanced to waypoint {}",
                             ctx.waypoint_manager().current_index());
            }
        }

        // Monitor battery, geofence, operator commands
        check_safety_conditions(ctx);
    }

    void on_exit(MissionContext& ctx) override {
        ctx.set_planning_active(false);
    }

    std::string name() const override { return "CRUISE"; }

    std::unique_ptr<MissionState> transition(MissionContext& ctx) override {
        if (ctx.has_command(MissionCommand::RTL))
            return std::make_unique<RTLState>();
        if (ctx.has_command(MissionCommand::LOITER))
            return std::make_unique<LoiterState>();
        if (ctx.battery_critical())
            return std::make_unique<RTLState>();
        if (ctx.geofence_breached())
            return std::make_unique<RTLState>();
        if (ctx.waypoint_manager().mission_complete())
            return std::make_unique<RTLState>();
        return nullptr;  // remain in CRUISE
    }

private:
    void check_safety_conditions(MissionContext& ctx);
};
