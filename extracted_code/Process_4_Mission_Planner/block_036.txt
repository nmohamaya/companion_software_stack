=== Listing 9-3: Trajectory Executor Thread (50 Hz) ===

// trajectory/trajectory_thread.cpp
#include "trajectory/trapezoidal_profile.h"
#include "mission/mission_context.h"
#include "util/realtime.h"

void trajectory_executor_thread(MissionContext& ctx,
                                 std::atomic<bool>& running) {
    set_thread_params("TrajExec", 5, SCHED_FIFO, 80);

    TrapezoidalProfile traj;
    const auto period = std::chrono::milliseconds(20);  // 50 Hz
    bool trajectory_valid = false;
    double traj_start_time = 0.0;

    while (running.load(std::memory_order_relaxed)) {
        auto loop_start = std::chrono::steady_clock::now();

        if (!ctx.planning_active()) {
            trajectory_valid = false;
            std::this_thread::sleep_for(period);
            continue;
        }

        // Check for new planned path
        auto new_path = ctx.consume_planned_path();
        if (new_path && new_path->size() >= 2) {
            // Apply avoidance adjustments
            double heading_adj, speed_factor;
            ctx.get_avoidance_output(heading_adj, speed_factor);

            double max_vel = ctx.config().max_velocity * speed_factor;
            double max_acc = ctx.config().max_acceleration;

            traj.generate(*new_path, max_vel, max_acc);
            traj_start_time = now_seconds();
            trajectory_valid = true;
        }

        if (trajectory_valid) {
            double t = now_seconds() - traj_start_time;

            if (t > traj.duration()) {
                // Trajectory complete — notify waypoint manager
                trajectory_valid = false;
                ctx.waypoint_manager().check_reached(
                    traj.evaluate(traj.duration()).position);
                continue;
            }

            auto pt = traj.evaluate(t);

            // Publish to SHM for Process 5
            ctx.publish_trajectory_cmd(pt);
        }

        std::this_thread::sleep_until(loop_start + period);
    }
}
