=== Listing 15-2: Mission Status SHM Writer ===

// ipc/mission_status_writer.h
#pragma once
#include "data/mission_status.h"
#include <atomic>
#include <sys/mman.h>

struct ShmMissionStatus {
    std::atomic<uint64_t> seq;
    uint64_t timestamp_ns;
    MissionStatus status;
};

class MissionStatusWriter {
public:
    bool create(const char* name = "/mission_status");

    void write(const MissionStatus& status) {
        if (!ptr_) return;
        uint64_t s = ptr_->seq.load(std::memory_order_relaxed);
        ptr_->seq.store(s + 1, std::memory_order_release);

        ptr_->timestamp_ns = current_time_ns();
        ptr_->status = status;

        ptr_->seq.store(s + 2, std::memory_order_release);
    }

    ~MissionStatusWriter();  // munmap + shm_unlink + close

private:
    int fd_ = -1;
    ShmMissionStatus* ptr_ = nullptr;
};
