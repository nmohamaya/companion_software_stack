=== Listing 4-1: Lock-Free Pose Reader (SeqLock) ===

// ipc/pose_reader.h
#pragma once
#include <cstring>
#include <atomic>
#include <sys/mman.h>
#include <fcntl.h>
#include <unistd.h>
#include <Eigen/Core>
#include <Eigen/Geometry>

struct ShmPoseStamped {
    std::atomic<uint64_t> seq;       // SeqLock sequence counter
    uint64_t timestamp_ns;
    double position[3];              // x, y, z  (NED frame, metres)
    double quaternion[4];            // w, x, y, z
    double velocity[3];              // vx, vy, vz (m/s)
    float  covariance[6];            // diagonal of position covariance
    uint8_t quality;                 // 0-100 pose confidence
    uint8_t padding[7];
};

class PoseReader {
public:
    bool open(const char* name = "/slam_pose") {
        fd_ = shm_open(name, O_RDONLY, 0);
        if (fd_ < 0) return false;
        ptr_ = static_cast<ShmPoseStamped*>(
            mmap(nullptr, sizeof(ShmPoseStamped), PROT_READ, MAP_SHARED, fd_, 0));
        return ptr_ != MAP_FAILED;
    }

    bool read(Eigen::Vector3d& pos, Eigen::Quaterniond& quat,
              Eigen::Vector3d& vel, uint8_t& quality) const {
        if (!ptr_) return false;
        for (int attempt = 0; attempt < 4; ++attempt) {
            uint64_t s1 = ptr_->seq.load(std::memory_order_acquire);
            if (s1 & 1) continue;  // writer in progress
            // Copy fields
            pos = Eigen::Map<const Eigen::Vector3d>(ptr_->position);
            quat = Eigen::Quaterniond(ptr_->quaternion[0], ptr_->quaternion[1],
                                       ptr_->quaternion[2], ptr_->quaternion[3]);
            vel   = Eigen::Map<const Eigen::Vector3d>(ptr_->velocity);
            quality = ptr_->quality;
            std::atomic_thread_fence(std::memory_order_acquire);
            uint64_t s2 = ptr_->seq.load(std::memory_order_relaxed);
            if (s1 == s2) return true;  // consistent read
        }
        return false;  // torn read after 4 attempts
    }

    ~PoseReader() {
        if (ptr_ && ptr_ != MAP_FAILED) munmap(ptr_, sizeof(ShmPoseStamped));
        if (fd_ >= 0) close(fd_);
    }

private:
    int fd_ = -1;
    ShmPoseStamped* ptr_ = nullptr;
};
