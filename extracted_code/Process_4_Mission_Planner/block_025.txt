=== Listing 7-2: Obstacle Avoidance Thread (20 Hz) ===

// avoidance/avoidance_thread.cpp
#include "avoidance/vfh_plus.h"
#include "mission/mission_context.h"
#include "util/realtime.h"

void obstacle_avoidance_thread(MissionContext& ctx,
                                std::atomic<bool>& running) {
    set_thread_params("OAvoid", 5, SCHED_FIFO, 70);

    VFHPlus vfh;
    const auto period = std::chrono::milliseconds(50);  // 20 Hz

    while (running.load(std::memory_order_relaxed)) {
        auto start = std::chrono::steady_clock::now();

        if (!ctx.planning_active()) {
            std::this_thread::sleep_for(period);
            continue;
        }

        // Get current pose
        Eigen::Vector3d pos;
        Eigen::Quaterniond quat;
        Eigen::Vector3d vel;
        if (!ctx.get_pose(pos, quat, vel)) {
            std::this_thread::sleep_for(period);
            continue;
        }

        // Get desired heading from planned path
        double desired_heading = ctx.get_desired_heading();
        double current_heading = yaw_from_quaternion(quat);

        // Get detected objects
        std::vector<DetectedObject> objects;
        ctx.object_reader().read(objects);

        // Run VFH+
        auto result = vfh.compute(pos, current_heading, desired_heading, objects);

        if (result.emergency_stop) {
            spdlog::error("EMERGENCY STOP — obstacle at {:.1f}m",
                           result.closest_obstacle_m);
            ctx.push_command(MissionCommand::EMERGENCY_STOP);
        }

        // Publish safe heading and speed to trajectory executor
        ctx.set_avoidance_output(result.desired_heading, result.speed_factor);

        std::this_thread::sleep_until(start + period);
    }
}
