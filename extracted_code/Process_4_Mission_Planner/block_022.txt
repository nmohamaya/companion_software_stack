=== Listing 6-5: Path Planner Thread with Dynamic Strategy Switching ===

// planning/path_planner_thread.cpp
#include "planning/planning_strategy.h"
#include "planning/astar_planner.h"
#include "planning/rrt_star_planner.h"
#include "planning/occupancy_grid.h"
#include "mission/mission_context.h"
#include "util/realtime.h"

void path_planner_thread(MissionContext& ctx,
                          std::shared_ptr<OccupancyGrid> grid,
                          std::atomic<bool>& running) {
    set_thread_params("PathPlan", 4, SCHED_FIFO, 40);

    // Default to A*, can switch at runtime
    std::unique_ptr<PlanningStrategy> planner = std::make_unique<AStarPlanner>();
    const auto period = std::chrono::milliseconds(500);  // 2 Hz

    while (running.load(std::memory_order_relaxed)) {
        auto start = std::chrono::steady_clock::now();

        if (!ctx.planning_active()) {
            std::this_thread::sleep_for(period);
            continue;
        }

        // Read current pose
        Eigen::Vector3d pos;
        Eigen::Quaterniond quat;
        Eigen::Vector3d vel;
        if (!ctx.get_pose(pos, quat, vel)) {
            spdlog::warn("PathPlanner: pose unavailable");
            std::this_thread::sleep_for(period);
            continue;
        }

        // Update occupancy grid with latest detections
        std::vector<DetectedObject> objects;
        ctx.object_reader().read(objects);
        grid->clear();
        grid->update_from_objects(objects, /*safety_margin=*/0.5);

        // Get next waypoint as goal
        auto goal = ctx.waypoint_manager().current_target();
        if (!goal) {
            std::this_thread::sleep_for(period);
            continue;
        }

        // Plan path
        auto result = planner->plan(pos, *goal, *grid, /*timeout_ms=*/200.0);
        if (result) {
            ctx.set_planned_path(std::move(result->waypoints));
            spdlog::debug("Path planned: {} pts, cost={:.1f}m, time={:.1f}ms",
                          result->waypoints.size(), result->total_cost,
                          result->compute_time_ms);
        } else {
            spdlog::warn("No feasible path to waypoint");
            // Could switch to RRT* for exploration
            if (planner->name() == "A*") {
                spdlog::info("Switching to RRT* planner");
                planner = std::make_unique<RRTStarPlanner>();
            }
        }

        std::this_thread::sleep_until(start + period);
    }
}
